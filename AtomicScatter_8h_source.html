133 KOKKOS_INLINE_FUNCTION void operator()(const size_t i_in) const {
134 using grid_value_t = typename decltype(args.grid)::non_const_value_type;
135
136 size_t p = i_in;
137 if constexpr (Policy::use_sorting)
138 p = args.permute(i_in);
139
140 CoordinateTransform<RealType, Dim> transform{args.origin, args.invdx, args.n_grid};
141 const RealType inv_hw = args.inv_hw;
142 const ValueType v = args.values(p);
143
144 // Per-dimension stencil base + per-stencil-point weights.
145 int base[Dim];
146 RealType kw[Dim][W];
147 for (unsigned d = 0; d < Dim; ++d) {
148 const RealType g = transform.toGridCoordinate(args.x(p)[d], d);
149 const int idx = transform.getStencilBase(g - RealType(0.5), W);
150 base[d] = idx - args.local_offset[d] + args.nghost;
151 const RealType g0 = (g - (RealType(idx) + RealType(0.5))) * inv_hw;
152 for (int k = 0; k < W; ++k) {
153 const RealType xi = g0 - RealType(k) * inv_hw;
154 if constexpr (Types::KernelType::has_width_template) {
155 kw[d][k] = args.kernel.template eval<W>(xi);
156 } else {
157 kw[d][k] = args.kernel(xi);
158 }
159 }
160 }
161
162 auto grid = args.grid;
163 if constexpr (Dim == 1) {
164 for (int i0 = 0; i0 < W; ++i0) {
165 auto& cell = grid(base[0] + i0);
166 Kokkos::atomic_add(&to_grid_value<grid_value_t>(cell),
167 static_cast<grid_value_t>(v * kw[0][i0]));
168 }
169 } else if constexpr (Dim == 2) {
170 for (int i1 = 0; i1 < W; ++i1) {
171 for (int i0 = 0; i0 < W; ++i0) {
172 auto& cell = grid(base[0] + i0, base[1] + i1);
173 const RealType w = kw[0][i0] * kw[1][i1];
174 Kokkos::atomic_add(&to_grid_value<grid_value_t>(cell),
175 static_cast<grid_value_t>(v * w));
176 }
177 }
178 } else if constexpr (Dim == 3) {
179 for (int i2 = 0; i2 < W; ++i2) {
180 for (int i1 = 0; i1 < W; ++i1) {
181 for (int i0 = 0; i0 < W; ++i0) {
182 auto& cell = grid(base[0] + i0, base[1] + i1, base[2] + i2);
183 const RealType w = kw[0][i0] * kw[1][i1] * kw[2][i2];
184 Kokkos::atomic_add(&to_grid_value<grid_value_t>(cell),
185 static_cast<grid_value_t>(v * w));
186 }
187 }
188 }
189 }

192 KOKKOS_INLINE_FUNCTION void operator()(const team_member& team) const {
193 using grid_value_t = typename decltype(args.grid)::non_const_value_type;
194
195 const int team_rank = team.team_rank();
196 const int team_size = team.team_size();
197
198 const size_t base_particle = size_t(team.league_rank()) * size_t(team_size);
199 size_t p_global = base_particle + size_t(team_rank);
200 if (p_global >= args.n_particles)
201 return;
202
203 if constexpr (Policy::use_sorting)
204 p_global = args.permute(p_global);
205
206 CoordinateTransform<RealType, Dim> transform{args.origin, args.invdx, args.n_grid};
207
208 // -------------------------
209 // scratch weights: 1D + computed padding (force odd stride for W=2..10)
210 // -------------------------
211 using ScratchSpace = typename team_member::scratch_memory_space;
212 using Unmanaged = Kokkos::MemoryUnmanaged;
213
214 auto scratch = team.team_scratch(0);
215
216 constexpr int KW_BASE = int(Dim) * int(W);
217 constexpr int KW_PAD = (KW_BASE & 1) ? 0 : 1; // make stride odd
218 constexpr int KW_STRIDE = KW_BASE + KW_PAD;
219
220 using Kw1D = Kokkos::View<RealType*, ScratchSpace, Unmanaged>;
221 Kw1D kw(scratch, team_size * KW_STRIDE);
222
223 RealType* kw_ptr = kw.data() + team_rank * KW_STRIDE;
224
225 const RealType inv_hw = args.inv_hw;
226
227 // helper: fill weights for a compile-time dimension D using g0 computed on-demand
228 auto fill_dim = [&](auto Dtag, RealType g0d) {
229 constexpr int D = decltype(Dtag)::value;
230 Kokkos::parallel_for(Kokkos::ThreadVectorRange(team, W), [&](const int i) {
231 const RealType xi = g0d - RealType(i) * inv_hw;
232 if constexpr (Types::KernelType::has_width_template)
233 kw_ptr[D * W + i] = args.kernel.template eval<W>(xi);
234 else
235 kw_ptr[D * W + i] = args.kernel(xi);
236 });
237 };
238
239 // -------------------------
240 // Dimension-specialized paths (LayoutLeft only)
241 // Compute (base, g0) only where they're needed and keep live ranges tight.
242 // -------------------------
243 if constexpr (Dim == 1) {
244 // compute dim0 and immediately fill weights
245 const RealType x0 = args.x(p_global)[0];
246 const RealType gp0 = transform.template toGridCoordinate<0>(x0);
247 const int idx0 = transform.template getStencilBase<W>(gp0 - RealType(0.5));
248 const int b0 = idx0 - args.local_offset[0] + args.nghost;
249 fill_dim(std::integral_constant<int, 0>{},
250 (gp0 - (RealType(idx0) + RealType(0.5))) * inv_hw);
251
252 // stencil
253 const ValueType my_val = args.values(p_global);
254 auto grid = args.grid;
255
256 Kokkos::parallel_for(Kokkos::ThreadVectorRange(team, W), [&](const int i0) {
257 const RealType w = kw_ptr[0 * W + i0];
258 auto& cell = grid(b0 + i0);
259 Kokkos::atomic_add(&to_grid_value<grid_value_t>(cell),
260 static_cast<grid_value_t>(my_val * w));
261 });
262
263 } else if constexpr (Dim == 2) {
264 // dim0
265 const RealType x0 = args.x(p_global)[0];
266 const RealType gp0 = transform.template toGridCoordinate<0>(x0);
267 const int idx0 = transform.template getStencilBase<W>(gp0 - RealType(0.5));
268 const int b0 = idx0 - args.local_offset[0] + args.nghost;
269 fill_dim(std::integral_constant<int, 0>{},
270 (gp0 - (RealType(idx0) + RealType(0.5))) * inv_hw);
271
272 // dim1
273 const RealType x1 = args.x(p_global)[1];
274 const RealType gp1 = transform.template toGridCoordinate<1>(x1);
275 const int idx1 = transform.template getStencilBase<W>(gp1 - RealType(0.5));
276 const int b1 = idx1 - args.local_offset[1] + args.nghost;
277 fill_dim(std::integral_constant<int, 1>{},
278 (gp1 - (RealType(idx1) + RealType(0.5))) * inv_hw);
279
280 // stencil: LayoutLeft => i0 fastest, then i1
281 const ValueType my_val = args.values(p_global);
282 auto grid = args.grid;
283 constexpr int STENCIL_SIZE = W * W;
284
285 Kokkos::parallel_for(Kokkos::ThreadVectorRange(team, STENCIL_SIZE),
286 [&](const int flat) {
287 const int i0 = flat % W;
288 const int i1 = flat / W;
289
290 const RealType w = kw_ptr[0 * W + i0] * kw_ptr[1 * W + i1];
291 auto& cell = grid(b0 + i0, b1 + i1);
292 Kokkos::atomic_add(&to_grid_value<grid_value_t>(cell),
293 static_cast<grid_value_t>(my_val * w));
294 });
295
296 } else if constexpr (Dim == 3) {
297 // dim0
298 const RealType x0 = args.x(p_global)[0];
299 const RealType gp0 = transform.template toGridCoordinate<0>(x0);
300 const int idx0 = transform.template getStencilBase<W>(gp0 - RealType(0.5));
301 const int b0 = idx0 - args.local_offset[0] + args.nghost;
302 fill_dim(std::integral_constant<int, 0>{},
303 (gp0 - (RealType(idx0) + RealType(0.5))) * inv_hw);
304
305 // dim1
306 const RealType x1 = args.x(p_global)[1];
307 const RealType gp1 = transform.template toGridCoordinate<1>(x1);
308 const int idx1 = transform.template getStencilBase<W>(gp1 - RealType(0.5));
309 const int b1 = idx1 - args.local_offset[1] + args.nghost;
310 fill_dim(std::integral_constant<int, 1>{},
311 (gp1 - (RealType(idx1) + RealType(0.5))) * inv_hw);
312
313 // dim2
314 const RealType x2 = args.x(p_global)[2];
315 const RealType gp2 = transform.template toGridCoordinate<2>(x2);
316 const int idx2 = transform.template getStencilBase<W>(gp2 - RealType(0.5));
317 const int b2 = idx2 - args.local_offset[2] + args.nghost;
318 fill_dim(std::integral_constant<int, 2>{},
319 (gp2 - (RealType(idx2) + RealType(0.5))) * inv_hw);
320
321 // stencil: LayoutLeft => i0 fastest, then i1, then i2
322 const ValueType my_val = args.values(p_global);
323 auto grid = args.grid;
324 constexpr int STENCIL_SIZE = W * W * W;
325
326 Kokkos::parallel_for(
327 Kokkos::ThreadVectorRange(team, STENCIL_SIZE), [&](const int flat) {
328 const int i0 = flat % W;
329 const int t = flat / W;
330 const int i1 = t % W;
331 const int i2 = t / W;
332
333 const RealType w =
334 kw_ptr[0 * W + i0] * kw_ptr[1 * W + i1] * kw_ptr[2 * W + i2];
335
336 auto& cell = grid(b0 + i0, b1 + i1, b2 + i2);
337 Kokkos::atomic_add(&to_grid_value<grid_value_t>(cell),
338 static_cast<grid_value_t>(my_val * w));
339 });
340 }
341 }

343 void run(size_t) {
344 if constexpr (W < 3) {
345 // NGP / CIC: stencils are too small for team-policy overhead
346 // to pay back. Flat parallel_for with global atomic_add wins
347 // (matches the original IPPL atomic-scatter pattern).
348 Kokkos::parallel_for(
349 "AtomicScatter::Simple",
350 Kokkos::RangePolicy<execution_space>(0, args.n_particles), *this);
351 return;
352 }
353
354 // Team-based path for W >= 3.
355 // GPU per-team thread limit: team_size * vector_length must fit in
356 // 1024 threads (HIP enforces strictly less-than 1024). Defensive
357 // clamp here protects against CSV/auto-tune values produced for a
358 // different (e.g. CUDA) device being replayed on HIP.
359 int team_size = args.particles_per_team;
360 if constexpr (!is_host_backend) {
361 const int hw_cap = is_hip ? 1023 : 1024;
362 const int max_team_size = std::max(1, hw_cap / vector_length);
363 if (team_size > max_team_size) team_size = max_team_size;
364 }
365 if (team_size < 1) team_size = 1;
366
367 const size_t scratch_size = compute_scratch_size<true>(team_size);
368 const size_t n_teams = (args.n_particles + team_size - 1) / team_size;
369
370 team_policy policy(n_teams, team_size, vector_length);
371 policy.set_scratch_size(0, Kokkos::PerTeam(scratch_size));
372
373 Kokkos::parallel_for("AtomicScatter", policy, *this);
374 }
375 };
