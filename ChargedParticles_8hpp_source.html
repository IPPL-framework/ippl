332 bool balance(size_type totalP, const unsigned int nstep) {
333 if (ippl::Comm->size() < 2) {
334 return false;
335 }
336 if (std::strcmp(TestName, "UniformPlasmaTest") == 0) {
337 return (nstep % loadbalancefreq_m == 0);
338 } else {
339 int local = 0;
340 std::vector<int> res(ippl::Comm->size());
341 double equalPart = (double)totalP / ippl::Comm->size();
342 double dev = std::abs((double)this->getLocalNum() - equalPart) / totalP;
343 if (dev > loadbalancethreshold_m) {
344 local = 1;
345 }
346 MPI_Allgather(&local, 1, MPI_INT, res.data(), 1, MPI_INT,
347 ippl::Comm->getCommunicator());
348
349 for (unsigned int i = 0; i < res.size(); i++) {
350 if (res[i] == 1) {
351 return true;
352 }
353 }
354 return false;
355 }

388 void scatterCIC(size_type totalP, unsigned int iteration, Vector_t<double, Dim>& hrField) {
389 Inform m("scatter ");
390
391 rho_m = 0.0;
392 scatter(q, rho_m, this->R);
393
394 static IpplTimings::TimerRef sumTimer = IpplTimings::getTimer("Check");
395 IpplTimings::startTimer(sumTimer);
396 double Q_grid = rho_m.sum();
397
398 size_type Total_particles = 0;
399 size_type local_particles = this->getLocalNum();
400
401 ippl::Comm->reduce(local_particles, Total_particles, 1, std::plus<size_type>());
402
403 double rel_error = std::fabs((Q_m - Q_grid) / Q_m);
404 m << "Rel. error in charge conservation = " << rel_error << endl;
405
406 if (ippl::Comm->rank() == 0) {
407 if (Total_particles != totalP || rel_error > 1e-10) {
408 m << "Time step: " << iteration << endl;
409 m << "Total particles in the sim. " << totalP << " "
410 << "after update: " << Total_particles << endl;
411 m << "Rel. error in charge conservation: " << rel_error << endl;
412 ippl::Comm->abort();
413 }
414 }
415
416 double cellVolume =
417 std::reduce(hrField.begin(), hrField.end(), 1., std::multiplies<double>());
418 rho_m = rho_m / cellVolume;
419
420 rhoNorm_m = norm(rho_m);
421 IpplTimings::stopTimer(sumTimer);
422
423 // dumpVTK(rho_m, nr_m[0], nr_m[1], nr_m[2], iteration, hrField[0], hrField[1], hrField[2]);
424
425 // rho = rho_e - rho_i (only if periodic BCs)
426 if (stype_m != "OPEN") {
427 double size = 1;
428 for (unsigned d = 0; d < Dim; d++) {
429 size *= rmax_m[d] - rmin_m[d];
430 }
431 rho_m = rho_m - (Q_m / size);
432 }

493 void initSolverWithParams(const ippl::ParameterList& sp) {
494 solver_m.template emplace<Solver>();
495 Solver& solver = std::get<Solver>(solver_m);
496
497 solver.mergeParameters(sp);
498
499 solver.setRhs(rho_m);
500
501 if constexpr (std::is_same_v<Solver, CGSolver_t<T, Dim>>) {
502 // The CG solver computes the potential directly and
503 // uses this to get the electric field
504 solver.setLhs(phi_m);
505 solver.setGradient(E_m);
506 } else {
507 // The periodic Poisson solver, Open boundaries solver,
508 // and the P3M solver compute the electric field directly
509 solver.setLhs(E_m);
510 }

581 void dumpData() {
582 auto Pview = P.getView();
583
584 double kinEnergy = 0.0;
585 double potEnergy = 0.0;
586
587 rho_m = dot(E_m, E_m);
588 potEnergy = 0.5 * hr_m[0] * hr_m[1] * hr_m[2] * rho_m.sum();
589
590 Kokkos::parallel_reduce(
591 "Particle Kinetic Energy", this->getLocalNum(),
592 KOKKOS_LAMBDA(const int i, double& valL) {
593 double myVal = dot(Pview(i), Pview(i)).apply();
594 valL += myVal;
595 },
596 Kokkos::Sum<double>(kinEnergy));
597
598 kinEnergy *= 0.5;
599 double gkinEnergy = 0.0;
600
601 ippl::Comm->reduce(kinEnergy, gkinEnergy, 1, std::plus<double>());
602
603 const int nghostE = E_m.getNghost();
604 auto Eview = E_m.getView();
605 Vector_t<T, Dim> normE;
606
607 using index_array_type = typename ippl::RangePolicy<Dim>::index_array_type;
608 for (unsigned d = 0; d < Dim; ++d) {
609 T temp = 0.0;
610 ippl::parallel_reduce(
611 "Vector E reduce", ippl::getRangePolicy(Eview, nghostE),
612 KOKKOS_LAMBDA(const index_array_type& args, T& valL) {
613 // ippl::apply accesses the view at the given indices and obtains a
614 // reference; see src/Expression/IpplOperations.h
615 T myVal = std::pow(ippl::apply(Eview, args)[d], 2);
616 valL += myVal;
617 },
618 Kokkos::Sum<T>(temp));
619 T globaltemp = 0.0;
620 ippl::Comm->reduce(temp, globaltemp, 1, std::plus<T>());
621 normE[d] = std::sqrt(globaltemp);
622 }
623
624 if (ippl::Comm->rank() == 0) {
625 std::stringstream fname;
626 fname << "data/ParticleField_";
627 fname << ippl::Comm->size();
628 fname << ".csv";
629
630 Inform csvout(NULL, fname.str().c_str(), Inform::APPEND);
631 csvout.precision(10);
632 csvout.setf(std::ios::scientific, std::ios::floatfield);
633
634 if (time_m == 0.0) {
635 csvout << "time, Potential energy, Kinetic energy, Total energy, Rho_norm2";
636 for (unsigned d = 0; d < Dim; d++) {
637 csvout << ", E" << static_cast<char>((Dim <= 3 ? 'x' : '1') + d) << "_norm2";
638 }
639 csvout << endl;
