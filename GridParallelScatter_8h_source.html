122 // binning flattens with dimension 2 fastest:
123 // bin = tx*(ny*nz) + ty*nz + tz -> decode in order z, y, x.
124 KOKKOS_INLINE_FUNCTION void decode_tile_base(const size_t tile_id_in, int& tx, int& ty,
125 int& tz) const {
126 size_t t = tile_id_in;
127 const size_t nt2 = static_cast<size_t>(args.num_tiles[2]);
128 const size_t nt1 = static_cast<size_t>(args.num_tiles[1]);
129 const int iz = static_cast<int>(t % nt2);
130 t /= nt2;
131 const int iy = static_cast<int>(t % nt1);
132 t /= nt1;
133 const int ix = static_cast<int>(t);
134 tx = ix * args.tile_size[0];
135 ty = iy * args.tile_size[1];
136 tz = iz * args.tile_size[2];
137 }

139 template <bool NeedsImag>
140 KOKKOS_FORCEINLINE_FUNCTION static void scatter_particle_fast(
141 const team_member& team, const int sx, const int sy, const int sz, const int pitch0,
142 const int pitch1, const RealType* __restrict__ kw, const RealType vr, const RealType vi,
143 RealType* __restrict__ local_r, RealType* __restrict__ local_i) {
144 const RealType* __restrict__ kwx = kw;
145 const RealType* __restrict__ kwy = kw + W;
146 const RealType* __restrict__ kwz = kw + 2 * W;
147
148 constexpr int plane = W * W;
149#if defined(__CUDA_ARCH__) || defined(__HIP_DEVICE_COMPILE__)
150#pragma unroll 1
151#endif
152 for (int idx = team.team_rank(); idx < stencil_total; idx += team.team_size()) {
153 const int zz = idx / plane;
154 const int rem = idx - zz * plane;
155 const int yy = rem / W;
156 const int xx = rem - yy * W;
157
158 const int hidx = (sx + xx) + (sy + yy) * pitch0 + (sz + zz) * pitch1;
159 const RealType w = kwx[xx] * kwy[yy] * kwz[zz];
160 local_r[hidx] += vr * w;
161 if constexpr (NeedsImag)
162 local_i[hidx] += vi * w;
163 }
164 }

327 const int rem = static_cast<int>(lid) - kz * pitch1;
328 const int jy = rem / pitch0;
329 const int ix = rem - jy * pitch0;

334 // are non-zero in finite precision, so the float `==` is safe.
335 if constexpr (needs_imag) {
336 if (rr == RealType(0) && local_i[lid] == RealType(0))

338 } else {
339 if (rr == RealType(0))
340 continue;
341 }
342
343 const int local_x = tile_base_x + ix - half_left;
344 const int local_y = tile_base_y + jy - half_left;
345 const int local_z = tile_base_z + kz - half_left;

347 if (local_x < -args.nghost || local_x >= args.n_grid_local[0] + args.nghost
348 || local_y < -args.nghost || local_y >= args.n_grid_local[1] + args.nghost
349 || local_z < -args.nghost || local_z >= args.n_grid_local[2] + args.nghost)
350 continue;
351
352 const int gx = local_x + args.nghost;
353 const int gy = local_y + args.nghost;
354 const int gz = local_z + args.nghost;
355
356 if constexpr (grid_complex) {
357 RealType* ptr = reinterpret_cast<RealType*>(&args.grid(gx, gy, gz));
358 Kokkos::atomic_add(&ptr[0], rr);
359 if constexpr (needs_imag)
360 Kokkos::atomic_add(&ptr[1], local_i[lid]);
361 } else {
362 Kokkos::atomic_add(&args.grid(gx, gy, gz), static_cast<grid_value_t>(rr));
363 }
364 }

367 template <bool NeedsImag>
368 static size_t compute_scratch_size(const Vector<int, Dim>& tile_size, int /*team_size*/,
369 int z_batches) {
370 const int batch_np = z_batches > 0 ? z_batches : 1;
371 const int hs0 = tile_size[0] + padded_extra;
372 const int hs1 = tile_size[1] + padded_extra;
373 const int hs2 = tile_size[2] + padded_extra;
374 const size_t total = static_cast<size_t>(hs0) * hs1 * hs2;
375
376 size_t s = 0;
377 s += scratch_real_view::shmem_size(total);
378 if constexpr (NeedsImag)
379 s += scratch_real_view::shmem_size(total);
380 s += scratch_real_view::shmem_size(static_cast<size_t>(batch_np) * ker_stride);
381 s += scratch_real_view::shmem_size(batch_np);
382 if constexpr (NeedsImag)
383 s += scratch_real_view::shmem_size(batch_np);
384 s += scratch_int_view::shmem_size(batch_np); // shifts_x
385 s += scratch_int_view::shmem_size(batch_np); // shifts_y
386 s += scratch_int_view::shmem_size(batch_np); // shifts_z
387 return s;
388 }

390 void run(size_t n_particles) {
391 using grid_value_t = typename decltype(args.grid)::non_const_value_type;
392 constexpr bool grid_complex = std::is_same_v<grid_value_t, Kokkos::complex<RealType>>;
393 constexpr bool value_complex = std::is_same_v<ValueType, Kokkos::complex<RealType>>;
394 constexpr bool needs_imag = grid_complex && value_complex;
395
396 size_t n_tiles = 1;
397 for (unsigned d = 0; d < Dim; ++d)
398 n_tiles *= static_cast<size_t>(args.num_tiles[d]);
399 if (n_tiles == 0 || n_particles == 0)
400 return;
401
402 const int hs0 = args.tile_size[0] + padded_extra;
403 const int hs1 = args.tile_size[1] + padded_extra;
404 const int hs2 = args.tile_size[2] + padded_extra;
405 total_ = static_cast<size_t>(hs0) * hs1 * hs2;
406
407 size_t league_size;
408 if constexpr (fixed_oversubscription) {
409 sub_teams_per_tile_ = 1;
410 league_size = n_tiles;
411 } else {
412 sub_teams_per_tile_ =
413 std::max<size_t>(1, static_cast<size_t>(args.oversubscription_factor));
414 league_size = n_tiles * sub_teams_per_tile_;
415 }
416
417 const size_t scratch =
418 compute_scratch_size<needs_imag>(args.tile_size, args.team_size, args.batch_np);
419
420 auto policy = team_policy(league_size, args.team_size, 1)
421 .set_scratch_size(0, Kokkos::PerTeam(scratch));
422 Kokkos::parallel_for("GridParallelScatterOutputDrivenBatched3D", policy, *this);
423 }
424 };

512 template <bool NeedsImag>
513 KOKKOS_FORCEINLINE_FUNCTION static void scatter_particle_fast(
514 const team_member& team, const int* __restrict__ shift,
515 const size_t* __restrict__ stride, const RealType* __restrict__ kw, const RealType vr,
516 const RealType vi, RealType* __restrict__ local_r, RealType* __restrict__ local_i) {
517 for (int idx = team.team_rank(); idx < stencil_total; idx += team.team_size()) {
518 int tmp = idx;
519 size_t hidx = 0;
520 RealType w = RealType(1);
521 for (unsigned d = 0; d < Dim; ++d) {
522 const int wi = tmp % W;
523 tmp /= W;
524 hidx += static_cast<size_t>(shift[d] + wi) * stride[d];
525 w *= kw[d * W + wi];
526 }
527 local_r[hidx] += vr * w;
528 if constexpr (NeedsImag)
529 local_i[hidx] += vi * w;
530 }

578
579 const auto tile_base = decode_tile_base(tile_id);
580 const size_t total = total_;
581 const int batch_np = args.batch_np;
582
583 auto scratch = team.team_scratch(0);
584
585 scratch_real_view local_r_v(scratch, total);
586 RealType* const local_r = local_r_v.data();
587
588 RealType* local_i = nullptr;
589 scratch_real_view local_i_v;
590 if constexpr (needs_imag) {
591 local_i_v = scratch_real_view(scratch, total);
592 local_i = local_i_v.data();
593 }
594
595 scratch_real_view kerevals_v(scratch, static_cast<size_t>(batch_np) * ker_stride);
596 RealType* const kerevals = kerevals_v.data();
597
598 scratch_real_view vals_r_v(scratch, batch_np);
599 RealType* const vals_r = vals_r_v.data();
600
601 RealType* vals_i = nullptr;
602 scratch_real_view vals_i_v;
603 if constexpr (needs_imag) {
604 vals_i_v = scratch_real_view(scratch, batch_np);
605 vals_i = vals_i_v.data();
606 }
607
608 scratch_int_view shifts_v(scratch, static_cast<size_t>(batch_np) * Dim);
609 int* const shifts = shifts_v.data();
610
611 for (size_t i = static_cast<size_t>(team.team_rank()); i < total;
612 i += static_cast<size_t>(team.team_size())) {
613 local_r[i] = RealType(0);
614 if constexpr (needs_imag)
615 local_i[i] = RealType(0);
616 }
617 team.team_barrier();
618
619 const CoordinateTransform<RealType, Dim> transform{args.origin, args.invdx,
620 args.n_grid};
621
622 for (size_t batch_begin = pstart; batch_begin < pend;
623 batch_begin += static_cast<size_t>(batch_np)) {
624 const int batch_size = static_cast<int>(
625 Kokkos::min(pend - batch_begin, static_cast<size_t>(batch_np)));
626
627 Kokkos::parallel_for(Kokkos::TeamThreadRange(team, batch_size), [&](const int bi) {
628 const size_t p = args.permute(batch_begin + static_cast<size_t>(bi));
629 RealType* const kw = kerevals + static_cast<size_t>(bi) * ker_stride;
630
631 for (unsigned d = 0; d < Dim; ++d) {
632 const RealType gp = transform.toGridCoordinate(args.x(p)[d], d);
633 const int idx0 = transform.getStencilBase(gp - RealType(0.5), W);
634 for (int wi = 0; wi < W; ++wi)
635 kw[d * W + wi] = args.kernel(
636 (gp - (RealType(idx0 + wi) + RealType(0.5))) * args.inv_hw);
637 shifts[static_cast<size_t>(bi) * Dim + d] =
638 idx0 - args.local_offset[d] + half_left - tile_base[d];
639 }
640
641 if constexpr (value_complex) {
642 const auto v = args.values(p);
643 vals_r[bi] = v.real();
644 if constexpr (needs_imag)
645 vals_i[bi] = v.imag();
646 } else {
647 vals_r[bi] = static_cast<RealType>(args.values(p));
648 }
649 });
650 team.team_barrier();
651
652 for (int bi = 0; bi < batch_size; ++bi) {
653 int shift[Dim];
654 for (unsigned d = 0; d < Dim; ++d)
655 shift[d] = shifts[static_cast<size_t>(bi) * Dim + d];
656 const RealType* const kw = kerevals + static_cast<size_t>(bi) * ker_stride;
657 const RealType vr = vals_r[bi];
658 const RealType vi = needs_imag ? vals_i[bi] : RealType(0);
659 scatter_particle_fast<needs_imag>(team, shift, stride, kw, vr, vi, local_r,
660 local_i);
661 team.team_barrier();
662 }
663 }
664
665 for (size_t lid = static_cast<size_t>(team.team_rank()); lid < total;
666 lid += static_cast<size_t>(team.team_size())) {
667 size_t tmp = lid;
668 int coord[Dim];
669 size_t sidx = 0;
670 for (unsigned d = 0; d < Dim; ++d) {
671 coord[d] = static_cast<int>(tmp % static_cast<size_t>(hs[d]));
672 tmp /= static_cast<size_t>(hs[d]);
673 sidx += static_cast<size_t>(coord[d]) * stride[d];
674 }
675
676 const RealType rr = local_r[sidx];
677 if constexpr (needs_imag) {
678 if (rr == RealType(0) && local_i[sidx] == RealType(0))
679 continue;
680 } else {
681 if (rr == RealType(0))
682 continue;
683 }
684
685 Kokkos::Array<int, Dim> gc{};
686 bool in_bounds = true;
687 for (unsigned d = 0; d < Dim; ++d) {
688 const int local = tile_base[d] + coord[d] - half_left;
689 if (local < -args.nghost || local >= args.n_grid_local[d] + args.nghost) {
690 in_bounds = false;
691 break;
692 }
693 gc[d] = local + args.nghost;
694 }
695 if (!in_bounds)
696 continue;
697
698 [&]<std::size_t... Is>(std::index_sequence<Is...>) {
699 if constexpr (grid_complex) {
700 RealType* ptr = reinterpret_cast<RealType*>(&args.grid(gc[Is]...));
701 Kokkos::atomic_add(&ptr[0], rr);
702 if constexpr (needs_imag)
703 Kokkos::atomic_add(&ptr[1], local_i[sidx]);
704 } else {
705 Kokkos::atomic_add(&args.grid(gc[Is]...), static_cast<grid_value_t>(rr));
706 }
707 }(std::make_index_sequence<Dim>{});
708 }
709 }

711 template <bool NeedsImag>
712 static size_t compute_scratch_size(const Vector<int, Dim>& tile_size, int /*team_size*/,
713 int z_batches) {
714 const int batch_np = z_batches > 0 ? z_batches : 1;
715 size_t total = 1;
716 for (unsigned d = 0; d < Dim; ++d)
717 total *= static_cast<size_t>(tile_size[d] + padded_extra);
718
719 size_t s = 0;
720 s += scratch_real_view::shmem_size(total);
721 if constexpr (NeedsImag)
722 s += scratch_real_view::shmem_size(total);
723 s += scratch_real_view::shmem_size(static_cast<size_t>(batch_np) * ker_stride);
724 s += scratch_real_view::shmem_size(batch_np);
725 if constexpr (NeedsImag)
726 s += scratch_real_view::shmem_size(batch_np);
727 s += scratch_int_view::shmem_size(static_cast<size_t>(batch_np) * Dim);
728 return s;
729 }

731 void run(size_t n_particles) {
732 using grid_value_t = typename decltype(args.grid)::non_const_value_type;
733 constexpr bool grid_complex = std::is_same_v<grid_value_t, Kokkos::complex<RealType>>;
734 constexpr bool value_complex = std::is_same_v<ValueType, Kokkos::complex<RealType>>;
735 constexpr bool needs_imag = grid_complex && value_complex;
736
737 size_t n_tiles = 1;
738 for (unsigned d = 0; d < Dim; ++d)
739 n_tiles *= static_cast<size_t>(args.num_tiles[d]);
740 if (n_tiles == 0 || n_particles == 0)
741 return;
742
743 total_ = 1;
744 for (unsigned d = 0; d < Dim; ++d)
745 total_ *= static_cast<size_t>(args.tile_size[d] + padded_extra);
746
747 size_t league_size;
748 if constexpr (fixed_oversubscription) {
749 sub_teams_per_tile_ = 1;
750 league_size = n_tiles;
751 } else {
752 sub_teams_per_tile_ =
753 std::max<size_t>(1, static_cast<size_t>(args.oversubscription_factor));
754 league_size = n_tiles * sub_teams_per_tile_;
755 }
756
757 const size_t scratch =
758 compute_scratch_size<needs_imag>(args.tile_size, args.team_size, args.batch_np);
759 auto policy = team_policy(league_size, args.team_size, 1)
760 .set_scratch_size(0, Kokkos::PerTeam(scratch));
761 Kokkos::parallel_for("GridParallelScatterOutputDrivenBatchedND", policy, *this);
762 }
