1//
2// Class ParticleSpatialLayout
3// Particle layout based on spatial decomposition.
4//
5// This is a specialized version of ParticleLayout, which places particles
6// on processors based on their spatial location relative to a fixed grid.
7// In particular, this can maintain particles on processors based on a
8// specified FieldLayout or RegionLayout, so that particles are always on
9// the same node as the node containing the Field region to which they are
10// local. This may also be used if there is no associated Field at all,
11// in which case a grid is selected based on an even distribution of
12// particles among processors.
13//
14// After each 'time step' in a calculation, which is defined as a period
15// in which the particle positions may change enough to affect the global
16// layout, the user must call the 'update' routine, which will move
17// particles between processors, etc. After the Nth call to update, a
18// load balancing routine will be called instead. The user may set the
19// frequency of load balancing (N), or may supply a function to
20// determine if load balancing should be done or not.
21//

62 std::fill(nRecvs_m.begin(), nRecvs_m.end(), 0);

66 if (rank == Comm->rank())

75 template <typename T, unsigned Dim, class Mesh, typename... Properties>
76 void ParticleSpatialLayout<T, Dim, Mesh, Properties...>::countExchangeP2P() {
77 const int myRank = Comm->rank();
78 const int tag = Comm->next_tag(mpi::tag::P_SPATIAL_LAYOUT, mpi::tag::P_LAYOUT_CYCLE);
79
80 // Zero the receive-count buffer on the device
81 Kokkos::deep_copy(position_execution_space{}, recvCounts_d_, 0);
82 Kokkos::fence();
83
84 // Device pointer to send-count array
85 int* d_sendCounts = rankSendCount_d_.data();
86 int* d_recvCounts = recvCounts_d_.data();
87
88 std::vector<MPI_Request> reqs;
89 reqs.reserve(2 * std::max(0, nRanks_ - 1));
90
91 // Post receives from all ranks (except self)
92 for (int r = 0; r < nRanks_; ++r) {
93 if (r == myRank)
94 continue;
95 MPI_Irecv(d_recvCounts + r, 1, MPI_INT, r, tag, Comm->getCommunicator(),
96 &reqs.emplace_back());
97 }
98
99 for (int r = 0; r < nRanks_; ++r) {
100 if (r == myRank)
101 continue;
102 MPI_Isend(d_sendCounts + r, 1, MPI_INT, r, tag, Comm->getCommunicator(),
103 &reqs.emplace_back());
104 }
105
106 MPI_Waitall(static_cast<int>(reqs.size()), reqs.data(), MPI_STATUSES_IGNORE);
107 }

115 template <typename T, unsigned Dim, class Mesh, typename... Properties>
116 template <class ParticleContainer>
117 void ParticleSpatialLayout<T, Dim, Mesh, Properties...>::update(ParticleContainer& pc) {
118 /* Apply Boundary Conditions */
119 static IpplTimings::TimerRef ParticleBCTimer = IpplTimings::getTimer("particleBC");
120 IpplTimings::startTimer(ParticleBCTimer);
121 this->applyBC(pc.R, rlayout_m->getDomain());
122 IpplTimings::stopTimer(ParticleBCTimer);
123
124 /* Update Timer for the rest of the function */
125 static IpplTimings::TimerRef ParticleUpdateTimer = IpplTimings::getTimer("updateParticle");
126 IpplTimings::startTimer(ParticleUpdateTimer);
127
128 if (nRanks_ < 2) {
129 IpplTimings::stopTimer(ParticleUpdateTimer);
130 return;
131 }
132
133 ensureNeighborsCached();
134
135 /* particle MPI exchange:
136 * 1. figure out which particles need to go where -> locateParticles(...)
137 * 2. fill send buffer and send particles
138 * 3. delete invalidated particles
139 * 4. receive particles
140 */
141
142 // 1. figure out which particles need to go where -> locateParticles(...) ============= //
143
144 static IpplTimings::TimerRef locateTimer = IpplTimings::getTimer("locateParticles");
145 IpplTimings::startTimer(locateTimer);
146
147 const size_type nInvalid = locateParticlesPacked(pc);
148
149 // Copy metadata to host
150 size_type nDest = 0;
151 Kokkos::deep_copy(position_execution_space{}, nDest, nDest_d_);
152
153 // destRanks prefix
154 if (nDest > 0) {
155 Kokkos::deep_copy(
156 position_execution_space{},
157 Kokkos::subview(destRanks_h_, std::make_pair(size_t(0), size_t(nDest))),
158 Kokkos::subview(destRanks_d_, std::make_pair(size_t(0), size_t(nDest))));
159 }
160
161 // counts + offsets
162 Kokkos::deep_copy(position_execution_space{}, rankSendCount_h_, rankSendCount_d_);
163 Kokkos::deep_copy(position_execution_space{}, sendOffsets_h_, sendOffsets_d_);
164
165 // Build host destination list without allocation
166 destinationRanks_host_.clear();
167 for (size_t i = 0; i < (size_t)nDest; ++i)
168 destinationRanks_host_.push_back(destRanks_h_(i));
169
170 IpplTimings::stopTimer(locateTimer);
171
172 // 2. fill send buffer and send particles =============================================== //
173
174 // 2.1 Count Exchange
175
176 static IpplTimings::TimerRef preprocTimer = IpplTimings::getTimer("sendPreprocess");
177 IpplTimings::startTimer(preprocTimer);
178
179 if (countExchangeMode_ == CountExchange::RMA) {
180 countExchangeRMA();
181 } else if (countExchangeMode_ == CountExchange::P2P) {
182 countExchangeP2P();
183 } else {
184 countExchangeAlltoall();
185 }
186
187 IpplTimings::stopTimer(preprocTimer);
188
189 // 2.2 Particle Sends
190
191 static IpplTimings::TimerRef sendTimer = IpplTimings::getTimer("particleSend");
192 IpplTimings::startTimer(sendTimer);
193
194 std::vector<MPI_Request> requests;
195 requests.reserve(destinationRanks_host_.size());
196
197 int tag = Comm->next_tag(mpi::tag::P_SPATIAL_LAYOUT, mpi::tag::P_LAYOUT_CYCLE);
198
199 for (int rank : destinationRanks_host_) {
200 if (rank == Comm->rank())
201 continue;
202 const size_type count = static_cast<size_type>(rankSendCount_h_(rank));
203 if (count == 0)
204 continue;
205 const size_type begin = static_cast<size_type>(sendOffsets_h_(rank));
206 auto ids_sub =
207 Kokkos::subview(sendIds_d_, std::make_pair((size_t)begin, (size_t)(begin + count)));
208 requests.push_back(pc.sendToRank(rank, tag, ids_sub));
209 }
210
211 IpplTimings::stopTimer(sendTimer);
212
213 // 2.3 Post receives
214
215 static IpplTimings::TimerRef recvTimer = IpplTimings::getTimer("particleRecv");
216 IpplTimings::startTimer(recvTimer);
217
218 std::vector<std::pair<int, size_type>> recvList;
219 size_type totalRecvs = 0;
220
221 if (countExchangeMode_ == CountExchange::RMA) {
222 for (int rank = 0; rank < nRanks_; ++rank) {
223 if (nRecvs_m[rank] > 0) {
224 recvList.push_back({rank, nRecvs_m[rank]});
225 totalRecvs += nRecvs_m[rank];
226 }
227 }
228 } else {
229 auto recvCounts_h =
230 Kokkos::create_mirror_view_and_copy(Kokkos::HostSpace(), recvCounts_d_);
231 for (int rank = 0; rank < nRanks_; ++rank) {
232 if (recvCounts_h(rank) > 0) {
233 recvList.push_back({rank, recvCounts_h(rank)});
234 totalRecvs += static_cast<size_type>(recvCounts_h(rank));
235 }
236 }
237 }
238
239 std::vector<MPI_Request> recvRequests(recvList.size(), MPI_REQUEST_NULL);
240 std::vector<std::function<void(size_type)>> finalizers(recvList.size());
241
242 for (size_t i = 0; i < recvList.size(); ++i) {
243 auto [rank, count] = recvList[i];
244 auto [req, fin] = pc.postRecvFromRank(rank, tag, count);
245 recvRequests[i] = req;
246 finalizers[i] = std::move(fin);
247 }
248
249 IpplTimings::stopTimer(recvTimer);
250
251 // 3. Internal destruction of invalid particles ======================================= //
252
253 static IpplTimings::TimerRef destroyTimer = IpplTimings::getTimer("particleDestroy");
254 IpplTimings::startTimer(destroyTimer);
255
256 // locateParticlesPacked already wrote the per-particle "is leaving"
257 // mask into leaving_d_. Reuse it instead of re-running the full
258 // region search inside the destroy predicate.
259 auto leaving = leaving_d_;
260 pc.template internalDestroy<position_memory_space, position_execution_space>(
261 KOKKOS_LAMBDA(size_t i) { return leaving(i); }, nInvalid);
262 Kokkos::fence();
263 const size_type localAfterDestroy = pc.getLocalNum();
264
265 IpplTimings::stopTimer(destroyTimer);
266
267 static IpplTimings::TimerRef waitTimer = IpplTimings::getTimer("particleWait");
268 IpplTimings::startTimer(waitTimer);
269
270 requests.insert(requests.end(), recvRequests.begin(), recvRequests.end());
271 if (!requests.empty()) {
272 MPI_Waitall(static_cast<int>(requests.size()), requests.data(), MPI_STATUSES_IGNORE);
273 }
274
275 IpplTimings::stopTimer(waitTimer);
276
277 static IpplTimings::TimerRef freeBufferTimer = IpplTimings::getTimer("particleFreeBuffers");
278 IpplTimings::startTimer(freeBufferTimer);
279
280 Comm->freeAllBuffers();
281
282 IpplTimings::stopTimer(freeBufferTimer);
283
284 // 5. Deserialize
285 static IpplTimings::TimerRef deserializeTimer =
286 IpplTimings::getTimer("particleDeserialize");
287 IpplTimings::startTimer(deserializeTimer);
288
289 static IpplTimings::TimerRef deserializeResizeTimer =
290 IpplTimings::getTimer("particleDeserResize");
291 IpplTimings::startTimer(deserializeResizeTimer);
292
293 if (totalRecvs > 0) {
294 const size_type receiveCapacity = localAfterDestroy + totalRecvs;
295 pc.forAllAttributes([&]<typename Attribute>(Attribute*& attribute) {
296 attribute->reserve(receiveCapacity);
297 });
298 }
299
300 IpplTimings::stopTimer(deserializeResizeTimer);
301
302 static IpplTimings::TimerRef deserializeCopyTimer =
303 IpplTimings::getTimer("particleDeserCopy");
304 IpplTimings::startTimer(deserializeCopyTimer);
305
306 for (auto& finalize : finalizers) {
307 finalize(pc.getLocalNum());
308 }
309
310 IpplTimings::stopTimer(deserializeCopyTimer);
311 IpplTimings::stopTimer(deserializeTimer);
312
313 IpplTimings::stopTimer(ParticleUpdateTimer);
314 }

347 template <typename T, unsigned Dim, class Mesh, typename... Properties>
348 template <typename ParticleContainer>
349 size_t ParticleSpatialLayout<T, Dim, Mesh, Properties...>::locateParticlesPacked(
350 const ParticleContainer& pc) {
351 const int nRanks = Comm->size();
352 const size_type myRank = Comm->rank();
353
354 ensureNeighborsCached();
355
356 auto positions = pc.R.getView();
357 region_view_type Regions = rlayout_m->getdLocalRegions();
358 const auto is = std::make_index_sequence<Dim>{};
359
360 using exec_space = position_execution_space;
361 using policy_type = Kokkos::RangePolicy<size_t, exec_space>;
362
363 // Reset small device buffers
364 Kokkos::deep_copy(rankSendCount_d_, size_type(0));
365 Kokkos::deep_copy(cursor_d_, size_type(0));
366 Kokkos::deep_copy(nDest_d_, size_type(0));
367
368 const size_type neighbors_used = neighbors_used_;
369 auto& neighbours_d = neighbors_d_;
370
371 // Destination rank computation (no per-particle storage)
372 auto destRankOf = KOKKOS_LAMBDA(const size_t i)->size_type {
373 if (positionInRegion(is, positions(i), Regions(myRank)))
374 return myRank;
375
376 for (int j = 0; j < static_cast<int>(neighbors_used); ++j) {
377 const int r = neighbours_d(j);
378 if (positionInRegion(is, positions(i), Regions(r)))
379 return r;
380 }
381
382 // slow-path: global scan
383 for (int r = 0; r < static_cast<int>(Regions.extent(0)); ++r) {
384 if (positionInRegion(is, positions(i), Regions(r)))
385 return r;
386 }
387
388 // Inclusive fallback: catches particles sitting exactly on a region
389 // lower boundary that the strict > check above missed (e.g. (0,0,0))
390 for (int r = 0; r < static_cast<int>(Regions.extent(0)); ++r) {
391 if (positionInRegionInclusive(is, positions(i), Regions(r)))
392 return r;
393 }
394 return myRank; // truly outside all regions - applyBC should have prevented this
395 };
396
397 // Make sure the leaving-mask buffer is large enough; it's reused across
398 // updates to avoid reallocation.
399 ensureLeavingCapacity(pc.getLocalNum());
400 auto& leaving_d = leaving_d_;
401
402 // Pass 1: compute send counts + nInvalid + per-particle "is leaving"
403 size_type nInvalid = 0;
404 auto& rankSendCount_d = rankSendCount_d_;
405 Kokkos::parallel_reduce(
406 "PSL::packed_count", policy_type(0, pc.getLocalNum()),
407 KOKKOS_LAMBDA(const size_t i, size_type& inval) {
408 const size_type dest = destRankOf(i);
409 const bool leaves = (dest != myRank);
410 leaving_d(i) = leaves;
411 inval += leaves;
412 if (leaves)
413 Kokkos::atomic_fetch_add(&rankSendCount_d(dest), size_type(1));
414 },
415 nInvalid);
416 Kokkos::fence();
417
418 // Ensure sendIds capacity after we know nInvalid
419 ensureSendCapacity(nInvalid);
420
421 // Exclusive scan to offsets (length nRanks+1)
422 auto& sendOffsets_d = sendOffsets_d_;
423 Kokkos::parallel_scan(
424 "PSL::packed_offsets", policy_type(0, (size_t)nRanks + 1),
425 KOKKOS_LAMBDA(const size_t r, size_type& upd, const bool final) {
426 if (final)
427 sendOffsets_d(r) = upd;
428 if (r < (size_t)nRanks)
429 upd += rankSendCount_d(r);
430 });
431 Kokkos::fence();
432
433 // Pass 2: fill packed send IDs into sendIds_d_ (prefix [0, nInvalid))
434 auto& cursor_d = cursor_d_;
435 auto& sendIds_d = sendIds_d_;
436 Kokkos::parallel_for(
437 "PSL::packed_fill", policy_type(0, pc.getLocalNum()), KOKKOS_LAMBDA(const size_t i) {
438 const size_type dest = destRankOf(i);
439 if (dest == myRank)
440 return;
441
442 const size_type pos = Kokkos::atomic_fetch_add(&cursor_d(dest), size_type(1));
443 const size_type base = sendOffsets_d(dest);
444
445 sendIds_d(base + pos) = static_cast<typename hash_type::non_const_value_type>(i);
446 });
447 Kokkos::fence();
448
449 // Build destination rank list on device (compact), store length in nDest_d_
450 auto& destRanks_d = destRanks_d_;
451 auto& nDest_d = nDest_d_;
452 Kokkos::parallel_for(
453 "PSL::packed_destRanks", policy_type(0, (size_t)nRanks), KOKKOS_LAMBDA(const size_t r) {
454 if ((size_type)r == myRank)
455 return;
456 if (rankSendCount_d(r) > 0) {
457 const size_type idx = Kokkos::atomic_fetch_add(&nDest_d(), size_type(1));
458 destRanks_d(idx) = static_cast<int>(r);
459 }
460 });
461 Kokkos::fence();
462
463 return nInvalid;
464 }

513 template <typename T, unsigned Dim, class Mesh, typename... Properties>
514 void ParticleSpatialLayout<T, Dim, Mesh, Properties...>::ensureNeighborsCached() {
515 if (!neighbors_dirty_)
516 return;
517
518 const neighbor_list& neighbors = flayout_m.getNeighbors();
519 const size_type neighborSize = getNeighborSize(neighbors);
520 neighbors_used_ = neighborSize;
521
522 if (neighborSize > neighbors_capacity_) {
523 neighbors_capacity_ = neighborSize;
524 Kokkos::realloc(neighbors_d_, neighbors_capacity_);
525 }
526
527 neighbors_host_.clear();
528 neighbors_host_.reserve(neighborSize);
529
530 auto neighbors_h = Kokkos::create_mirror_view(neighbors_d_);
531 size_t k = 0;
532 for (const auto& comp : neighbors) {
533 for (size_t j = 0; j < comp.size(); ++j) {
534 neighbors_h(k) = comp[j];
535 neighbors_host_.push_back(comp[j]);
536 ++k;
537 }
538 }
539
540 Kokkos::deep_copy(
541 Kokkos::subview(neighbors_d_, std::make_pair(size_t(0), size_t(neighborSize))),
542 Kokkos::subview(neighbors_h, std::make_pair(size_t(0), size_t(neighborSize))));
543
544 neighbors_dirty_ = false;
545 }
