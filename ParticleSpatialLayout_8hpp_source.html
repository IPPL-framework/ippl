1//
2// Class ParticleSpatialLayout
3// Particle layout based on spatial decomposition.
4//
5// This is a specialized version of ParticleLayout, which places particles
6// on processors based on their spatial location relative to a fixed grid.
7// In particular, this can maintain particles on processors based on a
8// specified FieldLayout or RegionLayout, so that particles are always on
9// the same node as the node containing the Field region to which they are
10// local. This may also be used if there is no associated Field at all,
11// in which case a grid is selected based on an even distribution of
12// particles among processors.
13//
14// After each 'time step' in a calculation, which is defined as a period
15// in which the particle positions may change enough to affect the global
16// layout, the user must call the 'update' routine, which will move
17// particles between processors, etc. After the Nth call to update, a
18// load balancing routine will be called instead. The user may set the
19// frequency of load balancing (N), or may supply a function to
20// determine if load balancing should be done or not.
21//

44 template <typename T, unsigned Dim, class Mesh, typename... Properties>
45 template <class ParticleContainer>
46 void ParticleSpatialLayout<T, Dim, Mesh, Properties...>::update(ParticleContainer& pc) {
47 static IpplTimings::TimerRef ParticleBCTimer = IpplTimings::getTimer("particleBC");
48 IpplTimings::startTimer(ParticleBCTimer);
49 this->applyBC(pc.R, rlayout_m.getDomain());
50 IpplTimings::stopTimer(ParticleBCTimer);
51
52 static IpplTimings::TimerRef ParticleUpdateTimer = IpplTimings::getTimer("updateParticle");
53 IpplTimings::startTimer(ParticleUpdateTimer);
54 int nRanks = Comm->size();
55
56 if (nRanks < 2) {
57 return;
58 }
59
60 /* particle MPI exchange:
61 * 1. figure out which particles need to go where
62 * 2. fill send buffer and send particles
63 * 3. delete invalidated particles
64 * 4. receive particles
65 */
66
67 static IpplTimings::TimerRef locateTimer = IpplTimings::getTimer("locateParticles");
68 IpplTimings::startTimer(locateTimer);
69 size_type localnum = pc.getLocalNum();
70
71 // 1st step
72
73 /* the values specify the rank where
74 * the particle with that index should go
75 */
76 locate_type ranks("MPI ranks", localnum);
77
78 /* 0 --> particle valid
79 * 1 --> particle invalid
80 */
81 bool_type invalid("invalid", localnum);
82
83 size_type invalidCount = locateParticles(pc, ranks, invalid);
84 IpplTimings::stopTimer(locateTimer);
85
86 // 2nd step
87
88 // figure out how many receives
89 static IpplTimings::TimerRef preprocTimer = IpplTimings::getTimer("sendPreprocess");
90 IpplTimings::startTimer(preprocTimer);
91 mpi::rma::Window<mpi::rma::Active> window;
92 std::vector<size_type> nRecvs(nRanks, 0);
93 window.create(*Comm, nRecvs.begin(), nRecvs.end());
94
95 std::vector<size_type> nSends(nRanks, 0);
96
97 window.fence(0);
98
99 for (int rank = 0; rank < nRanks; ++rank) {
100 if (rank == Comm->rank()) {
101 // we do not need to send to ourselves
102 continue;
103 }
104 nSends[rank] = numberOfSends(rank, ranks);
105 window.put<size_type>(nSends.data() + rank, rank, Comm->rank());
106 }
107 window.fence(0);
108 IpplTimings::stopTimer(preprocTimer);
109
110 static IpplTimings::TimerRef sendTimer = IpplTimings::getTimer("particleSend");
111 IpplTimings::startTimer(sendTimer);
112 // send
113 std::vector<MPI_Request> requests(0);
114
115 int tag = Comm->next_tag(mpi::tag::P_SPATIAL_LAYOUT, mpi::tag::P_LAYOUT_CYCLE);
116
117 for (int rank = 0; rank < nRanks; ++rank) {
118 if (nSends[rank] > 0) {
119 hash_type hash("hash", nSends[rank]);
120 fillHash(rank, ranks, hash);
121
122 pc.sendToRank(rank, tag, requests, hash);
123 }
124 }
125 IpplTimings::stopTimer(sendTimer);
126
127 // 3rd step
128 static IpplTimings::TimerRef destroyTimer = IpplTimings::getTimer("particleDestroy");
129 IpplTimings::startTimer(destroyTimer);
130
131 pc.internalDestroy(invalid, invalidCount);
132 Kokkos::fence();
133
134 IpplTimings::stopTimer(destroyTimer);
135 static IpplTimings::TimerRef recvTimer = IpplTimings::getTimer("particleRecv");
136 IpplTimings::startTimer(recvTimer);
137 // 4th step
138
139 for (int rank = 0; rank < nRanks; ++rank) {
140 if (nRecvs[rank] > 0) {
141 pc.recvFromRank(rank, tag, nRecvs[rank]);
142 }
143 }
144 IpplTimings::stopTimer(recvTimer);
145
146 IpplTimings::startTimer(sendTimer);
147
148 if (requests.size() > 0) {
149 MPI_Waitall(requests.size(), requests.data(), MPI_STATUSES_IGNORE);
150 }
151 Comm->freeAllBuffers();
152
153 IpplTimings::stopTimer(sendTimer);
154
155 IpplTimings::stopTimer(ParticleUpdateTimer);
156 }

166 template <typename T, unsigned Dim, class Mesh, typename... Properties>
167 template <typename ParticleContainer>
168 detail::size_type ParticleSpatialLayout<T, Dim, Mesh, Properties...>::locateParticles(
169 const ParticleContainer& pc, locate_type& ranks, bool_type& invalid) const {
170 auto& positions = pc.R.getView();
171 typename RegionLayout_t::view_type Regions = rlayout_m.getdLocalRegions();
172
173 using mdrange_type = Kokkos::MDRangePolicy<Kokkos::Rank<2>, position_execution_space>;
174
175 int myRank = Comm->rank();
176
177 const auto is = std::make_index_sequence<Dim>{};
178
179 size_type invalidCount = 0;
180 Kokkos::parallel_reduce(
181 "ParticleSpatialLayout::locateParticles()",
182 mdrange_type({0, 0}, {ranks.extent(0), Regions.extent(0)}),
183 KOKKOS_LAMBDA(const size_t i, const size_type j, size_type& count) {
184 bool xyz_bool = positionInRegion(is, positions(i), Regions(j));
185 if (xyz_bool) {
186 ranks(i) = j;
187 invalid(i) = (myRank != ranks(i));
188 count += invalid(i);
189 }
190 },
191 Kokkos::Sum<size_type>(invalidCount));
192 Kokkos::fence();
193
194 return invalidCount;
195 }

197 template <typename T, unsigned Dim, class Mesh, typename... Properties>
198 void ParticleSpatialLayout<T, Dim, Mesh, Properties...>::fillHash(int rank,
199 const locate_type& ranks,
200 hash_type& hash) {
201 /* Compute the prefix sum and fill the hash
202 */
203 using policy_type = Kokkos::RangePolicy<position_execution_space>;
204 Kokkos::parallel_scan(
205 "ParticleSpatialLayout::fillHash()", policy_type(0, ranks.extent(0)),
206 KOKKOS_LAMBDA(const size_t i, int& idx, const bool final) {
207 if (final) {
208 if (rank == ranks(i)) {
209 hash(idx) = i;
210 }
211 }
212
213 if (rank == ranks(i)) {
214 idx += 1;
215 }
216 });
217 Kokkos::fence();
218 }
