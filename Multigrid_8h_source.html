63 template <typename BCType>
64 Level(std::shared_ptr<mesh_type> m, std::shared_ptr<layout_type> l, BCType& bcs)
65 : mesh_ptr(m)
66 , layout_ptr(l)

90 template <typename LevelType>
91 double compute_diag(const LevelType& lev) {
92 double diag = 0.0;
93
94 // We iterate up to LevelType::Dim.
95 // The compiler knows Dim at compile time, so it can heavily
96 // optimize or completely "unroll" this loop.
97 for (unsigned d = 0; d < LevelType::Dim; ++d) {
98 double ihxd2 = 1.0 / (lev.hx[d] * lev.hx[d]);
99 diag += 2.0 * ihxd2;
100 }
101
102 return diag;
103 }

172 void operator()(Field& b, Field& result) override {
173 IpplTimings::TimerRef mg = IpplTimings::getTimer("MG-PRECOND");
174 IpplTimings::startTimer(mg);
175
176 L_[0].f = b.deepCopy();
177
178 // Remove Volume average if periodic
179 if (is_all_periodic_) {
180 auto avg = L_[0].f.getVolumeAverage();
181 L_[0].f = L_[0].f - avg;
182 if (communication_)
183 L_[0].f.fillHalo();
184 }
185
186 for (size_t level = 0; level < L_.size(); ++level)
187 L_[level].u = 0.0;
188 vcycle(0);
189
190 // Remove Volume average if periodic
191 if (is_all_periodic_) {
192 auto avg = L_[0].u.getVolumeAverage();
193 L_[0].u = L_[0].u - avg;
194 if (communication_)
195 L_[0].u.fillHalo();
196 }
197
198 // Write the result into the caller-provided buffer
199 Kokkos::deep_copy(result.getView(), L_[0].u.getView());
200
201 IpplTimings::stopTimer(mg);
202 }

212 void init_fields(Field& b) override {
213 IpplTimings::TimerRef init_fields = IpplTimings::getTimer("init_fields");
214 IpplTimings::startTimer(init_fields);
215
216 auto& fine_mesh = b.get_mesh();
217 auto& fine_layout = b.getLayout();
218 auto fine_domain = fine_layout.getDomain();
219 std::array<bool, Dim> decomp = fine_layout.isParallel();
220
221 auto bcs = b.getFieldBC();
222
223 // 1. Calculate number of levels
224 const auto& localFine = fine_layout.getLocalNDIndex(); // this rank's slab
225 int min_local = Kokkos::Experimental::finite_max<int>::value;
226
227 for (unsigned d = 0; d < Dim; ++d) {
228 // For serial dims, every rank's local extent == global extent, so this naturally
229 // covers them too. We do not have to check if dim is parallel or not.
230 min_local = Kokkos::min(min_local, static_cast<int>(localFine[d].length()));
231 }
232
233 // reduce over all ranks to find smallest dimensional extent
234 int min_local_global = min_local;
235 ippl::Comm->allreduce(min_local, min_local_global, 1, std::less<int>{});
236
237 int min_cells_on_coarsest = min_cells_per_rank_per_dim_;
238 int nlevels = 1;
239 int at_level = min_local_global;
240 while (at_level / 2 >= min_cells_on_coarsest) {
241 at_level /= 2;
242 ++nlevels;
243 }
244
245 // reserve full vector to avoid implicit copying when adding new elements
246 L_.clear();
247 L_.reserve(nlevels);
248
249 // 2. Build the hierarchy (Meshes, Layouts, and Levels)
250 for (int ell = 0; ell < nlevels; ++ell) {
251 // Create SubFieldLayout using strided index over the original mesh
252 ippl::NDIndex<Dim> sub_domain;
253 ippl::Vector<double, Dim> level_hx;
254
255 for (unsigned d = 0; d < Dim; ++d) {
256 const int stride = 1 << ell; // 2^ell
257 // strided sub-index: first, last, stride
258 sub_domain[d] =
259 ippl::Index(fine_domain[d].first(), fine_domain[d].last(), stride);
260 level_hx[d] = fine_mesh.getMeshSpacing(d) * stride;
261 }
262
263 auto level_layout = std::make_shared<ippl::SubFieldLayout<Dim>>(
264 fine_layout.comm, fine_domain, sub_domain, decomp, fine_layout.isAllPeriodic_m);
265
266 auto level_mesh =
267 std::make_shared<mesh_type>(sub_domain, level_hx, fine_mesh.getOrigin());
268
269 // create BCs for each level
270 ippl::BConds<Field, Dim> level_bcs;
271 for (size_t i = 0; i < 2 * Dim; ++i) {
272 if (bcs[i]) {
273 int bc_type = bcs[i]->getBCType();
274
275 if (bc_type == ippl::PERIODIC_FACE) {
276 // PeriodicFace has internal MPI state (faceNeighbors_m, haloData_m).
277 // We MUST create a fresh instance for every level so it binds
278 // correctly.
279 level_bcs[i] = std::make_shared<ippl::PeriodicFace<Field>>(i);
280
281 } else if (ell > 0
282 && (bc_type == ippl::CONSTANT_FACE
283 || bc_type == ippl::ZERO_FACE)) {
284 // Error equation on coarse grids: inhomogeneous Dirichlet becomes Zero
285 level_bcs[i] = std::make_shared<ippl::ZeroFace<Field>>(i);
286
287 } else {
288 // For ell == 0 CONSTANT_FACE, or ZERO_FACE, or NO_FACE,
289 // they have no internal layout state. We can safely share the pointer
290 level_bcs[i] = bcs[i];
291 }
292 }
293 }
294
295 L_.emplace_back(level_mesh, level_layout, level_bcs);
296 }
297
298 IpplTimings::stopTimer(init_fields);
299 }

309 void restrict_average(const size_t level) {
310 IpplTimings::TimerRef restrict = IpplTimings::getTimer("restrict_fullweight");
311 IpplTimings::startTimer(restrict);
312
313 if (level >= L_.size() - 1) {
314 std::cerr << "Trying to restrict at lowest level." << std::endl;
315 return;
316 }
317
318 auto& lev_fine = L_[level];
319 auto& lev_coarse = L_[level + 1];
320
321 // 1. Calculate and sync residual
322 Field residual_fine = residual(lev_fine.u, lev_fine.f);
323 residual_fine.fillHalo();
324 lev_coarse.f = 0.0;
325
326 // 2. Setup domains and views
327 const auto lDomF = residual_fine.getLayout().getLocalNDIndex();
328 const auto lDomC = lev_coarse.f.getLayout().getLocalNDIndex();
329
330 const int nghF = residual_fine.getNghost();
331 const int nghC = lev_coarse.f.getNghost();
332
333 auto rf = residual_fine.getView();
334 auto fc = lev_coarse.f.getView();
335
336 // 3. Calculate number of children per coarser cell
337 constexpr int num_children = 1 << Dim; // 2^Dim
338 constexpr double denom = static_cast<double>(num_children); // 2^Dim
339
340 using index_array_type = typename RangePolicy<Dim>::index_array_type;
341
342 // 4. N-Dimensional Kokkos Loop
343 ippl::parallel_for(
344 "restrict_average", lev_coarse.f.getFieldRangePolicy(),
345 KOKKOS_LAMBDA(const index_array_type& args) {
346 // Local coarse index -> local fine "lower corner" of the contained block
347 ippl::Vector<int, Dim> idxF_base;
348 for (unsigned d = 0; d < Dim; ++d) {
349 const int localC = static_cast<int>(args[d]) - nghC;
350
351 // Coarse point in the original fine-grid index space.
352 const int globalC = lDomC[d].first() + localC * lDomC[d].stride();
353
354 // Same physical/index-space point, expressed as fine-level view index.
355 idxF_base[d] = (globalC - lDomF[d].first()) / lDomF[d].stride() + nghF;
356 }
357
358 // Sum the 2^Dim contained fine cells
359 double sum = 0.0;
360 for (int s = 0; s < num_children; ++s) {
361 Vector<int, Dim> idxF = idxF_base;
362 for (unsigned d = 0; d < Dim; ++d) {
363 idxF[d] += (s >> d) & 1;
364 }
365 sum += apply(rf, idxF);
366 }
367
368 apply(fc, args) = sum / denom;
369 });
370
371 ippl::fence();
372
373 // Remove Volume average if periodic
374 if (is_all_periodic_) {
375 auto avg = lev_coarse.f.getVolumeAverage();
376 lev_coarse.f = lev_coarse.f - avg;
377 }
378
379 if (communication_)
380 lev_coarse.f.fillHalo();
381
382 IpplTimings::stopTimer(restrict);
383 }

393 void prolong_add(const size_t level) {
394 IpplTimings::TimerRef prolong = IpplTimings::getTimer("prolong");
395 IpplTimings::startTimer(prolong);
396
397 if (level >= L_.size() - 1) {
398 std::cerr << "Trying to prolong at invalid level" << std::endl;
399 return;
400 }
401
402 auto& lev_fine = L_[level];
403 auto& lev_coarse = L_[level + 1];
404
405 // 1. Sync coarse grid ghost cells (crucial because interpolation reads adjacent coarse
406 // nodes)
407 if (communication_)
408 lev_coarse.u.fillHalo();
409
410 // 2. Setup domains and views
411 const auto lDomF = lev_fine.u.getLayout().getLocalNDIndex();
412 const auto lDomC = lev_coarse.u.getLayout().getLocalNDIndex();
413
414 const int nghF = lev_fine.u.getNghost();
415 const int nghC = lev_coarse.u.getNghost();
416
417 auto uf = lev_fine.u.getView();
418 auto uc = lev_coarse.u.getView();
419
420 const auto gDomF = lev_fine.u.getLayout().getDomain();
421
422 // 3. Calculate number of corners
423 constexpr int num_corners = 1 << Dim; // 2^Dim contributing coarse cells
424
425 using index_array_type = typename RangePolicy<Dim>::index_array_type;
426
427 // 4. N-Dimensional Kokkos Loop over the Fine Grid
428 ippl::parallel_for(
429 "prolong_add", lev_fine.u.getFieldRangePolicy(),
430 KOKKOS_LAMBDA(const index_array_type& args) {
431 // For each dim: which coarse cell contains us, and on which side?
432 // sgn = -1 -> fine cell in lower half, neighbor is C-1
433 // sgn = +1 -> fine cell in upper half, neighbor is C+1
434 ippl::Vector<int, Dim> idxC_base;
435 ippl::Vector<int, Dim> sgn;
436 for (unsigned d = 0; d < Dim; ++d) {
437 // Fine-level point in original fine-grid index space.
438 const int localF = static_cast<int>(args[d]) - nghF;
439 const int globalF = lDomF[d].first() + localF * lDomF[d].stride();
440
441 // Logical index on this fine level:
442 // level 0: global 0,1,2,3 -> logical 0,1,2,3
443 // level 1: global 0,2,4,6 -> logical 0,1,2,3
444 // level 2: global 0,4,8 -> logical 0,1,2
445 const int logicalF = (globalF - gDomF[d].first()) / lDomF[d].stride();
446
447 const int logicalC = logicalF / 2;
448 sgn[d] = (logicalF % 2 == 0) ? -1 : +1;
449
450 // Coarse-level global coordinate corresponding to logicalC.
451 const int globalC = gDomF[d].first() + logicalC * lDomC[d].stride();
452
453 // Map coarse global coordinate to coarse view index.
454 idxC_base[d] = (globalC - lDomC[d].first()) / lDomC[d].stride() + nghC;
455 }
456
457 // Tensor product of {3/4 (containing), 1/4 (neighbor)} per dim
458 double interp_val = 0.0;
459 for (int s = 0; s < num_corners; ++s) {
460 Vector<int, Dim> idxC = idxC_base;
461 double weight = 1.0;
462 for (unsigned d = 0; d < Dim; ++d) {
463 int b = (s >> d) & 1; // 0 = self, 1 = neighbor
464 idxC[d] += b * sgn[d];
465 weight *= (b == 0) ? 0.75 : 0.25;
466 }
467 interp_val += weight * apply(uc, idxC);
468 }
469
470 // Multigrid: ADD the correction onto the existing fine solution
471 apply(uf, args) += interp_val;
472 });
473 ippl::fence();
474
475 IpplTimings::stopTimer(prolong);
476 }

597 void vcycle(size_t level) {
598 if (level == L_.size() - 1) {
599 // Coarsest grid: just smooth a lot (or use a direct solver)
600 smooth_jacobi(level, 50);
601 return;
602 }
603 smooth_jacobi(level, nu1_); // Pre-smoothing
604 restrict_average(level); // Pass level
605 vcycle(level + 1); // Recursively go down one level
606 prolong_add(level); // Pass level
607 smooth_jacobi(level, nu2_); // Post-smoothing
608 }

616 void smooth_jacobi(const size_t level, const unsigned iters) {
617 IpplTimings::TimerRef jacobi = IpplTimings::getTimer("smooth_jacobi");
618 IpplTimings::startTimer(jacobi);
619
620 auto& lev = L_[level];
621 auto& u = lev.u;
622 const auto& f = lev.f;
623
624 const auto diag = multigrid::compute_diag(lev);
625
626 for (unsigned it = 0; it < iters; ++it) {
627 if (communication_)
628 u.fillHalo();
629
630 Field res = residual(u, f);
631 u = u + omega_ * (res / diag);
632 }
633 IpplTimings::stopTimer(jacobi);
634 }
