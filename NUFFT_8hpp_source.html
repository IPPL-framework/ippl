387 const ParticleAttrib<Vector<T, Dim>, Properties...>& R, ParticleAttrib<T, Properties...>& Q,
388 ComplexField& f) {
389 // The FINUFFT/cuFINUFFT path is hardcoded for Dim == 3 (3D mode-grid
390 // scratch + Rank<3> MDRangePolicy in transformFinufft). For Dim != 3
391 // we skip the constexpr branch entirely so the 3D-only template body
392 // is never instantiated -- otherwise the unit tests that exercise the
393 // 2D type instantiations of FFT<NUFFTransform, ...> fail to compile
394 // under a FINUFFT-enabled build.
395 if constexpr (Dim == 3 && fft::is_available_v<fft::Finufft>) {
396 if (useFinufft_m) {
397 transformFinufft(R, Q, f);
398 return;
399 }
400 }
401
402 transformNative(R, Q, f);
403 }

409 template <typename RealField>
410 template <class... Properties>
411 void FFT<NUFFTransform, RealField>::transformNative(
412 const ParticleAttrib<Vector<T, Dim>, Properties...>& R, ParticleAttrib<T, Properties...>& Q,
413 ComplexField& f) {
414 const auto localNp = R.getParticleCount();
415 const auto& layout = f.getLayout();
416 const auto& mesh = f.get_mesh();
417 const auto& dx = mesh.getMeshSpacing();
418 const auto& domain = layout.getDomain();
419
420 Vector<T, Dim> Len;
421 for (unsigned d = 0; d < Dim; ++d) {
422 int fullLength = domain[d].length();
423 if (useR2C_m && static_cast<int>(d) == r2cDir_m) {
424 fullLength = 2 * (fullLength - 1);
425 }
426 Len[d] = dx[d] * fullLength;
427 }
428
429 constexpr T twoPi = T(2.0 * M_PI);
430 auto Rview = R.getView();
431
432 Vector<T, Dim> scaleToTwoPi, scaleBack;
433 for (unsigned d = 0; d < Dim; ++d) {
434 scaleToTwoPi[d] = twoPi / Len[d];
435 scaleBack[d] = Len[d] / twoPi;
436 }
437
438 using ScaleFunctor = detail::ScalePositionsFunctor<T, Dim, decltype(Rview)>;
439 Kokkos::parallel_for("NUFFT_scale_to_2pi", Kokkos::RangePolicy<ExecSpace>(0, localNp),
440 ScaleFunctor(Rview, scaleToTwoPi));
441
442 if (type_m == 1) {
443 nativeNufft_m->type1(R, Q, f, useUpsampledInputs_m);
444 } else if (type_m == 2) {
445 nativeNufft_m->type2(f, R, Q, useUpsampledInputs_m);
446 } else {
447 throw IpplException("FFT<NUFFTransform>", "Only type 1 and type 2 NUFFT supported");
448 }
449
450 Kokkos::parallel_for("NUFFT_scale_back", Kokkos::RangePolicy<ExecSpace>(0, localNp),
451 ScaleFunctor(Rview, scaleBack));
452 }

457 template <typename RealField>
458 template <class... Properties>
459 void FFT<NUFFTransform, RealField>::transformFinufft(
460 const ParticleAttrib<Vector<T, Dim>, Properties...>& R, ParticleAttrib<T, Properties...>& Q,
461 ComplexField& f) {
462#ifdef ENABLE_FINUFFT
463 const auto localNp = R.getParticleCount();
464 const auto& layout = f.getLayout();
465 const auto& mesh = f.get_mesh();
466 const auto& dx = mesh.getMeshSpacing();
467 const auto& domain = layout.getDomain();
468 const int nghost = f.getNghost();
469
470 Vector<T, Dim> Len;
471 for (unsigned d = 0; d < Dim; ++d) {
472 Len[d] = dx[d] * domain[d].length();
473 }
474
475 constexpr T twoPi = T(2.0 * M_PI);
476
477 auto fview = f.getView();
478 auto Rview = R.getView();
479 auto Qview = Q.getView();
480
481 const auto& lDom = layout.getLocalNDIndex();
482 if (tempField_m.extent(0) != static_cast<std::size_t>(lDom[0].length())
483 || tempField_m.extent(1) != static_cast<std::size_t>(lDom[1].length())
484 || tempField_m.extent(2) != static_cast<std::size_t>(lDom[2].length())) {
485 Kokkos::realloc(tempField_m, lDom[0].length(), lDom[1].length(), lDom[2].length());
486 }
487
488 if (tempQ_m.extent(0) < localNp) {
489 Kokkos::realloc(tempQ_m, localNp);
490 }
491
492 for (unsigned d = 0; d < Dim; ++d) {
493 if (tempR_m[d].extent(0) < localNp) {
494 Kokkos::realloc(tempR_m[d], localNp);
495 }
496 }
497
498 auto tempField = tempField_m;
499 auto tempQ = tempQ_m;
500 auto tempRx = tempR_m[0];
501 auto tempRy = tempR_m[1];
502 auto tempRz = tempR_m[2];
503
504 Vector<T, Dim> scale;
505 for (unsigned d = 0; d < Dim; ++d) {
506 scale[d] = twoPi / Len[d];
507 }
508
509 using mdrange_type = Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<3>>;
510 using CopyToTemp = detail::CopyFieldToTempFunctor<decltype(fview), decltype(tempField)>;
511
512 int nx = lDom[0].length();
513 int ny = lDom[1].length();
514 int nz = lDom[2].length();
515 bool needShift = (type_m == 2);
516
517 Kokkos::parallel_for(
518 "FINUFFT_copy_field_to_temp",
519 mdrange_type({nghost, nghost, nghost}, {static_cast<int>(fview.extent(0)) - nghost,
520 static_cast<int>(fview.extent(1)) - nghost,
521 static_cast<int>(fview.extent(2)) - nghost}),
522 CopyToTemp(fview, tempField, nghost, nx, ny, nz, needShift));
523
524 using CopyParticles =
525 detail::CopyParticlesToTempFunctor<T, Dim, decltype(Rview), decltype(Qview),
526 decltype(tempRx), decltype(tempQ)>;
527
528 Kokkos::parallel_for("FINUFFT_copy_particles_to_temp", localNp,
529 CopyParticles(Rview, Qview, tempRx, tempRy, tempRz, tempQ, scale));
530
531 Kokkos::fence();
532
533 int err = Traits_t::setpts(finufftPlan_m, static_cast<FinufftCount_t>(localNp),
534 tempRx.data(), tempRy.data(), tempRz.data(), FinufftCount_t{0},
535 nullptr, nullptr, nullptr);
536
537 if (err != 0) {
538 throw IpplException("FFT<NUFFTransform>", "FINUFFT setpts failed");
539 }
540
541 err = Traits_t::execute(finufftPlan_m, tempQ.data(), tempField.data());
542
543 if (err != 0) {
544 throw IpplException("FFT<NUFFTransform>", "FINUFFT execute failed");
545 }
546
547 Kokkos::fence();
548
549 if (type_m == 1) {
550 using CopyFromTemp =
551 detail::CopyFieldFromTempFunctor<decltype(fview), decltype(tempField)>;
552
553 Kokkos::parallel_for("FINUFFT_copy_field_from_temp",
554 mdrange_type({nghost, nghost, nghost},
555 {static_cast<int>(fview.extent(0)) - nghost,
556 static_cast<int>(fview.extent(1)) - nghost,
557 static_cast<int>(fview.extent(2)) - nghost}),
558 CopyFromTemp(fview, tempField, nghost, nx, ny, nz));
559 } else if (type_m == 2) {
560 using CopyBack = detail::CopyParticlesFromTempFunctor<decltype(Qview), decltype(tempQ)>;
561
562 Kokkos::parallel_for("FINUFFT_copy_particles_from_temp", localNp,
563 CopyBack(Qview, tempQ));
564 }
565#else
566 throw std::runtime_error("FINUFFT is not activated. Rebuild with -DIPPL_ENABLE_FINUFFT=ON");
567 (void)R;
568 (void)Q;
569 (void)f;
570#endif // ENABLE_FINUFFT
571 }
