10 template <typename E>
11 struct meta_poisson : public Expression<meta_poisson<E>, sizeof(E)> {
12 constexpr static unsigned dim = E::dim;
13
14 KOKKOS_FUNCTION
15 meta_poisson(const E& u)
16 : u_m(u) {}

18 template <typename... Idx>
19 KOKKOS_INLINE_FUNCTION auto operator()(const Idx... args) const {
20 using index_type = std::tuple_element_t<0, std::tuple<Idx...>>;
21 using T = typename E::Mesh_t::value_type;
22 T res = 0;
23 index_type coords[dim] = {args...};
24 auto&& center = apply(u_m, coords);
25 for (unsigned d = 0; d < dim; d++) {
26 index_type coords[dim] = {args...};
27
28 coords[d] -= 1;
29 auto&& left = apply(u_m, coords);
30
31 coords[d] += 2;
32 auto&& right = apply(u_m, coords);
33 res += (2.0 * center - left - right);
34 }
35 return res;
36 }
37
38 private:
39 const E u_m;
40 };

60 /*
61 * n-dimensional lower triangular Laplacian
62 */
63 template <typename... Idx>
64 KOKKOS_INLINE_FUNCTION auto operator()(const Idx... args) const {
65 using index_type = std::tuple_element_t<0, std::tuple<Idx...>>;
66 using T = typename E::Mesh_t::value_type;
67 T res = 0;
68
69 for (unsigned d = 0; d < dim; d++) {
70 index_type coords[dim] = {args...};
71 const int global_index = coords[d] + ldom_m[d].first() - nghosts_m;
72 const int size = domain_m.length()[d];
73 const bool not_left_boundary = (global_index != 0);
74 const bool right_boundary = (global_index == size - 1);
75
76 coords[d] -= 1;
77 auto&& left = apply(u_m, coords);
78
79 coords[d] += 2;
80 auto&& right = apply(u_m, coords);
81
82 // not_left_boundary and right_boundary are boolean values
83 // Because of periodic boundary conditions we need to add this boolean mask to
84 // obtain the lower triangular part of the Laplace Operator
85 res += hvector_m[d] * (not_left_boundary * left + right_boundary * right);
86 }
87 return res;
88 }
89
90 private:
91 using Mesh_t = typename E::Mesh_t;
92 using Layout_t = typename E::Layout_t;
93 using vector_type = typename Mesh_t::vector_type;
94 using domain_type = typename Layout_t::NDIndex_t;
95 const E u_m;
96 const vector_type hvector_m;
97 const unsigned nghosts_m;
98 const domain_type ldom_m;
99 const domain_type domain_m;
100 };

323 template <typename Field>
324 Field negative_inverse_diagonal_laplace(Field& u) {
325 constexpr unsigned Dim = Field::dim;
326 using mesh_type = typename Field::Mesh_t;
327 using layout_type = typename Field::Layout_t;
328 mesh_type& mesh = u.get_mesh();
329 layout_type& layout = u.getLayout();
330 Field res(mesh, layout);
331 auto&& bc = u.getFieldBC();
332 res.setFieldBC(bc);
333 double sum = 0.0;
334 double factor = 1.0;
335 typename mesh_type::vector_type hvector(0);
336 for (unsigned d = 0; d < Dim; ++d) {
337 hvector[d] = std::pow(mesh.getMeshSpacing(d), 2);
338 sum += std::pow(mesh.getMeshSpacing(d), 2)
339 * std::pow(mesh.getMeshSpacing((d + 1) % Dim), 2);
340 factor *= hvector[d];
341 }
342
343 res = 0.5 * (factor / sum) * u;
344 return res;
345 }
