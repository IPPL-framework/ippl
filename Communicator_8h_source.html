160 template <class Buffer, typename Archive>
161 void recv(int src, int tag, Buffer& buffer, Archive& ar, size_type msize,
162 size_type nrecvs) {
163 // Temporary fix. MPI communication seems to have problems when the
164 // count argument exceeds the range of int, so large messages should
165 // be split into smaller messages
166 if (msize > INT_MAX) {
167 std::cerr << "Message size exceeds range of int" << std::endl;
168 this->abort();
169 }
170 MPI_Status status;
171 MPI_Recv(ar.getBuffer(), msize, MPI_BYTE, src, tag, *comm_m, &status);
172
173 buffer.deserialize(ar, nrecvs);
174 }
175
176 template <class Buffer, typename Archive>
177 void isend(int dest, int tag, Buffer& buffer, Archive& ar, MPI_Request& request,
178 size_type nsends) {
179 if (ar.getSize() > INT_MAX) {
180 std::cerr << "Message size exceeds range of int" << std::endl;
181 this->abort();
182 }
183 buffer.serialize(ar, nsends);
184 MPI_Isend(ar.getBuffer(), ar.getSize(), MPI_BYTE, dest, tag, *comm_m, &request);
185 }

191 this->abort();
192 }
193 MPI_Irecv(ar.getBuffer(), msize, MPI_BYTE, src, tag, *comm_m, &request);
