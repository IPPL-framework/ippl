78 // Map normalized value [0,1] to candidate index. With no candidates the
79 // tuner has nothing to choose from, so return 1 as a defensive default.
80 int map_to_candidate(double normalized) const {
81 if (candidates_.empty()) return 1;
82
83 normalized = std::clamp(normalized, 0.0, 1.0);
84
85 size_t idx = static_cast<size_t>(normalized * (candidates_.size() - 1) + 0.5);
86 idx = std::min(idx, candidates_.size() - 1);

93 // list cannot meaningfully be mapped.

147 // Scale down proportionally if tile doesn't fit in scratch.
148 TileConfig fit_to_scratch(const TileConfig& tile) const {
149 if (scratch_calc_(tile) <= max_scratch_) {
150 return tile;
151 }
152
153 // Binary search for the largest scale factor that fits. The result
154 // space is integer-valued (we snap to a discrete candidate after
155 // scaling), so the number of meaningful iterations is bounded by
156 // log2(max tile dim), capped at 20 for robustness.
157 int max_tile = 1;
158 for (unsigned d = 0; d < Dim; ++d) {
159 max_tile = std::max(max_tile, tile[d]);
160 }
161 int max_iter = std::min(20, static_cast<int>(std::ceil(std::log2(max_tile + 1)) + 1));
162
163 TileConfig result = tile;
164 double lo = 0.0, hi = 1.0;
165 for (int iter = 0; iter < max_iter; ++iter) {
166 double mid = (lo + hi) / 2.0;
167
168 TileConfig test;
169 for (unsigned d = 0; d < Dim; ++d) {
170 int scaled = static_cast<int>(tile[d] * mid);
171 test[d] = snap_to_candidate(std::max(1, scaled));
172 }
173
174 if (scratch_calc_(test) <= max_scratch_) {
175 lo = mid;
176 result = test;
177 } else {
178 hi = mid;
179 }
180 }
181
182 // Final fallback - smallest candidate. If even that overflows the
183 // scratch budget the kernel cannot run; throw so the caller doesn't
184 // get a confusing Kokkos::abort deeper in the dispatch.
185 if (scratch_calc_(result) > max_scratch_) {
186 for (unsigned d = 0; d < Dim; ++d) {
187 result[d] = candidates_.front();
188 }
189 if (scratch_calc_(result) > max_scratch_) {
190 throw std::runtime_error(
191 "TileSizeTuner::fit_to_scratch: even the smallest tile "
192 "exceeds the scratch budget");
193 }
194 }
195
196 return result;
197 }

199 int snap_to_candidate(int value) const {
200 if (candidates_.empty()) return value;
201
202 // Find largest candidate <= value
203 auto it = std::upper_bound(candidates_.begin(), candidates_.end(), value);
204 if (it == candidates_.begin()) {
205 return candidates_.front();
206 }
207 return *std::prev(it);
208 }
