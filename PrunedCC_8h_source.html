39 template <bool IsCPU, typename Func>
40 inline void runConcurrentBatch(int count, Func&& func) {
41 if constexpr (IsCPU) {
42 // Serial outer loop for CPU (Kokkos will parallelize the inner loops)
43 for (int local = 0; local < count; ++local) {
44 func(local);
45 }
46 } else {
47 // OpenMP outer loop for GPU (to overlap asynchronous stream launches)
48#if defined(_OPENMP)
49#pragma omp parallel for
50#endif
51 for (int local = 0; local < count; ++local) {
52 func(local);
53 }
54 }
55 }

169 auto inView = input.getView();
170 auto outView = output.getView();
171 const int ngIn = input.getNghost();
172 const int ngOut = output.getNghost();
173
174 const auto& lDomPruned = output.getLayout().getLocalNDIndex();

178 // Ensure temps
179 for (int s = 0; s < numConcurrent_; ++s) {
180 if (temps_[s].size() != output.getOwned().size()) {
181 temps_[s] = detail::shrinkView("pruned_temp_" + std::to_string(s), outView, ngOut);
182 }
183 }
184
185 Kokkos::deep_copy(outView, Complex_t(0, 0));
186
187 double scale = 1.0;
188 if (dir == 1) {
189 for (unsigned d = 0; d < Dim; ++d) {
190 scale *= double(modes[d]) / double(gDomFull[d].length());
191 }
192 }
193
194 std::array<Vector<long, Dim>, NumSubFFTs> offsets;
195 for (int k = 0; k < NumSubFFTs; ++k) {
196 for (unsigned d = 0; d < Dim; ++d) {
197 offsets[k][d] = (k >> d) & 1;
198 }
199 }
200
201 Vector<int, Dim> localFirst;
202 for (unsigned d = 0; d < Dim; ++d) {
203 localFirst[d] = lDomPruned[d].first();
204 }

206 auto owned = output.getOwned();
207 const int numBatches = (NumSubFFTs + numConcurrent_ - 1) / numConcurrent_;
208
209 constexpr bool is_cpu = false
210#ifdef KOKKOS_ENABLE_SERIAL
211 || std::is_same_v<ExecSpace, Kokkos::Serial>
212#endif
213#ifdef KOKKOS_ENABLE_OPENMP
214 || std::is_same_v<ExecSpace, Kokkos::OpenMP>
215#endif
216 ;
217
218 const long ext0 = static_cast<long>(owned[0].length());
219 const long ext1 = static_cast<long>(owned[1].length());
220 [[maybe_unused]] const long ext2 =
221 (Dim == 3) ? static_cast<long>(owned[Dim == 3 ? 2 : 0].length()) : 1L;
222
223 for (int batch = 0; batch < numBatches; ++batch) {
224 const int start = batch * numConcurrent_;
225 const int end = std::min(start + numConcurrent_, NumSubFFTs);
226 const int count = end - start;
227
228 IpplTimings::startTimer(subFFTTimer);
229
230 // Using the wrapper to evaluate thread parallelism safely
231 detail::runConcurrentBatch<is_cpu>(count, [&](int local) {
232 const int k = start + local;
233 auto offs = offsets[k];
234 auto& temp = temps_[local];
235
236 if constexpr (Dim == 3) {
237 auto copy_lambda = KOKKOS_LAMBDA(int i0, int i1, int i2) {
238 int si = i0 * 2 + int(offs[0]) + ngIn;
239 int sj = i1 * 2 + int(offs[1]) + ngIn;
240 int sk = i2 * 2 + int(offs[2]) + ngIn;
241 temp(i0, i1, i2) = inView(si, sj, sk);
242 };
243 if constexpr (is_cpu) {
244 Kokkos::parallel_for(
245 "strided_copy_forward",
246 Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<3>>(
247 {0, 0, 0}, {ext0, ext1, ext2}),
248 copy_lambda);
249 } else {
250 auto exec = GPUOps::instance(streams_[local]);
251 Kokkos::parallel_for(
252 "strided_copy_forward",
253 Kokkos::MDRangePolicy<DeviceExec, Kokkos::Rank<3>>(
254 exec, {0, 0, 0}, {ext0, ext1, ext2}),
255 copy_lambda);
256 GPUOps::sync(streams_[local]);
257 }
258 } else {
259 auto copy_lambda = KOKKOS_LAMBDA(int i0, int i1) {
260 int si = i0 * 2 + int(offs[0]) + ngIn;
261 int sj = i1 * 2 + int(offs[1]) + ngIn;
262 temp(i0, i1) = inView(si, sj);
263 };
264 if constexpr (is_cpu) {
265 Kokkos::parallel_for(
266 "strided_copy_forward",
267 Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<2>>({0, 0},
268 {ext0, ext1}),
269 copy_lambda);
270 } else {
271 auto exec = GPUOps::instance(streams_[local]);
272 Kokkos::parallel_for(
273 "strided_copy_forward",
274 Kokkos::MDRangePolicy<DeviceExec, Kokkos::Rank<2>>(exec, {0, 0},
275 {ext0, ext1}),
276 copy_lambda);
277 GPUOps::sync(streams_[local]);
278 }
279 }
280
281 if (dir == 1) {
282 backends_[local]->forward(temp.data(), temp.data());
283 } else {
284 backends_[local]->backward(temp.data(), temp.data());
285 }
286 });
287
288 Kokkos::fence();
289 IpplTimings::stopTimer(subFFTTimer);
290
291 IpplTimings::startTimer(twiddleTimer);
292
293 for (int local = 0; local < count; ++local) {
294 const int k = start + local;
295 auto offs = offsets[k];
296 auto& temp = temps_[local];
297
298 const long g0 = static_cast<long>(gDomFull[0].length());
299 const long g1 = static_cast<long>(gDomFull[1].length());
300 const long g2 =
301 (Dim == 3) ? static_cast<long>(gDomFull[Dim == 3 ? 2 : 0].length()) : 1L;
302 const long m0 = static_cast<long>(modes[0]);
303 const long m1 = static_cast<long>(modes[1]);
304 const long m2 = (Dim == 3) ? static_cast<long>(modes[Dim == 3 ? 2 : 0]) : 1L;
305 const int lf0 = localFirst[0];
306 const int lf1 = localFirst[1];
307 const int lf2 = (Dim == 3) ? localFirst[Dim == 3 ? 2 : 0] : 0;
308
309 if constexpr (Dim == 3) {
310 Kokkos::parallel_for(
311 "twiddle_add_forward",
312 Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<3>>(
313 {ngOut, ngOut, ngOut},
314 {int(outView.extent(0)) - ngOut, int(outView.extent(1)) - ngOut,
315 int(outView.extent(2)) - ngOut}),
316 KOKKOS_LAMBDA(int i, int j, int kk) {
317 int gi = i - ngOut + lf0;
318 int gj = j - ngOut + lf1;
319 int gk = kk - ngOut + lf2;
320
321 int64_t f0 = (gi < int64_t(m0) / 2) ? gi : int64_t(g0) - int64_t(m0) + gi;
322 int64_t f1 = (gj < int64_t(m1) / 2) ? gj : int64_t(g1) - int64_t(m1) + gj;
323 int64_t f2 = (gk < int64_t(m2) / 2) ? gk : int64_t(g2) - int64_t(m2) + gk;
324
325 Complex_t w(1.0, 0.0);
326 auto twiddle = [&](int64_t freq, int64_t N) {
327 double ang = -dir * 2.0 * M_PI * double(freq) / double(N);
328 return Complex_t(Kokkos::cos(ang), Kokkos::sin(ang));
329 };
330
331 if (offs[0]) w *= twiddle(f0, g0);
332 if (offs[1]) w *= twiddle(f1, g1);
333 if (offs[2]) w *= twiddle(f2, g2);
334
335 auto val = temp(i - ngOut, j - ngOut, kk - ngOut);
336 outView(i, j, kk) += w * val * scale;
337 });
338 } else {
339 Kokkos::parallel_for(
340 "twiddle_add_forward",
341 Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<2>>(
342 {ngOut, ngOut},
343 {int(outView.extent(0)) - ngOut, int(outView.extent(1)) - ngOut}),
344 KOKKOS_LAMBDA(int i, int j) {
345 int gi = i - ngOut + lf0;
346 int gj = j - ngOut + lf1;
347
348 int64_t f0 = (gi < int64_t(m0) / 2) ? gi : int64_t(g0) - int64_t(m0) + gi;
349 int64_t f1 = (gj < int64_t(m1) / 2) ? gj : int64_t(g1) - int64_t(m1) + gj;
350
351 Complex_t w(1.0, 0.0);
352 auto twiddle = [&](int64_t freq, int64_t N) {
353 double ang = -dir * 2.0 * M_PI * double(freq) / double(N);
354 return Complex_t(Kokkos::cos(ang), Kokkos::sin(ang));
355 };
356
357 if (offs[0]) w *= twiddle(f0, g0);
358 if (offs[1]) w *= twiddle(f1, g1);
359
360 auto val = temp(i - ngOut, j - ngOut);
361 outView(i, j) += w * val * scale;
362 });
363 }
364 }
365
366 IpplTimings::stopTimer(twiddleTimer);
367 }
