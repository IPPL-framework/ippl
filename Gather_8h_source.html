31 // Use the kernel's own value_type for the geometric real-precision
32 // computations (origin, invdx, weights). When the position attribute
33 // is float but the mesh is double, taking RealType from the position
34 // would silently downcast the mesh spacing and lose precision; the
35 // kernel knows the correct working precision and the legacy CIC
36 // path uses the mesh type explicitly via Kernel = LinearKernel<T>.
37 using RealType = typename std::decay_t<Kernel>::value_type;
38
39 using type = GatherTypes<FieldTr::dim, RealType, std::decay_t<Kernel>,
40 typename FieldTr::view_type, typename PosTr::view_type,
41 typename ValTr::view_type>;
42 };
