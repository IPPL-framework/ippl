4// ============================================================================
5// TileSizeCache
6// ============================================================================
7//
8// Singleton that loads optimal configurations produced by BenchmarkTileSweep
9// and exposes them to Scatter::dispatch at runtime.
10//
11// Supported CSV files (checked in order):
12// 1. $IPPL_TILE_CSV -- explicit path override
13// 2. tile_sweep_sa_optimal.csv -- BO-optimised + Atomic results (preferred)
14// 3. tile_sweep_optimal.csv -- uniform sweep results (fallback)
15//
16// -- Formats ------------------------------------------------------------------
17//
18// Uniform sweep (write_optimal_csv), with rho column:
19// method,kernel_width,rho,optimal_tile_size,throughput_Mpts_s,time_ms
20//
21// BO/Atomic (write_bo_csv), with rho column:
22// method,value_type,kernel_width,rho,
23// best_tile_x,best_tile_y,best_tile_z,
24// best_team_size,best_oversubscription_factor,best_z_batches,
25// throughput_Mpts_s,time_ms,kernel_evaluations,preflight_rejections
26//
27// Both formats are also parseable WITHOUT the rho column (old files).
28// In that case rho defaults to 0.0 ("unspecified").
29//
30// -- Density-aware lookup -----------------------------------------------------
31//
32// Multiple rows with the same (method, width, is_complex) but different rho
33// values coexist. get() / get_best() accept an optional rho parameter:
34// rho <= 0 -> return the entry with the highest throughput (ignore density)
35// rho > 0 -> return the entry whose recorded rho is closest to the query
36// (entries with rho=0 are a low-priority fallback)
37//
38// -- Method selection ---------------------------------------------------------
39//
40// get_best(width, is_complex, rho) returns the method+config with the highest
41// throughput for the query density across Atomic, Tiled, and OutputFocused.
42//
43// Thread safety: initialised once via std::call_once, then read-only.
44// ============================================================================
45
46// Header keeps only the type/API surface and the inline lookup logic.
47// CSV parsing, file IO, and load() live in TileSizeCache.cpp so that every
48// TU including this header doesn't drag in <fstream>/<sstream>.

135 // ------------------------------------------------------------------
136 // Lookup by explicit method + optional density hint.
137 // ------------------------------------------------------------------
138 std::optional<TileCacheEntry> get(ScatterMethod method, int kernel_width,
139 bool is_complex, double rho = 0.0) const {
140 TileCacheKey key{method, kernel_width, is_complex};
141 auto it = entries_.find(key);
142 if (it == entries_.end() || it->second.empty())
143 return std::nullopt;
144 const TileCacheEntry* e = closest_entry(it->second, rho);
145 return e ? std::optional<TileCacheEntry>(*e) : std::nullopt;
146 }
147
148 // ------------------------------------------------------------------
149 // Auto method selection: best throughput across ALL methods at the
150 // closest recorded density.
151 // ------------------------------------------------------------------
152 std::optional<BestCacheEntry> get_best(int kernel_width, bool is_complex,
153 double rho = 0.0) const {
154 std::optional<BestCacheEntry> best;
155 double best_tp = -1.0;
156
157 for (const auto& [key, vec] : entries_) {
158 if (key.kernel_width != kernel_width || key.is_complex != is_complex)
159 continue;
160 if (vec.empty())
161 continue;
162 const TileCacheEntry* e = closest_entry(vec, rho);
163 if (e && e->throughput_Mpts_s > best_tp) {
164 best_tp = e->throughput_Mpts_s;
165 best = BestCacheEntry{key.method, *e};
166 }
167 }
168 return best;
169 }

171 // Convenience: tile-only lookup (backward compat).
172 template <unsigned Dim>
173 std::optional<Vector<int, Dim>> get_tile(ScatterMethod method, int kernel_width,
174 bool is_complex, double rho = 0.0) const {
175 auto entry = get(method, kernel_width, is_complex, rho);
176 if (!entry.has_value())
177 return std::nullopt;
178 const auto& e = entry.value();
179 Vector<int, Dim> tile;
180 for (unsigned d = 0; d < Dim; ++d)
181 tile[d] = e.tile[d < 3 ? d : 2];
182 return tile;

221 // ------------------------------------------------------------------
222 // Density-aware entry selection.
223 // rho <= 0 -> highest throughput among all entries
224 // rho > 0 -> entry with closest positive rho; rho=0 entries are fallback
225 // ------------------------------------------------------------------
226 static const TileCacheEntry* closest_entry(const std::vector<TileCacheEntry>& vec,
227 double rho) {
228 if (vec.empty())
229 return nullptr;
230
231 if (rho <= 0.0) {
232 return &*std::max_element(vec.begin(), vec.end(),
233 [](const TileCacheEntry& a, const TileCacheEntry& b) {
234 return a.throughput_Mpts_s < b.throughput_Mpts_s;
235 });
236 }
237
238 const TileCacheEntry* best = nullptr;
239 double best_dist = std::numeric_limits<double>::max();
240
241 for (const auto& e : vec) {
242 // rho=0 entries used only when no positive-rho entry exists
243 double dist = (e.rho > 0.0) ? std::abs(e.rho - rho) : 1e30;
244 if (dist < best_dist) {
245 best_dist = dist;
246 best = &e;
247 }
248 }
249 // If only rho=0 entries exist, fall back to highest-throughput among them
250 if (best_dist >= 1e29) {
251 return &*std::max_element(vec.begin(), vec.end(),
252 [](const TileCacheEntry& a, const TileCacheEntry& b) {
253 return a.throughput_Mpts_s < b.throughput_Mpts_s;
254 });
255 }
256 return best;
257 }
