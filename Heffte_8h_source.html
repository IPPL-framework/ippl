72 inline size_t computeGlobalSize(const heffte::box3d<long long>& inbox,
73 const heffte::box3d<long long>& outbox, MPI_Comm comm) {
74 long long local_max[3], global_max[3];
75 local_max[0] = std::max(inbox.high[0], outbox.high[0]) + 1;
76 local_max[1] = std::max(inbox.high[1], outbox.high[1]) + 1;
77 local_max[2] = std::max(inbox.high[2], outbox.high[2]) + 1;
78
79 MPI_Allreduce(local_max, global_max, 3, MPI_LONG_LONG, MPI_MAX, comm);
80
81 return static_cast<size_t>(global_max[0]) * static_cast<size_t>(global_max[1])
82 * static_cast<size_t>(global_max[2]);
83 }

99 template <typename HeffteBackendT>
100 heffte::plan_options makeHeffteOptions(const ParameterList& params) {
101 auto opts = heffte::default_options<HeffteBackendT>();
102
103 if (!params.get<bool>("use_heffte_defaults")) {
104 opts.use_pencils = params.get<bool>("use_pencils");
105 opts.use_reorder = params.get<bool>("use_reorder");
106
107 if constexpr (is_available_v<HeffteGPU>) {
108 opts.use_gpu_aware = params.get<bool>("use_gpu_aware");
109 }
110
111 switch (params.get<int>("comm")) {
112 case a2a:
113 opts.algorithm = heffte::reshape_algorithm::alltoall;
114 break;
115 case a2av:
116 opts.algorithm = heffte::reshape_algorithm::alltoallv;
117 break;
118 case p2p:
119 opts.algorithm = heffte::reshape_algorithm::p2p;
120 break;
121 case p2p_pl:
122 opts.algorithm = heffte::reshape_algorithm::p2p_plined;
123 break;
124 default:
125 throw IpplException("FFT", "Unknown communication type");
126 }
127 } else {
128 opts.use_gpu_aware = true;
129 opts.algorithm = heffte::reshape_algorithm::p2p_plined;
130 }
131 return opts;
132 }

168 HeffteC2C(const heffte::box3d<long long>& inbox, const heffte::box3d<long long>& outbox,
169 MPI_Comm comm, const ParameterList& params, int maxBatchSize = 1)
170 : maxBatchSize_(maxBatchSize) {
171 static_assert(Dim == 2 || Dim == 3, "heFFTe only supports 2D and 3D");
172

174 heffte_ = std::make_shared<heffte_t>(inbox, outbox, comm, opts);
175
176 // Allocate workspace for maximum batch size
177 workspace_ = workspace_t(heffte_->size_workspace() * maxBatchSize);

180 globalSize_ = computeGlobalSize(inbox, outbox, comm);
