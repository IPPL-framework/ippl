87 void transform(TransformDirection direction, RealField& f, ComplexField& g) {
88 auto fview = f.getView();
89 auto gview = g.getView();
90 const int ngf = f.getNghost();
91 const int ngg = g.getNghost();
92
93 ensureTemps(f, g);
94 fft::copyToTemp<ExecSpace, decltype(tempReal_), decltype(fview)>(tempReal_, fview, ngf);
95 fft::copyToTemp<ExecSpace, decltype(tempComplex_), decltype(gview)>(tempComplex_, gview,
96 ngg);
97
98 if (direction == FORWARD) {
99 backend_->forward(tempReal_.data(), tempComplex_.data());
100 } else {
101 backend_->backward(tempComplex_.data(), tempReal_.data());
102 }
103
104 fft::copyFromTemp<ExecSpace, decltype(fview), decltype(tempReal_)>(fview, tempReal_,
105 ngf);
106 fft::copyFromTemp<ExecSpace, decltype(gview), decltype(tempComplex_)>(
107 gview, tempComplex_, ngg);
108 }
