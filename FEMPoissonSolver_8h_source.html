85 void solve() override {
86 // create load vector for the problem
87 this->rhs_mp->fillHalo();
88 lagrangeSpace_m.evaluateLoadVector(*(this->rhs_mp));
89
90 const Vector<size_t, Dim> zeroNdIndex = Vector<size_t, Dim>(0);
91
92 // We can pass the zeroNdIndex here, since the transformation jacobian does not depend
93 // on translation
94 const auto firstElementVertexPoints =
95 lagrangeSpace_m.getElementMeshVertexPoints(zeroNdIndex);
96
97 // Compute Inverse Transpose Transformation Jacobian ()
98 const Vector<Tlhs, Dim> DPhiInvT =
99 refElement_m.getInverseTransposeTransformationJacobian(firstElementVertexPoints);
100
101 // Compute absolute value of the determinant of the transformation jacobian (|det D
102 // Phi_K|)
103 const Tlhs absDetDPhi = Kokkos::abs(
104 refElement_m.getDeterminantOfTransformationJacobian(firstElementVertexPoints));
105
106 EvalFunctor<Tlhs, Dim, LagrangeType::numElementDOFs> poissonEquationEval(
107 DPhiInvT, absDetDPhi);
108
109 // get BC type of our RHS
110 BConds<FieldRHS, Dim>& bcField = (this->rhs_mp)->getFieldBC();
111 FieldBC bcType = bcField[0]->getBCType();
112
113 const auto algoOperator = [poissonEquationEval, &bcField, this](rhs_type field) -> lhs_type {
114 // set appropriate BCs for the field as the info gets lost in the CG iteration
115 field.setFieldBC(bcField);
116
117 field.fillHalo();
118
119 auto return_field = lagrangeSpace_m.evaluateAx(field, poissonEquationEval);
120
121 return return_field;
122 };
123
124 pcg_algo_m.setOperator(algoOperator);
125
126 // send boundary values to RHS (load vector) i.e. lifting (Dirichlet BCs)
127 if (bcType == CONSTANT_FACE) {
128 *(this->rhs_mp) = *(this->rhs_mp) -
129 lagrangeSpace_m.evaluateAx_lift(*(this->rhs_mp), poissonEquationEval);
130 }
131
132 // start a timer
133 static IpplTimings::TimerRef pcgTimer = IpplTimings::getTimer("pcg");
134 IpplTimings::startTimer(pcgTimer);
135
136 pcg_algo_m(*(this->lhs_mp), *(this->rhs_mp), this->params_m);
137
138 (this->lhs_mp)->fillHalo();
139
140 IpplTimings::stopTimer(pcgTimer);
141 }
