13 template <typename Tlhs, unsigned Dim, unsigned numElemDOFs>
14 struct EvalFunctor {
15 const Vector<Tlhs, Dim> DPhiInvT;
16 const Tlhs absDetDPhi;
17
18 EvalFunctor(Vector<Tlhs, Dim> DPhiInvT, Tlhs absDetDPhi)
19 : DPhiInvT(DPhiInvT)
20 , absDetDPhi(absDetDPhi) {}
21
22 KOKKOS_FUNCTION const auto operator()(
23 const size_t& i, const size_t& j,
24 const Vector<Vector<Tlhs, Dim>, numElemDOFs>& grad_b_q_k) const {
25 return dot((DPhiInvT * grad_b_q_k[j]), (DPhiInvT * grad_b_q_k[i])).apply() * absDetDPhi;
26 }
27 };

106 void solve() override {
107 // start a timer
108 static IpplTimings::TimerRef solve = IpplTimings::getTimer("solve");
109 IpplTimings::startTimer(solve);
110
111 const Vector<size_t, Dim> zeroNdIndex = Vector<size_t, Dim>(0);
112
113 // We can pass the zeroNdIndex here, since the transformation jacobian does not depend
114 // on translation
115 const auto firstElementVertexPoints =
116 lagrangeSpace_m.getElementMeshVertexPoints(zeroNdIndex);
117
118 // Compute Inverse Transpose Transformation Jacobian ()
119 const Vector<Tlhs, Dim> DPhiInvT =
120 refElement_m.getInverseTransposeTransformationJacobian(firstElementVertexPoints);
121
122 // Compute absolute value of the determinant of the transformation jacobian (|det D
123 // Phi_K|)
124 const Tlhs absDetDPhi = Kokkos::abs(
125 refElement_m.getDeterminantOfTransformationJacobian(firstElementVertexPoints));
126
127 EvalFunctor<Tlhs, Dim, this->lagrangeSpace_m.numElementDOFs> poissonEquationEval(
128 DPhiInvT, absDetDPhi);
129
130 // get BC type of our RHS
131 BConds<FieldRHS, Dim>& bcField = (this->rhs_mp)->getFieldBC();
132 FieldBC bcType = bcField[0]->getBCType();
133
134 const auto algoOperator = [poissonEquationEval, &bcField, this](rhs_type field) -> lhs_type {
135 // start a timer
136 static IpplTimings::TimerRef opTimer = IpplTimings::getTimer("operator");
137 IpplTimings::startTimer(opTimer);
138
139 // set appropriate BCs for the field as the info gets lost in the CG iteration
140 field.setFieldBC(bcField);
141
142 field.fillHalo();
143
144 auto return_field = lagrangeSpace_m.evaluateAx(field, poissonEquationEval);
145
146 IpplTimings::stopTimer(opTimer);
147
148 return return_field;
149 };
150
151 pcg_algo_m.setOperator(algoOperator);
152
153 // send boundary values to RHS (load vector) i.e. lifting (Dirichlet BCs)
154 if (bcType == CONSTANT_FACE) {
155 *(this->rhs_mp) = *(this->rhs_mp) -
156 lagrangeSpace_m.evaluateAx_lift(*(this->rhs_mp), poissonEquationEval);
157 }
158
159 // start a timer
160 static IpplTimings::TimerRef pcgTimer = IpplTimings::getTimer("pcg");
161 IpplTimings::startTimer(pcgTimer);
162
163 pcg_algo_m(*(this->lhs_mp), *(this->rhs_mp), this->params_m);
164
165 (this->lhs_mp)->fillHalo();
166
167 IpplTimings::stopTimer(pcgTimer);
168
169 int output = this->params_m.template get<int>("output_type");
170 if (output & Base::GRAD) {
171 *(this->grad_mp) = -grad(*(this->lhs_mp));
172 }
173
174 IpplTimings::stopTimer(solve);
175 }
