61 __global__ void transposeL2R(T* __restrict__ dst, const T* __restrict__ src, int n0,
62 int n1, int n2) {
63 int i = blockIdx.x * blockDim.x + threadIdx.x;
64 int j = blockIdx.y * blockDim.y + threadIdx.y;
65 int k = blockIdx.z * blockDim.z + threadIdx.z;
66
67 if (i < n0 && j < n1 && k < n2) {
68 size_t src_idx = i + j * n0 + k * n0 * n1; // LayoutLeft
69 size_t dst_idx = k + j * n2 + i * n1 * n2; // LayoutRight
70 dst[dst_idx] = src[src_idx];
71 }
72 }

260 void forward(complex_t* in, complex_t* out) {
261 using detail::checkCudaError;
262 using detail::checkCufftResult;
263
264 cuda_complex_t* desc_data =
265 static_cast<cuda_complex_t*>(desc_->descriptor->data[0]);
266
267 // Transpose input: LayoutLeft -> LayoutRight (into descriptor buffer)
268 launchTransposeL2R(desc_data, reinterpret_cast<cuda_complex_t*>(in));
269
270 // Execute forward FFT
271 checkCufftResult(cufftXtExecDescriptor(handle_, desc_, desc_, CUFFT_FORWARD),
272 "Forward FFT execution failed");
273
274 // Transpose output: LayoutRight -> LayoutLeft
275 launchTransposeR2L(reinterpret_cast<cuda_complex_t*>(out), desc_data);
276
277 // Apply scaling (1/N)
278 T scale = T(1) / static_cast<T>(total_elements_);
279 applyScaling(reinterpret_cast<cuda_complex_t*>(out), local_elements_, scale);
280
281 checkCudaError(cudaStreamSynchronize(stream_), "Stream sync failed");
282 }
