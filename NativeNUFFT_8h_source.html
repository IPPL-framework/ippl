117 NativeNUFFT(const Vector<size_t, Dim>& n_modes, bool use_upsampled, Config cfg = {})
118 : cfg_(cfg)
119 , kernel_(cfg.tol)
120 , n_modes_(n_modes)
121 , use_upsampled_(use_upsampled) {
122 // Compute upsampled grid sizes
123 for (unsigned d = 0; d < Dim; ++d) {
124 n_grid_[d] = std::bit_ceil<size_t>(
125 std::max<size_t>(cfg_.sigma * n_modes_[d], 2 * kernel_.width()));
126 }
127 initialized_ = false;
128 }

137 void initialize(const Layout_t& modes_layout, const MPI_Comm& comm = MPI_COMM_WORLD) {
138 if (initialized_)
139 return;
140
141 static IpplTimings::TimerRef initTimer = IpplTimings::getTimer("NativeNUFFT::init");
142 IpplTimings::startTimer(initTimer);
143
144 // Create index domain for upsampled grid
145 NDIndex<Dim> domain;
146 for (unsigned d = 0; d < Dim; ++d) {
147 domain[d] = Index(n_grid_[d]);
148 }
149
150 // Create decomposition
151 std::array<bool, Dim> isParallel;
152 isParallel.fill(true);
153
154 // For uneven kernel width W, a point in the upper half of the last
155 // segment needs both the cell value at the end and the kernel support
156 // that extends w/2 into the halo, hence (W + 1) / 2 ghost layers.
157 const int nghost = (kernel_.width()) / 2 + 1;
158
159 grid_layout_ = std::make_unique<Layout_t>(comm, domain, isParallel, true, nghost);
160
161 // Create mesh for upsampled grid
162 Vector<T, Dim> origin, hx;
163 for (unsigned d = 0; d < Dim; ++d) {
164 origin[d] = 0;
165 T extent = T(2.0) * Kokkos::numbers::pi_v<T>;
166 hx[d] = extent / n_grid_[d];
167 }
168
169 grid_mesh_ = std::make_unique<Mesh_t>(domain, hx, origin);
170
171 // Native NUFFT uses field ghosts directly instead of creating an
172 // extended grid; carry kernel-width ghost cells on the field itself.
173 grid_field_ = std::make_unique<ComplexField>(*grid_mesh_, *grid_layout_, nghost);
174
175 // Initialize heFFTe FFT
176 if (use_upsampled_) {
177 ParameterList fftParams;
178 fftParams.add("use_heffte_defaults", false);
179 fftParams.add("use_pencils", true);
180 fftParams.add("use_reorder", false);
181 fftParams.add("use_gpu_aware", true);
182 fftParams.add("comm", 3);
183 heffte_fft_ =
184 std::make_unique<FFT<CCTransform, ComplexField>>(*grid_layout_, fftParams);
185 } else {
186 ParameterList fftParams;
187 fftParams.add("use_heffte_defaults", false);
188 fftParams.add("use_pencils", true);
189 fftParams.add("use_reorder", false);
190 fftParams.add("use_gpu_aware", true);
191 fftParams.add("comm", 2);
192 fftParams.add("num_concurrent_ffts", 4);
193 PruningParams<Dim> pruning_params;
194 // Set pruning params to output the desired n_modes_, not n_grid_/2
195 for (int d = 0; d < static_cast<int>(Dim); ++d) {
196 pruning_params.n_modes[d] = n_modes_[d];
197 }
198
199 pruned_fft_ = std::make_unique<FFT<PrunedCCTransform, ComplexField>>(
200 *grid_layout_, modes_layout, pruning_params, fftParams);
201 }
202
203 // Precompute deconvolution factors
204 ESKernel<T> nufft_kernel(cfg_.tol);
205 for (unsigned d = 0; d < Dim; ++d) {
206 factors_[d] = complex_view_1d("deconv_factors", n_modes_[d]);
207
208 ippl::nufft::compute_deconvolution_factors<ExecSpace, T>(
209 factors_[d], static_cast<int64_t>(n_modes_[d]),
210 static_cast<int64_t>(n_grid_[d]), nufft_kernel);
211 }
212
213 Kokkos::fence();
214
215 initialized_ = true;
216 IpplTimings::stopTimer(initTimer);
217 }

232 template <class... Properties, typename OutField>
233 void type1(const ParticleAttrib<Vector<T, Dim>, Properties...>& R,
234 const ParticleAttrib<T, Properties...>& Q, OutField& f,
235 bool upsampled_output = false) {
236 if (!initialized_) {
237 throw IpplException("NativeNUFFT::type1",
238 "NUFFT not initialized. Call initialize() first.");
239 }
240 static IpplTimings::TimerRef NativeNUFFT1Timer = IpplTimings::getTimer("NativeNUFFT1");
241 IpplTimings::startTimer(NativeNUFFT1Timer);
242
243
244 static IpplTimings::TimerRef scatterTimer = IpplTimings::getTimer("scatterTimerNUFFT1");
245 IpplTimings::startTimer(scatterTimer);
246 Kokkos::deep_copy(grid_field_->getView(), 0.0);
247
248 auto scatter = ippl::Scatter(kernel_, cfg_.scatter_config);
249 scatter(*grid_field_, R, Q);
250 Kokkos::fence();
251 IpplTimings::stopTimer(scatterTimer);
252
253 static IpplTimings::TimerRef fftTimer = IpplTimings::getTimer("FFTNUFFT1");
254 IpplTimings::startTimer(fftTimer);
255 // Step 2: Inverse FFT
256 if (upsampled_output) {
257 performFFT(-1);
258 } else {
259 performPrunedFFT(1, f);
260 }
261 IpplTimings::stopTimer(fftTimer);
262
263 static IpplTimings::TimerRef deconvolutionTimer = IpplTimings::getTimer("deconvolutionNUFFT1");
264 IpplTimings::startTimer(deconvolutionTimer);
265 // Step 3: Deconvolution and truncation to output modes
266 if (upsampled_output) {
267 applyDeconvolutionType1<decltype(*grid_field_), decltype(f), ExecSpace, T, Dim>(
268 *grid_field_, factors_, f, n_modes_, n_grid_);
269 } else {
270 applyDeconvolutionPruned<decltype(f), ExecSpace, T, Dim>(
271 f, factors_, n_modes_, n_grid_);
272 }
273 IpplTimings::stopTimer(deconvolutionTimer);
274 IpplTimings::stopTimer(NativeNUFFT1Timer);
275
276 }

293 template <typename InField, class... Properties>
294 void type2(InField& f, const ParticleAttrib<Vector<T, Dim>, Properties...>& R,
295 ParticleAttrib<T, Properties...>& Q, bool upsampled_output = false) {
296 if (!initialized_) {
297 throw IpplException("NativeNUFFT::type2",
298 "NUFFT not initialized. Call initialize() first.");
299 }
300 static IpplTimings::TimerRef NativeNUFFT2Timer = IpplTimings::getTimer("NativeNUFFT2");
301 IpplTimings::startTimer(NativeNUFFT2Timer);
302
303 // ============================================================
304 // Step 1: Apply pre-correction
305 // ============================================================
306 static IpplTimings::TimerRef PrecorrectionTimer = IpplTimings::getTimer("PrecorrectionNUFFT2");
307 IpplTimings::startTimer(PrecorrectionTimer);
308
309 if (upsampled_output) {
310 applyPreCorrectionType2<decltype(f), decltype(*grid_field_), ExecSpace, T, Dim>(
311 f, factors_, *grid_field_, n_modes_, n_grid_);
312 } else {
313 applyPrecorrectionPruned<decltype(f), ExecSpace, T, Dim>(
314 f, factors_, n_modes_, n_grid_);
315 }
316 IpplTimings::stopTimer(PrecorrectionTimer);
317
318 Kokkos::fence();
319
320 // ============================================================
321 // Step 2: Inverse FFT
322 // ============================================================
323 static IpplTimings::TimerRef FFTTimer = IpplTimings::getTimer("FFTNUFFT2");
324 IpplTimings::startTimer(FFTTimer);
325 if (upsampled_output) {
326 performFFT(-1); // operates on grid_field_
327 } else {
328 performPrunedFFT(-1, f); // writes into grid_field_
329 }
330 IpplTimings::stopTimer(FFTTimer);
331 Kokkos::fence();
332
333 // ============================================================
334 // Step 3: Gather/interpolate at particle positions
335 // ============================================================
336 static IpplTimings::TimerRef GatherTimer = IpplTimings::getTimer("GatherNUFFT2");
337 IpplTimings::startTimer(GatherTimer);
338 auto gather = ippl::Gather(kernel_, cfg_.gather_config);
339 gather(*grid_field_, R, Q);
340 IpplTimings::stopTimer(GatherTimer);
341 Kokkos::fence();
342 IpplTimings::stopTimer(NativeNUFFT2Timer);
343 }

363 void performFFT(int sign) {
364 TransformDirection direction = (sign < 0) ? BACKWARD : FORWARD;
365 heffte_fft_->transform(direction, *grid_field_);
366 }
367
373 void performPrunedFFT(int sign, auto& output_field) {
374 TransformDirection direction = (sign < 0) ? BACKWARD : FORWARD;
375 if (sign < 0) {
376 pruned_fft_->transform(direction, output_field, *grid_field_, 1);
377 } else {
378 pruned_fft_->transform(direction, *grid_field_, output_field, -1);
379 }
