67 // Defensive backend clamp: a CSV preset tagged for sm_xx may
68 // be loaded on a build where the test fixture also instantiates
69 // host execution spaces. Force team_size=1 there so we never
70 // construct a TeamPolicy<Serial> with team_size > 1, which
71 // aborts with "Requested Team Size is too large!". Scatter.h

92 template <typename GridValueT>
93 struct Histogram {
94 static constexpr bool is_complex =
95 std::is_same_v<GridValueT, Kokkos::complex<RealType>>;
96
97 scratch_view data_r;
98 scratch_view data_i; // only used if is_complex
99 Vector<int, Dim> size;

101
102 KOKKOS_INLINE_FUNCTION void init(const team_member& team, scratch_view r,
103 scratch_view i, Vector<int, Dim> sz, size_t n) {
104 data_r = r;
105 if constexpr (is_complex)
106 data_i = i;
107 size = sz;
108 total = n;
109
110 Kokkos::parallel_for(Kokkos::TeamThreadRange(team, total), [&](size_t idx) {
111 data_r(idx) = 0;
112 if constexpr (is_complex)
113 data_i(idx) = 0;
114 });
115 }

117 KOKKOS_INLINE_FUNCTION size_t to_flat(const Kokkos::Array<int, Dim>& c) const {
118 size_t idx = 0, stride = 1;
119 for (unsigned d = 0; d < Dim; ++d) {
120 idx += static_cast<size_t>(c[d]) * stride;
121 stride *= static_cast<size_t>(size[d]);
122 }
123 return idx;
124 }
125
126 KOKKOS_INLINE_FUNCTION Kokkos::Array<int, Dim> from_flat(size_t idx) const {
127 Kokkos::Array<int, Dim> c{};
128 for (unsigned d = 0; d < Dim; ++d) {
129 c[d] = static_cast<int>(idx % static_cast<size_t>(size[d]));
130 idx /= static_cast<size_t>(size[d]);
131 }
132 return c;
133 }
134
135 KOKKOS_INLINE_FUNCTION bool in_bounds(const Kokkos::Array<int, Dim>& c) const {
136 for (unsigned d = 0; d < Dim; ++d) {
137 if (c[d] < 0 || c[d] >= size[d])
138 return false;
139 }
140 return true;
141 }
142
143 template <typename V>
144 KOKKOS_INLINE_FUNCTION void atomic_add(const Kokkos::Array<int, Dim>& c, const V& v,
145 RealType w) {
146 if (!in_bounds(c))
147 return;
148 const size_t idx = to_flat(c);
149
150 if constexpr (is_complex && std::is_same_v<V, Kokkos::complex<RealType>>) {
151 Kokkos::atomic_add(&data_r(idx), v.real() * w);
152 Kokkos::atomic_add(&data_i(idx), v.imag() * w);
153 } else if constexpr (is_complex) {
154 Kokkos::atomic_add(&data_r(idx), v * w);
155 } else {
156 Kokkos::atomic_add(&data_r(idx), v * w);
157 }
158 }
159
160 KOKKOS_INLINE_FUNCTION GridValueT operator()(size_t idx) const {
161 if constexpr (is_complex)
162 return GridValueT(data_r(idx), data_i(idx));
163 else
164 return data_r(idx);
165 }
166 };

168 KOKKOS_INLINE_FUNCTION Vector<int, Dim> hist_size() const {
169 Vector<int, Dim> hs;
170 for (unsigned d = 0; d < Dim; ++d)
171 hs[d] = args.tile_size[d] + W + 1;
172 return hs;
173 }
174
175 KOKKOS_INLINE_FUNCTION size_t hist_total() const {
176 size_t n = 1;
177 for (unsigned d = 0; d < Dim; ++d)
178 n *= static_cast<size_t>(args.tile_size[d] + W + 1);
179 return n;
180 }
181
182 KOKKOS_INLINE_FUNCTION Vector<int, Dim> decode_tile_base(size_t tile_id) const {
183 Vector<int, Dim> tile_base;
184 for (size_t t = tile_id, d = Dim; d-- > 0;) {
185 tile_base[d] = static_cast<int>(t % static_cast<size_t>(args.num_tiles[d]))
186 * args.tile_size[d];
187 t /= static_cast<size_t>(args.num_tiles[d]);
188 }
189 return tile_base;
190 }

192 KOKKOS_INLINE_FUNCTION void operator()(const team_member& team) const {
193 using grid_value_t = typename decltype(args.grid)::non_const_value_type;
194
195 const size_t tile_id = team.league_rank();
196 const auto tile_base = decode_tile_base(tile_id);
197
198 const auto hs = hist_size();
199 const size_t total = hist_total();
200
201 // Per-team scratch allocation: bump allocator semantics (your point).
202 scratch_view scratch_r(team.team_scratch(0), total);
203 scratch_view scratch_i;
204 if constexpr (Histogram<grid_value_t>::is_complex) {
205 scratch_i = scratch_view(team.team_scratch(0), total);
206 }
207
208 Histogram<grid_value_t> hist;
209 hist.init(team, scratch_r, scratch_i, hs, total);
210 team.team_barrier();
211
212 // Hoist transform: same for all particles.
213 const CoordinateTransform<RealType, Dim> transform{args.origin, args.invdx,
214 args.n_grid};
215
216 const size_t pstart = args.bin_offsets(tile_id);
217 const size_t pend = args.bin_offsets(tile_id + 1);
218
219 Kokkos::parallel_for(Kokkos::TeamThreadRange(team, pstart, pend), [&](size_t ip) {
220 const size_t p = args.permute(ip);
221 const auto val = args.values(p);
222
223 Stencil stencil{};
224
225 for_constexpr(std::make_integer_sequence<int, Dim>{}, [&]<int d>() {
226 const RealType g = transform.toGridCoordinate(args.x(p)[d], d);
227 const int idx0 = transform.getStencilBase(g - RealType(0.5), W);
228
229 stencil.base[d] = idx0 - args.local_offset[d];
230
231 for (int i = 0; i < W; ++i) {
232 stencil.kw[d][i] = args.kernel((g - (RealType(idx0 + i) + RealType(0.5))) * args.inv_hw);
233 }
234 });
235
236 auto scatter = [&]<unsigned D>(auto&& self, RealType wprod,
237 Kokkos::Array<int, Dim> hc) {
238 for (int i = 0; i < W; ++i) {
239 hc[D] = stencil.base[D] + i + half_left - tile_base[D];
240 const RealType w = wprod * stencil.kw[D][i];
241
242 if constexpr (D == 0) {
243 hist.atomic_add(hc, val, w);
244 } else {
245 self.template operator()<D - 1>(self, w, hc);
246 }
247 }
248 };
249 scatter.template operator()<Dim - 1>(scatter, RealType(1), {});
250 });
251
252 team.team_barrier();
253
254 // Flush histogram to grid
255 Kokkos::parallel_for(Kokkos::TeamThreadRange(team, total), [&](size_t idx) {
256 const auto hc = hist.from_flat(idx);
257
258 Kokkos::Array<int, Dim> gc{};
259 for (unsigned d = 0; d < Dim; ++d) {
260 const int local = tile_base[d] + hc[d] - half_left;
261 if (local < -args.nghost || local >= args.n_grid_local[d] + args.nghost)
262 return;
263 gc[d] = local + args.nghost;
264 }
265
266 [&]<std::size_t... Is>(std::index_sequence<Is...>) {
267 // There is some bug in Kokkos 5.0.0 that leads to deadlocks with complex
268 // atomics and that I was unable of reproducing in a MWE
269 if constexpr (Histogram<grid_value_t>::is_complex) {
270 RealType* ptr = reinterpret_cast<RealType*>(&args.grid(gc[Is]...));
271 Kokkos::atomic_add(&ptr[0], hist.data_r(idx));
272 Kokkos::atomic_add(&ptr[1], hist.data_i(idx));
273 } else {
274 Kokkos::atomic_add(&args.grid(gc[Is]...),
275 static_cast<grid_value_t>(hist(idx)));
276 }
277 }(std::make_index_sequence<Dim>{});
278 });
279 }

281 template <bool IsComplex>
282 static size_t compute_scratch_size(const Vector<int, Dim>& tile_size, int /* team_size */,
283 int /* z_batches */ = 1) {
284 size_t n = 1;
285 for (unsigned d = 0; d < Dim; ++d)
286 n *= static_cast<size_t>(tile_size[d] + W + 1);
287 // Use shmem_size() to account for Kokkos alignment overhead
288 return (IsComplex ? 2 : 1) * scratch_view::shmem_size(n);
289 }

291 void run(size_t) {
292 using grid_value_t = typename decltype(args.grid)::non_const_value_type;
293 constexpr bool cplx = std::is_same_v<grid_value_t, Kokkos::complex<RealType>>;
294 const size_t scratch = compute_scratch_size<cplx>(args.tile_size, 0);
295
296 size_t n_tiles = 1;
297 for (unsigned d = 0; d < Dim; ++d)
298 n_tiles *= static_cast<size_t>(args.num_tiles[d]);
299
300 Kokkos::parallel_for(
301 "TiledScatter",
302 team_policy(n_tiles, args.team_size).set_scratch_size(0, Kokkos::PerTeam(scratch)),
303 *this);
304 }
