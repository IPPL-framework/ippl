142 template <class Buffer, typename Archive>
143 void Communicate::recv(int src, int tag, Buffer& buffer, Archive& ar, size_type msize,
144 size_type nrecvs) {
145 // Temporary fix. MPI communication seems to have problems when the
146 // count argument exceeds the range of int, so large messages should
147 // be split into smaller messages
148 if (msize > INT_MAX) {
149 std::cerr << "Message size exceeds range of int" << std::endl;
150 this->abort();
151 }
152 MPI_Status status;
153 MPI_Recv(ar.getBuffer(), msize, MPI_BYTE, src, tag, comm_m, &status);
154
155 buffer.deserialize(ar, nrecvs);
156 }

158 template <class Buffer, typename Archive>
159 void Communicate::isend(int dest, int tag, Buffer& buffer, Archive& ar, MPI_Request& request,
160 size_type nsends) {
161 if (ar.getBufferSize() > INT_MAX) {
162 std::cerr << "Message size exceeds range of int" << std::endl;
163 this->abort();
164 }
165 buffer.serialize(ar, nsends);
166 MPI_Isend(ar.getBuffer(), ar.getSize(), MPI_BYTE, dest, tag, comm_m, &request);
167 }
