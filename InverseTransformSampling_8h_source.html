116 void updateBounds(Vector<T, Dim>& rmax, Vector<T, Dim>& rmin, Vector<T, Dim>& locrmax, Vector<T, Dim>& locrmin) {
117
118 int rank = ippl::Comm->rank();
119 Vector<T, Dim> nr_m, dr_m;
120 for (unsigned d = 0; d < Dim; ++d) {
121 nr_m[d] = dist_m.getCdf(locrmax[d], d) - dist_m.getCdf(locrmin[d], d);
122 dr_m[d] = dist_m.getCdf(rmax[d], d) - dist_m.getCdf(rmin[d], d);
123 umin_m[d] = dist_m.getCdf(locrmin[d], d);
124 umax_m[d] = dist_m.getCdf(locrmax[d], d);
125 }
126 T pnr = std::accumulate(nr_m.begin(), nr_m.end(), 1.0, std::multiplies<T>());
127 T pdr = std::accumulate(dr_m.begin(), dr_m.end(), 1.0, std::multiplies<T>());
128
129 T factor = pnr / pdr;
130 nlocal_m = (size_type)(factor * ntotal_m);
131
132 size_type nglobal = 0;
133 ippl::Comm->allreduce(&nlocal_m, &nglobal, 1, std::plus<size_type>());
134
135 int rest = (int)(ntotal_m - nglobal);
136 if (rank < rest) {
137 ++nlocal_m;
138 }
139 }

205 typename GeneratorPool::generator_type rand_gen = pool_m.get_state();

207 value_type u = 0.0;
