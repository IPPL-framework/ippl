52 const typename RegionLayout::host_mirror_type regions = rlayout_r.gethLocalRegions();
53
54 int rank = ippl::Comm->rank();
55 Vector<T, Dim> nr_m, dr_m;
56 for (unsigned d = 0; d < Dim; ++d) {
57 nr_m[d] = dist_m.getCdf(regions(rank)[d].max(), d) - dist_m.getCdf(regions(rank)[d].min(),d);
58 dr_m[d] = dist_m.getCdf(rmax_m[d],d) - dist_m.getCdf(rmin_m[d],d);
59 umin_m[d] = dist_m.getCdf(regions(rank)[d].min(),d);
60 umax_m[d] = dist_m.getCdf(regions(rank)[d].max(),d);
61 }
62 T pnr = std::accumulate(nr_m.begin(), nr_m.end(), 1.0, std::multiplies<T>());
63 T pdr = std::accumulate(dr_m.begin(), dr_m.end(), 1.0, std::multiplies<T>());
64
65 T factor = pnr / pdr;
66 nlocal_m = (size_type) (factor * ntotal_m);
67 size_type nglobal = 0;
68
69 MPI_Allreduce(&nlocal_m, &nglobal, 1, MPI_UNSIGNED_LONG, MPI_SUM,
70 ippl::Comm->getCommunicator());
71
72 int rest = (int)(ntotal_m - nglobal);
73
74 if (rank < rest) {
75 ++nlocal_m;
76 }
