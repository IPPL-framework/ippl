220 void LeapFrogStep(){
221 // LeapFrog time stepping https://en.wikipedia.org/wiki/Leapfrog_integration
222 // Here, we assume a constant charge-to-mass ratio of -1 for
223 // all the particles hence eliminating the need to store mass as
224 // an attribute
225 static IpplTimings::TimerRef PTimer = IpplTimings::getTimer("pushVelocity");
226 static IpplTimings::TimerRef RTimer = IpplTimings::getTimer("pushPosition");
227 static IpplTimings::TimerRef updateTimer = IpplTimings::getTimer("update");
228 static IpplTimings::TimerRef domainDecomposition = IpplTimings::getTimer("loadBalance");
229 static IpplTimings::TimerRef SolveTimer = IpplTimings::getTimer("solve");
230
231 double dt = this->dt_m;
232 std::shared_ptr<ParticleContainer_t> pc = this->pcontainer_m;
233 std::shared_ptr<FieldContainer_t> fc = this->fcontainer_m;
234
235 IpplTimings::startTimer(PTimer);
236 pc->P = pc->P - 0.5 * dt * pc->E;
237 IpplTimings::stopTimer(PTimer);
238
239 // drift
240 IpplTimings::startTimer(RTimer);
241 pc->R = pc->R + dt * pc->P;
242 IpplTimings::stopTimer(RTimer);
243
244 // Since the particles have moved spatially update them to correct processors
245 IpplTimings::startTimer(updateTimer);
246 pc->update();
247 IpplTimings::stopTimer(updateTimer);
248
249 size_type totalP = this->totalP_m;
250 int it = this->it_m;
251 bool isFirstRepartition = false;
252 if (this->loadbalancer_m->balance(totalP, it + 1)) {
253 IpplTimings::startTimer(domainDecomposition);
254 auto* mesh = &fc->getRho().get_mesh();
255 auto* FL = &fc->getFL();
256 this->loadbalancer_m->repartition(FL, mesh, isFirstRepartition);
257 IpplTimings::stopTimer(domainDecomposition);
258 }
259
260 // scatter the charge onto the underlying grid
261 this->par2grid();
262
263 // Field solve
264 IpplTimings::startTimer(SolveTimer);
265 this->fsolver_m->runSolver();
266 IpplTimings::stopTimer(SolveTimer);
267
268 // gather E field
269 this->grid2par();
270
271 // kick
272 IpplTimings::startTimer(PTimer);
273 pc->P = pc->P - 0.5 * dt * pc->E;
274 IpplTimings::stopTimer(PTimer);
275 }
