21 /*
22 * Packs a sparse selection of scalar particle-attribute entries into
23 * the archive buffer. `hash(i)` gives the source index in `view_data`
24 * for the i-th item being sent; the functor copies that value as raw
25 * bytes to `buf + wpos + i * sizeof(T)`. This is used by
26 * Archive::serialize(view, hash, nsends) during particle migration,
27 * where only particles listed in the send hash are serialized.
28 */
29 template <typename T, typename HashView, typename BufferPtr>
30 struct SerializeHashFunctor {
31 const T* view_data;
32 HashView hash;
33 BufferPtr buf;
34 size_t elem_size;
35 size_t wpos;
36
37 KOKKOS_INLINE_FUNCTION void operator()(const size_t i) const {
38 const char* src = reinterpret_cast<const char*>(view_data + hash(i));
39 char* dst = buf + i * elem_size + wpos;
40 copyBytes(dst, src, elem_size);
41 }
42 };
43
44 template <typename T, unsigned Dim, typename HashView, typename BufferPtr>
45 struct SerializeHashVectorFunctor {
46 const Vector<T, Dim>* view_data;
47 HashView hash;
48 BufferPtr buf;
49 size_t elem_size;
50 size_t wpos;
51
52 KOKKOS_INLINE_FUNCTION void operator()(const size_t i, const size_t d) const {
53 const Vector<T, Dim>* vec = view_data + hash(i);
54 const T value = (*vec)[d];
55 const char* src = reinterpret_cast<const char*>(&value);
56 char* dst = buf + (Dim * i + d) * elem_size + wpos;
57 copyBytes(dst, src, elem_size);
58 }
59 };
