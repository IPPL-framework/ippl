55 template <class... PolicyArgs, typename View>
56 typename RangePolicy<View::rank, typename View::execution_space, PolicyArgs...>::policy_type
57 getRangePolicy(const View& view, int shift = 0) {
58 constexpr unsigned Dim = View::rank;
59 using exec_space = typename View::execution_space;
60 using policy_type = typename RangePolicy<Dim, exec_space, PolicyArgs...>::policy_type;
61 if constexpr (Dim == 1) {
62 return policy_type(shift, view.size() - shift);
63 } else {
64 using index_type = typename RangePolicy<Dim, exec_space, PolicyArgs...>::index_type;
65 Kokkos::Array<index_type, Dim> begin, end;
66 for (unsigned int d = 0; d < Dim; d++) {
67 begin[d] = shift;
68 end[d] = view.extent(d) - shift;
69 }
70 return policy_type(begin, end);
71 }
72 // Silences nvcc "missing return" at end of exhaustive if constexpr.
73 __builtin_unreachable();
74 }

127 template <int Dim, int W, typename Team, typename F>
128 KOKKOS_FORCEINLINE_FUNCTION void thread_vector_stencil_for(const Team& team, F&& f) {
129 constexpr int TotalCells = [] {
130 int r = 1;
131 for (int i = 0; i < Dim; ++i)
132 r *= W;
133 return r;
134 }();
135
136 Kokkos::parallel_for(Kokkos::ThreadVectorRange(team, TotalCells), [&](const int flat_idx) {
137 // Compile-time unrollable index decomposition
138 Kokkos::Array<int, Dim> idx;
139 int tmp = flat_idx;
140 for (int d = Dim - 1; d >= 0; --d) {
141 idx[d] = tmp % W;
142 tmp /= W;
143 }
144
145 // Call with index pack
146 [&]<int... Is>(std::integer_sequence<int, Is...>) {
147 f(idx[Is]...);
148 }(std::make_integer_sequence<int, Dim>{});
149 });

193 // Dispatches F(i) for i in [0, n) either in parallel (OpenMP host)
194 // when MPI_THREAD_MULTIPLE is available, or serially otherwise.
195 // `f(i)` must be synchronous w.r.t. its own work — this dispatcher
196 // fences after the parallel_for so the caller's next operation sees
197 // a consistent state.
198 template <typename F>
199 void parallelForMPI(size_t n, F&& f) {
200 constexpr bool useGPU = isGPUSpace<Kokkos::DefaultExecutionSpace>;
201 const bool threadSafe = Env->threadMultiple();
202
203 if constexpr (useGPU) {
204 if (threadSafe) {
205 Kokkos::parallel_for(
206 "Parallel dispatch",
207 Kokkos::RangePolicy<Kokkos::DefaultHostExecutionSpace>(0, n), [&](int i) {
208 f(i);
209 });
210 Kokkos::fence();
211 return;
212 }
213 }
214 for (size_t i = 0; i < n; ++i) {
215 f(i);
216 }
217 }
