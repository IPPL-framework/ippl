24 //=========================================================================
25 // Pruned Real-to-Complex Transform
26 //
27 // The key insight that makes this completely communication-free per
28 // transform is the choice of heFFTe outbox:
29 //
30 // R2C dim (d_r):
31 // Each rank's outbox lower bound in d_r = pruned field lower bound.
32 // The last rank (hi_pruned == K_r - 1) extends its outbox to cover
33 // [K_r, N_r/2], which are the "extra" full-complex modes discarded by
34 // the pruning. This ensures every rank's outbox contains exactly the
35 // full-complex indices its pruned modes map to (fi = gi, direct).
36 //
37 // Non-R2C dims:
38 // Every rank owns the FULL extent [0, N_d - 1]. The wrapping formula
39 // fj = gj < K/2 ? gj : N-K+gj always maps into [0, N-1], which is
40 // always within this rank's outbox.
41 //
42 // Result: the local index in tempComplexFull_ is computable from the
43 // pruned global index with no inter-rank communication.
44 //=========================================================================
45

128 template <typename RealField>
129 FFT<PrunedRCTransform, RealField>::FFT(const Layout_t& layoutReal,
130 const Layout_t& /*layoutComplexFull*/,
131 const Layout_t& layoutComplexPruned,
132 const PruningParams<Dim>& pruning,
133 const ParameterList& params)
134 : pruning_(pruning) {
135 static_assert(Dim == 3, "PrunedRCTransform currently only supports 3D");
136
137 r2c_dir_ = params.get<int>("r2c_direction", 0);
138
139 // Global real-grid sizes
140 const auto& gDomReal = layoutReal.getDomain();
141 for (int d = 0; d < 3; ++d)
142 globalRealDims_[d] = gDomReal[d].length();
143
144 // Inbox: this rank's local real slab
145 std::array<long long, 3> lowReal, highReal;
146 fft::domainToBounds<Dim>(layoutReal.getLocalNDIndex(), lowReal, highReal);
147
148 // Outbox: custom -- aligned to pruned in r2c_dir, full extent everywhere else
149 //
150 // In r2c_dir d_r:
151 // outbox lo = pruned lo (so fi = gi_p is always within this rank's buffer)
152 // outbox hi = pruned hi (except the last rank, which absorbs [K_r, N_r/2])
153 //
154 // In every other dim d:
155 // outbox = [0, N_d - 1] (whole extent; wrapping formula always stays local)
156 const auto& lDomPruned = layoutComplexPruned.getLocalNDIndex();
157
158 std::array<long long, 3> lowOut, highOut;
159 for (int d = 0; d < 3; ++d) {
160 if (d == r2c_dir_) {
161 const long long lo_p = lDomPruned[d].first();
162 const long long hi_p = lDomPruned[d].last();
163 const long long K_d = static_cast<long long>(pruning_.n_modes[d]);
164 const long long N_d = globalRealDims_[d];
165 const long long top = N_d / 2; // last valid index of R2C output (N/2+1 elements)
166
167 lowOut[d] = lo_p;
168 // If this rank owns the last pruned mode, extend outbox to cover
169 // all remaining full-complex elements [K_d .. N_d/2]
170 highOut[d] = (hi_p == K_d - 1) ? top : hi_p;
171 } else {
172 // Every rank owns the full extent of this dimension
173 lowOut[d] = 0;
174 highOut[d] = globalRealDims_[d] - 1;
175 }
176 }
177
178 heffte::box3d<long long> inbox{lowReal, highReal};
179 heffte::box3d<long long> outbox{lowOut, highOut};
180
181 backend_ =
182 std::make_unique<Backend_t>(inbox, outbox, r2c_dir_, Comm->getCommunicator(), params);
183
184 for (int d = 0; d < 3; ++d) {
185 lowComplexFull_[d] = lowOut[d];
186 fullComplexDims_[d] = static_cast<std::size_t>(highOut[d] - lowOut[d] + 1);
187 }
188 }

194 template <typename RealField>
195 void FFT<PrunedRCTransform, RealField>::transform(TransformDirection direction, RealField& f,
196 ComplexField& g) {
197 auto fview = f.getView();
198 auto gview = g.getView();
199 const int ngf = f.getNghost();
200 const int ngg = g.getNghost();
201
202 // Ensure temp buffers
203 if (tempReal_.size() != f.getOwned().size())
204 tempReal_ = detail::shrinkView("pruned_r2c_real", fview, ngf);
205
206 const std::size_t fullSize =
207 fullComplexDims_[0] * fullComplexDims_[1] * fullComplexDims_[2];
208 if (tempComplexFull_.size() != fullSize)
209 tempComplexFull_ = TempComplex_t("pruned_r2c_complex_full", fullComplexDims_[0],
210 fullComplexDims_[1], fullComplexDims_[2]);
211
212 // Pruned domain info for the kernel
213 const auto& lDomPruned = g.getLayout().getLocalNDIndex();
214 const long long lp0 = lDomPruned[0].first();
215 const long long lp1 = lDomPruned[1].first();
216 const long long lp2 = lDomPruned[2].first();
217
218 // Mode counts and real-grid sizes for the wrapping formula
219 const long long K0 = pruning_.n_modes[0];
220 const long long K1 = pruning_.n_modes[1];
221 const long long K2 = pruning_.n_modes[2];
222 const long long N0 = globalRealDims_[0];
223 const long long N1 = globalRealDims_[1];
224 const long long N2 = globalRealDims_[2];
225
226 // outbox origins -- only non-zero in r2c_dir
227 const long long lcf0 = lowComplexFull_[0];
228 const long long lcf1 = lowComplexFull_[1];
229 const long long lcf2 = lowComplexFull_[2];
230 const int r2c = r2c_dir_;
231
232 auto owned = g.getOwned(); // ghost-free extent of pruned field
233
234 if (direction == FORWARD) {
235 // 1. Strip ghosts -> tempReal_
236 auto tempreal = tempReal_;
237 Kokkos::parallel_for(
238 "r2c_copy_real_fwd",
239 Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<3>>(
240 {ngf, ngf, ngf}, {int(fview.extent(0)) - ngf, int(fview.extent(1)) - ngf,
241 int(fview.extent(2)) - ngf}),
242 KOKKOS_LAMBDA(int i, int j, int k) {
243 tempreal(i - ngf, j - ngf, k - ngf) = fview(i, j, k);
244 });
245 Kokkos::fence();
246
247 // 2. Distributed R2C FFT -> tempComplexFull_
248 backend_->forward(tempReal_.data(), tempComplexFull_.data());
249
250 // 3. Extract pruned modes from tempComplexFull_ -> pruned output field
251 //
252 // Every access is local by construction of the outbox:
253 // fi0_l = gi0 - lcf0 (R2C dim 0 assumed here; generalises below)
254 // fi1_l = wrap(gi1) (lcf1=0 for non-R2C dims)
255 // fi2_l = wrap(gi2)
256 auto& tcf = tempComplexFull_;
257 Kokkos::parallel_for(
258 "extract_pruned_r2c_fwd",
259 Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<3>>(
260 {0, 0, 0},
261 {int(owned[0].length()), int(owned[1].length()), int(owned[2].length())}),
262 KOKKOS_LAMBDA(int i0, int i1, int i2) {
263 const long long gi0 = i0 + lp0;
264 const long long gi1 = i1 + lp1;
265 const long long gi2 = i2 + lp2;
266
267 const int fi0 =
268 (r2c == 0) ? int(gi0 - lcf0) : int((gi0 < K0 / 2) ? gi0 : (N0 - K0 + gi0));
269 const int fi1 =
270 (r2c == 1) ? int(gi1 - lcf1) : int((gi1 < K1 / 2) ? gi1 : (N1 - K1 + gi1));
271 const int fi2 =
272 (r2c == 2) ? int(gi2 - lcf2) : int((gi2 < K2 / 2) ? gi2 : (N2 - K2 + gi2));
273
274 gview(i0 + ngg, i1 + ngg, i2 + ngg) = tcf(fi0, fi1, fi2);
275 });
276
277 } else { // BACKWARD
278
279 // 1. Zero-fill tempComplexFull_, then scatter pruned modes into it
280 Kokkos::deep_copy(tempComplexFull_, Complex_t(0, 0));
281
282 auto& tcf = tempComplexFull_;
283 Kokkos::parallel_for(
284 "scatter_pruned_r2c_bwd",
285 Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<3>>(
286 {0, 0, 0},
287 {int(owned[0].length()), int(owned[1].length()), int(owned[2].length())}),
288 KOKKOS_LAMBDA(int i0, int i1, int i2) {
289 const long long gi0 = i0 + lp0;
290 const long long gi1 = i1 + lp1;
291 const long long gi2 = i2 + lp2;
292
293 const int fi0 =
294 (r2c == 0) ? int(gi0 - lcf0) : int((gi0 < K0 / 2) ? gi0 : (N0 - K0 + gi0));
295 const int fi1 =
296 (r2c == 1) ? int(gi1 - lcf1) : int((gi1 < K1 / 2) ? gi1 : (N1 - K1 + gi1));
297 const int fi2 =
298 (r2c == 2) ? int(gi2 - lcf2) : int((gi2 < K2 / 2) ? gi2 : (N2 - K2 + gi2));
299
300 tcf(fi0, fi1, fi2) = gview(i0 + ngg, i1 + ngg, i2 + ngg);
301 });
302 Kokkos::fence();
303
304 // 2. Distributed C2R backward
305 backend_->backward(tempComplexFull_.data(), tempReal_.data());
306 Kokkos::fence();
307
308 // 3. Copy tempReal_ back (restore ghost padding)
309 auto tempreal = tempReal_;
310 Kokkos::parallel_for(
311 "r2c_copy_real_bwd",
312 Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<3>>(
313 {ngf, ngf, ngf}, {int(fview.extent(0)) - ngf, int(fview.extent(1)) - ngf,
314 int(fview.extent(2)) - ngf}),
315 KOKKOS_LAMBDA(int i, int j, int k) {
316 fview(i, j, k) = tempreal(i - ngf, j - ngf, k - ngf);
317 });
318 }
319 }
