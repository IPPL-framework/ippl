60 // Pack a 3D view region into a linear buffer via the real-part of each
61 // cell (FFT-solver semantics: tolerates real or Kokkos::complex
62 // views; discards imaginary parts of complex views).
63 template <typename Tb, typename Tf>
64 inline void pack(const ippl::NDIndex<3> intersect, Kokkos::View<Tf***>& view,
65 ippl::detail::FieldBufferData<Tb>& fd, int nghost,
66 const ippl::NDIndex<3> ldom,
67 ippl::mpi::Communicator::size_type& nsends) {
68 Kokkos::View<Tb*>& buffer = fd.buffer;
69
70 size_t size = intersect.size();
71 nsends = size;
72 if (buffer.size() < size) {
73 const int overalloc = ippl::Comm->getDefaultOverallocation();
74 Kokkos::realloc(buffer, size * overalloc);
75 }
76
77 const int first0 = intersect[0].first() + nghost - ldom[0].first();
78 const int first1 = intersect[1].first() + nghost - ldom[1].first();
79 const int first2 = intersect[2].first() + nghost - ldom[2].first();
80
81 const int last0 = intersect[0].last() + nghost - ldom[0].first() + 1;
82 const int last1 = intersect[1].last() + nghost - ldom[1].first() + 1;
83 const int last2 = intersect[2].last() + nghost - ldom[2].first() + 1;
84
85 using mdrange_type = Kokkos::MDRangePolicy<Kokkos::Rank<3>>;
86 Kokkos::parallel_for(
87 "ippl::detail::pack()",
88 mdrange_type({first0, first1, first2},
89 {(long int)last0, (long int)last1, (long int)last2}),
90 KOKKOS_LAMBDA(const size_t i, const size_t j, const size_t k) {
91 const int ig = i - first0;
92 const int jg = j - first1;
93 const int kg = k - first2;
94
95 int l = ig + jg * intersect[0].length()
96 + kg * intersect[1].length() * intersect[0].length();
97
98 Kokkos::complex<Tb> val = view(i, j, k);
99 buffer(l) = Kokkos::real(val);
100 });
101 Kokkos::fence();
102 }

147 // Unpack a linear buffer into a 3D view region, with per-axis conditional
148 // reflection of the buffer index. Setting x / y / z to true reverses the
149 // buffer ordering along that axis as it is placed into the view — the
150 // primitive operation behind `mirrorField` and behind the Vico solver's
151 // reflected-quadrant assembly.
152 template <int tensorRank, typename Tb, typename Tf>
153 inline void unpack_impl(const ippl::NDIndex<3> intersect,
154 const Kokkos::View<Tf***>& view,
155 ippl::detail::FieldBufferData<Tb>& fd, int nghost,
156 const ippl::NDIndex<3> ldom, size_t dim1 = 0, size_t dim2 = 0,
157 bool x = false, bool y = false, bool z = false) {
158 Kokkos::View<Tb*>& buffer = fd.buffer;
159
160 const int first0 = intersect[0].first() + nghost - ldom[0].first();
161 const int first1 = intersect[1].first() + nghost - ldom[1].first();
162 const int first2 = intersect[2].first() + nghost - ldom[2].first();
163
164 const int last0 = intersect[0].last() + nghost - ldom[0].first() + 1;
165 const int last1 = intersect[1].last() + nghost - ldom[1].first() + 1;
166 const int last2 = intersect[2].last() + nghost - ldom[2].first() + 1;
167
168 using mdrange_type = Kokkos::MDRangePolicy<Kokkos::Rank<3>>;
169 Kokkos::parallel_for(
170 "ippl::detail::unpack_impl()",
171 mdrange_type({first0, first1, first2}, {last0, last1, last2}),
172 KOKKOS_LAMBDA(const size_t i, const size_t j, const size_t k) {
173 int ig = i - first0;
174 int jg = j - first1;
175 int kg = k - first2;
176
177 ig = x * (intersect[0].length() - 2 * ig - 1) + ig;
178 jg = y * (intersect[1].length() - 2 * jg - 1) + jg;
179 kg = z * (intersect[2].length() - 2 * kg - 1) + kg;
180
181 int l = ig + jg * intersect[0].length()
182 + kg * intersect[1].length() * intersect[0].length();
183
184 ippl::detail::ViewAccess<tensorRank, decltype(view)>::get(view, dim1, dim2, i,
185 j, k) = buffer(l);
186 });
187 Kokkos::fence();
188 }

215 // Async MPI_Isend wrapper: pack the intersection into a device-resident
216 // buffer (picked up from the view's memory space via Comm->getBuffer),
217 // issue Comm->isend, and append the resulting request to `requests`.
218 template <typename Tb, typename Tf, unsigned Dim>
219 inline void solver_send(int TAG, int id, int i, const ippl::NDIndex<Dim> intersection,
220 const ippl::NDIndex<Dim> ldom, int nghost,
221 Kokkos::View<Tf***>& view,
222 ippl::detail::FieldBufferData<Tb>& fd,
223 std::vector<MPI_Request>& requests) {
224 using memory_space = typename Kokkos::View<Tf***>::memory_space;
225
226 requests.resize(requests.size() + 1);
227
228 ippl::mpi::Communicator::size_type nsends;
229 pack(intersection, view, fd, nghost, ldom, nsends);
230
231 ippl::mpi::Communicator::buffer_type<memory_space> buf =
232 ippl::Comm->getBuffer<memory_space, Tf>(nsends);
233
234 int tag = TAG + id;
235
236 ippl::Comm->isend(i, tag, fd, *buf, requests.back(), nsends);
237 buf->resetWritePos();
238 }

240 template <typename Tb, typename Tf, unsigned Dim>
241 inline void solver_recv(int TAG, int id, int i, const ippl::NDIndex<Dim> intersection,
242 const ippl::NDIndex<Dim> ldom, int nghost,
243 Kokkos::View<Tf***>& view,
244 ippl::detail::FieldBufferData<Tb>& fd, bool x = false,
245 bool y = false, bool z = false) {
246 using memory_space = typename Kokkos::View<Tf***>::memory_space;
247
248 ippl::mpi::Communicator::size_type nrecvs;
249 nrecvs = intersection.size();
250
251 ippl::mpi::Communicator::buffer_type<memory_space> buf =
252 ippl::Comm->getBuffer<memory_space, Tf>(nrecvs);
253
254 int tag = TAG + id;
255
256 ippl::Comm->recv(i, tag, fd, *buf, nrecvs * sizeof(Tf), nrecvs);
257 buf->resetReadPos();
258
259 unpack(intersection, view, fd, nghost, ldom, x, y, z);
260 }
