28 template <typename ExecSpace, typename IndexView>
29 void validatePermutation(const IndexView& permute, size_t n,
30 const char* label = "sortParticles") {
31 using size_type = size_t;
32
33 size_type num_invalid = 0;
34 Kokkos::parallel_reduce(
35 label, Kokkos::RangePolicy<ExecSpace>(0, n),
36 KOKKOS_LAMBDA(size_type i, size_type & count) {
37 if (permute(i) >= n)
38 count++;
39 },
40 num_invalid);
41 Kokkos::fence();
42
43 if (num_invalid > 0) {
44 size_type first_invalid_idx = n;
45 Kokkos::parallel_reduce(
46 "find_first_invalid", Kokkos::RangePolicy<ExecSpace>(0, n),
47 KOKKOS_LAMBDA(size_type i, size_type & first_idx) {
48 if (permute(i) >= n && i < first_idx)
49 first_idx = i;
50 },
51 Kokkos::Min<size_type>(first_invalid_idx));
52 Kokkos::fence();
53
54 auto permute_host =
55 Kokkos::create_mirror_view_and_copy(Kokkos::HostSpace{}, permute);
56
57 throw std::runtime_error(std::string(label) + ": Found "
58 + std::to_string(num_invalid)
59 + " invalid permutation indices. First invalid: permute["
60 + std::to_string(first_invalid_idx)
61 + "] = " + std::to_string(permute_host(first_invalid_idx))
62 + " (n = " + std::to_string(n) + ")");
63 }
64 }

69 template <unsigned Dim, typename T, typename IndexType = size_t>
70 KOKKOS_INLINE_FUNCTION uint64_t computeMortonCode(const Vector<T, Dim>& position,
71 const Vector<T, Dim>& origin,
72 const Vector<T, Dim>& invdx,
73 const Vector<IndexType, Dim>& ngrid) {
74 uint64_t morton = 0;
75 constexpr int bits_per_dim = 21; // 21*3 = 63 bits, safe for 3D
76
77 uint32_t gridIndices[Dim];
78 for (unsigned d = 0; d < Dim; ++d) {
79 T sx = (position[d] - origin[d]) * invdx[d];
80 sx -= ngrid[d] * Kokkos::floor(sx / ngrid[d]); // wrap to [0, ngrid)
81 gridIndices[d] = static_cast<uint32_t>(sx);
82 }
83
84 for (int bit = 0; bit < bits_per_dim; ++bit) {
85 for (unsigned d = 0; d < Dim; ++d) {
86 if (gridIndices[d] & (1u << bit)) {
87 morton |= (uint64_t(1) << (bit * Dim + d));
88 }
89 }
90 }
91 return morton;
92 }

121 template <unsigned Dim, typename T>
122 Kokkos::View<size_t*, Kokkos::HostSpace> sortParticlesHost(
123 Kokkos::View<Vector<T, Dim>*, Kokkos::HostSpace> positions,
124 const Vector<T, Dim>& origin, const Vector<T, Dim>& invdx,
125 const Vector<size_t, Dim>& ngrid, size_t n) {
126 auto& bufs = ippl::detail::getDefaultBinSortBuffers<Kokkos::HostSpace>();
127 bufs.ensureCapacity(n, /*n_bins_p1=*/1);
128
129 auto& permute = bufs.permute();
130
131 // Build (key, index) pairs, sort, then extract permutation
132 std::vector<std::pair<uint64_t, size_t>> pairs(n);
133 for (size_t i = 0; i < n; ++i) {
134 pairs[i] = {computeMortonCode<Dim, T, size_t>(positions(i), origin, invdx, ngrid),
135 i};
136 }
137 std::sort(pairs.begin(), pairs.end()); // pair has lexicographic < by .first
138
139 for (size_t i = 0; i < n; ++i) {
140 permute(i) = pairs[i].second;
141 }
142
143 return permute;
144 }

158 auto& bufs = ippl::detail::getDefaultBinSortBuffers<Kokkos::CudaSpace>();
159 bufs.ensureCapacity(n, /*n_bins_p1=*/1);
160
161 // Step 1: compute Morton codes into binKeys, init permute to identity
162 auto keys = bufs.binKeys();
163 auto permute = bufs.permute();
164
165 Kokkos::parallel_for(
166 "MortonSort::ComputeKeys", Kokkos::RangePolicy<Kokkos::Cuda>(0, n),
167 ComputeMortonCodesFunctor<Dim, decltype(positions), T, decltype(keys)>{
168 positions, keys, origin, invdx, ngrid});
169
170 Kokkos::parallel_for(
171 "MortonSort::InitPermute", Kokkos::RangePolicy<Kokkos::Cuda>(0, n),
172 KOKKOS_LAMBDA(size_type i) { permute(i) = i; });
173 Kokkos::fence();
174
175 // Step 2: query temp storage, grow buffer if needed, then sort
176 auto keys_out = bufs.keysOut();
177 auto perm_out = bufs.permOut();
178
179 size_t temp_bytes = 0;
180 cub::DeviceRadixSort::SortPairs(nullptr, temp_bytes, keys.data(), keys_out.data(),
181 permute.data(), perm_out.data(), static_cast<int>(n));
182
183 bufs.ensureTempStorage(temp_bytes);
184
185 cub::DeviceRadixSort::SortPairs(bufs.tempStorage().data(), temp_bytes, keys.data(),
186 keys_out.data(), permute.data(), perm_out.data(),
187 static_cast<int>(n));
188 Kokkos::fence();
189

243 template <unsigned Dim, typename ExecSpace, typename T>
244 Kokkos::View<size_t*, typename ExecSpace::memory_space> sortParticles(
245 Kokkos::View<Vector<T, Dim>*, typename ExecSpace::memory_space> positions,
246 const Vector<T, Dim>& origin, const Vector<T, Dim>& invdx,
247 const Vector<size_t, Dim>& ngrid, size_t n) {
248 using memory_space = typename ExecSpace::memory_space;
249
250#ifdef KOKKOS_ENABLE_CUDA
251 if constexpr (std::is_same_v<ExecSpace, Kokkos::Cuda>) {
252 return sortParticlesCuda<Dim, T>(positions, origin, invdx, ngrid, n);
253 }
254#endif
255#ifdef KOKKOS_ENABLE_HIP
256 if constexpr (std::is_same_v<ExecSpace, Kokkos::HIP>) {
257 return sortParticlesHip<Dim, T>(positions, origin, invdx, ngrid, n);
258 }
259#endif
260 if constexpr (std::is_same_v<memory_space, Kokkos::HostSpace>) {
261 return sortParticlesHost<Dim, T>(positions, origin, invdx, ngrid, n);
262 } else {
263 // Generic fallback: sort on host, copy result to device buffer
264 auto positions_host =
265 Kokkos::create_mirror_view_and_copy(Kokkos::HostSpace{}, positions);
266
267 Vector<size_t, Dim> ngrid_host;
268 for (unsigned d = 0; d < Dim; ++d)
269 ngrid_host[d] = ngrid[d];
270
271 auto permute_host =
272 sortParticlesHost<Dim, T>(positions_host, origin, invdx, ngrid_host, n);
273
274 auto& bufs = ippl::detail::getDefaultBinSortBuffers<memory_space>();
275 bufs.ensureCapacity(n, /*n_bins_p1=*/1);
276
277 Kokkos::deep_copy(Kokkos::subview(bufs.permute(), std::make_pair(size_t(0), n)),
278 Kokkos::subview(permute_host, std::make_pair(size_t(0), n)));
279
280 return bufs.permute();
281 }

307 template <typename ExecSpace, typename DataView, typename IndexView>
308 void applyPermutation(DataView& data, const IndexView& permute, size_t n) {
309 DataView temp("temp", n);
310 Kokkos::parallel_for(
311 "apply_permutation", Kokkos::RangePolicy<ExecSpace>(0, n),
312 ApplyPermutationFunctor<DataView, DataView, IndexView>{data, temp, permute});
313 Kokkos::deep_copy(Kokkos::subview(data, Kokkos::make_pair(size_t(0), n)), temp);
