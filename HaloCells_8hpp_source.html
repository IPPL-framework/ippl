28 template <typename T, unsigned Dim, class... ViewArgs>
29 template <class Op>
30 void HaloCells<T, Dim, ViewArgs...>::exchangeBoundaries(view_type& view, Layout_t* layout,
31 SendOrder order) {
32 using neighbor_list = typename Layout_t::neighbor_list;
33 using range_list = typename Layout_t::neighbor_range_list;
34
35 auto& comm = layout->comm;
36
37 const neighbor_list& neighbors = layout->getNeighbors();
38 const range_list &sendRanges = layout->getNeighborsSendRange(),
39 &recvRanges = layout->getNeighborsRecvRange();
40
41 size_t totalRequests = 0;
42 for (const auto& componentNeighbors : neighbors) {
43 totalRequests += componentNeighbors.size();
44 }
45
46 using memory_space = typename view_type::memory_space;
47 using buffer_type = mpi::Communicator::buffer_type<memory_space>;
48 std::vector<MPI_Request> requests(totalRequests);
49
50 // sending loop
51 constexpr size_t cubeCount = detail::countHypercubes(Dim) - 1;
52 size_t requestIndex = 0;
53 for (size_t index = 0; index < cubeCount; index++) {
54 int tag = mpi::tag::HALO + index;
55 const auto& componentNeighbors = neighbors[index];
56 for (size_t i = 0; i < componentNeighbors.size(); i++) {
57 int targetRank = componentNeighbors[i];
58
59 bound_type range;
60 if (order == INTERNAL_TO_HALO) {
61 /*We store only the sending and receiving ranges
62 * of INTERNAL_TO_HALO and use the fact that the
63 * sending range of HALO_TO_INTERNAL is the receiving
64 * range of INTERNAL_TO_HALO and vice versa
65 */
66 range = sendRanges[index][i];
67 } else {
68 range = recvRanges[index][i];
69 }
70
71 size_type nsends;
72 pack(range, view, haloData_m, nsends);
73
74 buffer_type buf = comm.template getBuffer<memory_space, T>(nsends);
75
76 comm.isend(targetRank, tag, haloData_m, *buf, requests[requestIndex++], nsends);
77 buf->resetWritePos();
78 }
79 }
80
81 // receiving loop
82 for (size_t index = 0; index < cubeCount; index++) {
83 int tag = mpi::tag::HALO + Layout_t::getMatchingIndex(index);
84 const auto& componentNeighbors = neighbors[index];
85 for (size_t i = 0; i < componentNeighbors.size(); i++) {
86 int sourceRank = componentNeighbors[i];
87
88 bound_type range;
89 if (order == INTERNAL_TO_HALO) {
90 range = recvRanges[index][i];
91 } else {
92 range = sendRanges[index][i];
93 }
94
95 size_type nrecvs = range.size();
96
97 buffer_type buf = comm.template getBuffer<memory_space, T>(nrecvs);
98
99 comm.recv(sourceRank, tag, haloData_m, *buf, nrecvs * sizeof(T), nrecvs);
100 buf->resetReadPos();
101
102 unpack<Op>(range, view, haloData_m);
103 }
104 }
105
106 if (totalRequests > 0) {
107 MPI_Waitall(totalRequests, requests.data(), MPI_STATUSES_IGNORE);
108 }
109 comm.freeAllBuffers();
110 }

114 databuffer_type& fd, size_type& nsends) {

116
117 auto& buffer = fd.buffer;
118
119 size_t size = subview.size();
120 nsends = size;
121 if (buffer.size() < size) {

123 Kokkos::realloc(buffer, size * overalloc);

125
126 using index_array_type =
127 typename RangePolicy<Dim, typename view_type::execution_space>::index_array_type;
128 ippl::parallel_for(
129 "HaloCells::pack()", getRangePolicy(subview),
130 KOKKOS_LAMBDA(const index_array_type& args) {
131 int l = 0;
132
133 for (unsigned d1 = 0; d1 < Dim; d1++) {
134 int next = args[d1];

136 next *= subview.extent(d2);

138 l += next;
139 }

188 template <typename Op>
189 void HaloCells<T, Dim, ViewArgs...>::applyPeriodicSerialDim(view_type& view,
190 const Layout_t* layout,
191 const int nghost) {
192 int myRank = layout->comm.rank();
193 const auto& lDomains = layout->getHostLocalDomains();
194 const auto& domain = layout->getDomain();
195
196 using exec_space = typename view_type::execution_space;
197 using index_type = typename RangePolicy<Dim, exec_space>::index_type;
198
199 Kokkos::Array<index_type, Dim> ext, begin, end;
200
201 for (size_t i = 0; i < Dim; ++i) {
202 ext[i] = view.extent(i);
203 begin[i] = 0;
204 }
205
206 Op op;
207
208 for (unsigned d = 0; d < Dim; ++d) {
209 end = ext;
210 end[d] = nghost;
211
212 if (lDomains[myRank][d].length() == domain[d].length()) {
213 int N = view.extent(d) - 1;
214
215 using index_array_type =
216 typename RangePolicy<Dim,
217 typename view_type::execution_space>::index_array_type;
218 ippl::parallel_for(
219 "applyPeriodicSerialDim", createRangePolicy<Dim, exec_space>(begin, end),
220 KOKKOS_LAMBDA(index_array_type & coords) {
221 // The ghosts are filled starting from the inside
222 // of the domain proceeding outwards for both lower
223 // and upper faces. The extra brackets and explicit
224 // mention
225
226 // nghost + i
227 coords[d] += nghost;
228 auto&& left = apply(view, coords);
229
230 // N - nghost - i
231 coords[d] = N - coords[d];
232 auto&& right = apply(view, coords);
233
234 // nghost - 1 - i
235 coords[d] += 2 * nghost - 1 - N;
236 op(apply(view, coords), right);
237
238 // N - (nghost - 1 - i) = N - (nghost - 1) + i
239 coords[d] = N - coords[d];
240 op(apply(view, coords), left);
241 });
242
243 Kokkos::fence();
244 }
