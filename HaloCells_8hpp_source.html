28 template <typename T, unsigned Dim, class... ViewArgs>
29 template <class Op>
30 void HaloCells<T, Dim, ViewArgs...>::exchangeBoundaries(view_type& view, Layout_t* layout,
31 SendOrder order) {
32 using neighbor_list = typename Layout_t::neighbor_list;
33 using range_list = typename Layout_t::neighbor_range_list;
34
35 auto& comm = layout->comm;
36
37 const neighbor_list& neighbors = layout->getNeighbors();
38 const range_list &sendRanges = layout->getNeighborsSendRange(),
39 &recvRanges = layout->getNeighborsRecvRange();
40
41 size_t totalRequests = 0;
42 for (const auto& componentNeighbors : neighbors) {
43 totalRequests += componentNeighbors.size();
44 }
45
46 using memory_space = typename view_type::memory_space;
47 using buffer_type = mpi::Communicator::buffer_type<memory_space>;
48 std::vector<MPI_Request> requests(totalRequests);
49
50 // sending loop
51 constexpr size_t cubeCount = detail::countHypercubes(Dim) - 1;
52 size_t requestIndex = 0;
53 for (size_t index = 0; index < cubeCount; index++) {
54 int tag = mpi::tag::HALO + index;
55 const auto& componentNeighbors = neighbors[index];
56 for (size_t i = 0; i < componentNeighbors.size(); i++) {
57 int targetRank = componentNeighbors[i];
58
59 bound_type range;
60 if (order == INTERNAL_TO_HALO) {
61 /*We store only the sending and receiving ranges
62 * of INTERNAL_TO_HALO and use the fact that the
63 * sending range of HALO_TO_INTERNAL is the receiving
64 * range of INTERNAL_TO_HALO and vice versa
65 */
66 range = sendRanges[index][i];
67 } else {
68 range = recvRanges[index][i];
69 }
70
71 size_type nsends;
72 pack(range, view, haloData_m, nsends);
73
74 buffer_type buf = comm.template getBuffer<memory_space, T>(
75 mpi::tag::HALO_SEND + i * cubeCount + index, nsends);
76
77 comm.isend(targetRank, tag, haloData_m, *buf, requests[requestIndex++], nsends);
78 buf->resetWritePos();
79 }
80 }
81
82 // receiving loop
83 for (size_t index = 0; index < cubeCount; index++) {
84 int tag = mpi::tag::HALO + Layout_t::getMatchingIndex(index);
85 const auto& componentNeighbors = neighbors[index];
86 for (size_t i = 0; i < componentNeighbors.size(); i++) {
87 int sourceRank = componentNeighbors[i];
88
89 bound_type range;
90 if (order == INTERNAL_TO_HALO) {
91 range = recvRanges[index][i];
92 } else {
93 range = sendRanges[index][i];
94 }
95
96 size_type nrecvs = range.size();
97
98 buffer_type buf = comm.template getBuffer<memory_space, T>(
99 mpi::tag::HALO_RECV + i * cubeCount + index, nrecvs);
100
101 comm.recv(sourceRank, tag, haloData_m, *buf, nrecvs * sizeof(T), nrecvs);
102 buf->resetReadPos();
103
104 unpack<Op>(range, view, haloData_m);
105 }
106 }
107
108 if (totalRequests > 0) {
109 MPI_Waitall(totalRequests, requests.data(), MPI_STATUSES_IGNORE);
110 }
111 }

114 void HaloCells<T, Dim, ViewArgs...>::pack(const bound_type& range, const view_type& view,
115 databuffer_type& fd, size_type& nsends) {
116 auto subview = makeSubview(view, range);
117
118 auto& buffer = fd.buffer;
119
120 size_t size = subview.size();
121 nsends = size;
122 if (buffer.size() < size) {
123 int overalloc = Comm->getDefaultOverallocation();
124 Kokkos::realloc(buffer, size * overalloc);
125 }
126
127 using index_array_type =
128 typename RangePolicy<Dim, typename view_type::execution_space>::index_array_type;
129 ippl::parallel_for(
130 "HaloCells::pack()", getRangePolicy(subview),
131 KOKKOS_LAMBDA(const index_array_type& args) {
132 int l = 0;
133
134 for (unsigned d1 = 0; d1 < Dim; d1++) {
135 int next = args[d1];
136 for (unsigned d2 = 0; d2 < d1; d2++) {
137 next *= subview.extent(d2);
138 }
139 l += next;
140 }
141
142 buffer(l) = apply(subview, args);
143 });

189 template <typename Op>
190 void HaloCells<T, Dim, ViewArgs...>::applyPeriodicSerialDim(view_type& view,
191 const Layout_t* layout,
192 const int nghost) {
193 int myRank = layout->comm.rank();
194 const auto& lDomains = layout->getHostLocalDomains();
195 const auto& domain = layout->getDomain();
196
197 using exec_space = typename view_type::execution_space;
198 using index_type = typename RangePolicy<Dim, exec_space>::index_type;
199
200 Kokkos::Array<index_type, Dim> ext, begin, end;
201
202 for (size_t i = 0; i < Dim; ++i) {
203 ext[i] = view.extent(i);
204 begin[i] = 0;
205 }
206
207 Op op;
208
209 for (unsigned d = 0; d < Dim; ++d) {
210 end = ext;
211 end[d] = nghost;
212
213 if (lDomains[myRank][d].length() == domain[d].length()) {
214 int N = view.extent(d) - 1;
215
216 using index_array_type =
217 typename RangePolicy<Dim,
218 typename view_type::execution_space>::index_array_type;
219 ippl::parallel_for(
220 "applyPeriodicSerialDim", createRangePolicy<Dim, exec_space>(begin, end),
221 KOKKOS_LAMBDA(index_array_type & coords) {
222 // The ghosts are filled starting from the inside
223 // of the domain proceeding outwards for both lower
224 // and upper faces. The extra brackets and explicit
225 // mention
226
227 // nghost + i
228 coords[d] += nghost;
229 auto&& left = apply(view, coords);
230
231 // N - nghost - i
232 coords[d] = N - coords[d];
233 auto&& right = apply(view, coords);
234
235 // nghost - 1 - i
236 coords[d] += 2 * nghost - 1 - N;
237 op(apply(view, coords), right);
238
239 // N - (nghost - 1 - i) = N - (nghost - 1) + i
240 coords[d] = N - coords[d];
241 op(apply(view, coords), left);
242 });
243
244 Kokkos::fence();
245 }
