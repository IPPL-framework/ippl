60 KOKKOS_INLINE_FUNCTION void operator()(size_t j) const {
61 using result_type = decltype(args.grid)::non_const_value_type;
62
63 // Get actual particle index (sorted or direct)
64 const size_t p = UseSorting ? args.permute(j) : j;
65
66 // Build stencil
67 CoordinateTransform<RealType, Dim> transform{args.origin, args.invdx, args.n_grid};
68 Stencil stencil{};
69 for_constexpr(std::make_integer_sequence<int, Dim>{}, [&]<int d>() {
70 const RealType g_pos = transform.toGridCoordinate(args.x(p)[d], d);
71 const RealType g_pos_cc = g_pos - RealType(0.5);
72 const int idx0 = transform.getStencilBase(g_pos_cc, W);
73
74 stencil.base[d] = idx0 - args.local_offset[d] + args.nghost;
75
76 auto& kernel_vals = stencil.kw[d];
77 for (int i = 0; i < W; ++i) {
78 kernel_vals[i] =
79 args.kernel((g_pos - (RealType(idx0 + i) + RealType(0.5))) * args.inv_hw);
80 }
81 });

83 // Gather W^d stencil around non-uniform pt
84 result_type out = result_type(0);
85 auto rec = [&]<unsigned D>(auto&& self, RealType wprod, auto... idx) -> void {
86 const int bD = get<D>(stencil.base);
87 const auto& kD = get<D>(stencil.kw);
88
89 for (int i = 0; i < W; ++i) {
90 const RealType w = wprod * kD[i];
91 if constexpr (D == 0) {
92 out += args.grid(bD + i, idx...) * w;
93 } else {
94 self.template operator()<D - 1>(self, w, bD + i, idx...);
95 }
96 }
97 };
98 rec.template operator()<Dim - 1>(rec, RealType(1));

104 } else {
