114 template <typename ValueT, typename FieldT, class Mesh, class Centering, class... ViewArgs,
115 typename ParticleT, class... PosProps, class... ValProps>
116 void operator()(Field<FieldT, Dim, Mesh, Centering, ViewArgs...>& field,
117 const ParticleAttrib<Vector<ParticleT, Dim>, PosProps...>& positions,
118 const ParticleAttrib<ValueT, ValProps...>& values) {
119 using Types =
120 Interpolation::detail::DeducedScatterTypes<Kernel, decltype(field),
121 decltype(positions), decltype(values)>;
122
123 constexpr bool is_complex_field = ippl::detail::is_kokkos_complex<FieldT>::value;
124
125 // -- Estimate particle density (rho = local particles / local grid pts) --
126 const size_t n_particles = positions.getParticleCount();
127 const double rho_est = estimate_rho(field, n_particles);
128
129 // Auto-select the best method from the benchmark cache when not
130 // tuning and not locked. Tile/team/osub/z_batches are applied later
131 // in resolve_config for all methods, whether auto-selected or
132 // explicitly locked.
133 if (!config_m.enable_tuning && !config_m.lock_method) {
134 auto& cache = Interpolation::TileSizeCache::instance();
135 if (cache.loaded()) {
136 if (auto best = cache.get_best(kernel_m.width(), is_complex_field, rho_est)) {
137 config_m.method = best->method;
138 }
139 }
140 }
141
142 // Small-grid override: the cache is keyed only by (method, width,
143 // value_type, rho), so a sweep recorded at a large grid (e.g.
144 // 256^3, the size at which Tiled/OutputFocused start beating
145 // Atomic because the grid spills L2) gets returned for runtime
146 // calls whose local grid is far smaller. In that regime the
147 // tile-blocked methods pay binning + radix-sort overhead with
148 // nothing to amortise it against, and Atomic wins easily. Force
149 // Atomic when the local grid has fewer cells than 256^3, i.e.
150 // what was benchmarked. Could definiitely be improved.
151 if (!config_m.lock_method) {
152 const auto& local_dom = field.getLayout().getLocalNDIndex();
153 std::size_t local_cells = 1;
154 for (unsigned d = 0; d < Dim; ++d) {
155 local_cells *= static_cast<std::size_t>(local_dom[d].length());
156 }
157 constexpr std::size_t kSmallGridThreshold =
158 static_cast<std::size_t>(256) * 256 * 256;
159 if (local_cells < kSmallGridThreshold) {
160 config_m.method = Interpolation::ScatterMethod::Atomic;
161 config_m.sort = false;
162 }
163 }
164
165 // Backend-feasibility clamp. CSV presets are tagged by GPU arch but
166 // a runtime mismatch (e.g. an sm_90 CSV being read by a Serial
167 // build) leaves the cache holding Tiled/OutputFocused configs with
168 // team_size > 1 - those abort inside their TeamPolicy construction
169 // on a host backend. Force Atomic on backends where teams are not
170 // a meaningful parallelism (Serial today; extend the trait below
171 // if more host-only backends ever land).
172#ifdef KOKKOS_ENABLE_SERIAL
173 using exec_space_ = typename Types::execution_space;
174#endif
175 constexpr bool host_only_backend =
176#ifdef KOKKOS_ENABLE_SERIAL
177 std::is_same_v<exec_space_, Kokkos::Serial>
178#else
179 false
180#endif
181 ;
182 if constexpr (host_only_backend) {
183 if (config_m.method != Interpolation::ScatterMethod::Atomic) {
184 config_m.method = Interpolation::ScatterMethod::Atomic;
185 }
186 }
187
188 const auto method = config_m.method;
189
190 const bool run_atomic = (method == Interpolation::ScatterMethod::Atomic);
191
192 if (run_atomic) {
193 if (config_m.sort) {
194 dispatch<Interpolation::detail::AtomicScatter, Types,
195 Interpolation::detail::SortedPolicy>(field, positions, values,
196 rho_est);
197 } else {
198 dispatch<Interpolation::detail::AtomicScatter, Types,
199 Interpolation::detail::UnsortedPolicy>(field, positions, values,
200 rho_est);
201 }
202 return;
203 }
204
205 if (method == Interpolation::ScatterMethod::Tiled) {
206 dispatch<Interpolation::detail::TiledScatter, Types,
207 Interpolation::detail::SortedPolicy>(field, positions, values, rho_est);
208 return;
209 }
210
211 if (method == Interpolation::ScatterMethod::OutputFocused) {
212 dispatch<Interpolation::detail::GridParallelScatter, Types,
213 Interpolation::detail::SortedPolicy>(field, positions, values, rho_est);
214 }
215 }

218 // ------------------------------------------------------------------
219 // estimate_rho: LOCAL particles / LOCAL grid points.
220 //
221 // Uses getLocalNDIndex() for the rank-local owned domain, matching
222 // getParticleCount() which returns the rank-local particle count.
223 // The previous version used getDomain() (global) with local particle
224 // counts, under-estimating rho by a factor of #ranks.
225 // ------------------------------------------------------------------
226 template <typename FieldT, class Mesh, class Centering, class... ViewArgs>
227 static double estimate_rho(const Field<FieldT, Dim, Mesh, Centering, ViewArgs...>& field,
228 size_t n_particles) {
229 const auto& local_dom = field.getLayout().getLocalNDIndex();
230 size_t n_grid = 1;
231 for (unsigned d = 0; d < Dim; ++d)
232 n_grid *= static_cast<size_t>(local_dom[d].length());
233 return (n_grid > 0) ? static_cast<double>(n_particles) / static_cast<double>(n_grid)
234 : 1.0;
235 }

237 // ------------------------------------------------------------------
238 // resolve_config: load cached config for the ACTIVE method.
239 //
240 // Works identically for lock_method=true and lock_method=false:
241 // by this point config_m.method is already set (either kept as-is
242 // when locked, or switched by get_best() in operator()).
243 //
244 // Priority:
245 // 1. enable_tuning -> return config_m unchanged (runtime tuner)
246 // 2. cache hit -> apply tile, team_size, osub, z_batches
247 // 3. no hit -> return config_m unchanged
248 // ------------------------------------------------------------------
249 template <template <int, class, class> class Impl, int W, class Types, class Policy,
250 bool IsComplex>
251 Interpolation::ScatterConfig<Dim> resolve_config(double rho_est) const {
252 if (config_m.enable_tuning)
253 return config_m;
254
255 auto& cache = Interpolation::TileSizeCache::instance();
256 auto cached = cache.get(config_m.method, W, IsComplex, rho_est);
257
258 if (!cached.has_value())
259 return config_m;
260
261 const auto& e = cached.value();
262
263 auto resolved = config_m;
264 Vector<int, Dim> tile;
265 for (unsigned d = 0; d < Dim; ++d)
266 tile[d] = e.tile[d < 3 ? d : 2];
267 resolved.set_tile_size(tile);
268 if (e.team_size > 0)
269 resolved.team_size = e.team_size;
270 if (e.oversubscription_factor > 0)
271 resolved.oversubscription_factor = e.oversubscription_factor;
272 if (e.z_batches > 0)
273 resolved.z_batches = e.z_batches;
274 return resolved;
275 }

280 template <template <int, class, class> class Impl, int W, class Types, class Policy,
281 bool IsComplex>
282 static void clamp_tile_to_shmem(Interpolation::ScatterConfig<Dim>& cfg) {
283 if constexpr (!Impl<W, Types, Policy>::requires_binning)
284 return;
285
286 using execution_space = typename Types::execution_space;
287 using team_policy = Kokkos::TeamPolicy<execution_space>;
288
289 Vector<int, Dim> tile = cfg.get_tile_size();
290
291 // Per-team scratch capacity is invariant across the loop iterations
292 // for a fixed team_size. Hoist the query.
293 const size_t avail = team_policy(1, cfg.team_size).scratch_size_max(0);
294
295 const int max_iter = static_cast<int>(Dim) * 64 + 8;
296 for (int itr = 0; itr < max_iter; ++itr) {
297 const size_t req = Impl<W, Types, Policy>::template compute_scratch_size<IsComplex>(
298 tile, cfg.team_size, cfg.z_batches);
299 if (req <= avail)
300 break;
301
302 int best_dim = -1;
303 size_t best_req = req;
304
305 for (unsigned d = 0; d < Dim; ++d) {
306 if (tile[d] <= 1)
307 continue;
308 Vector<int, Dim> trial = tile;
309 --trial[d];
310 size_t trial_req =
311 Impl<W, Types, Policy>::template compute_scratch_size<IsComplex>(
312 trial, cfg.team_size, cfg.z_batches);
313 if (trial_req < best_req) {
314 best_req = trial_req;
315 best_dim = static_cast<int>(d);
316 }
317 }
318
319 if (best_dim < 0) {
320 if (cfg.team_size > 1) {
321 cfg.team_size = std::max(1, cfg.team_size / 2);
322 } else {
323 break;
324 }
325 } else {
326 --tile[best_dim];
327 }
328 }
329
330 cfg.set_tile_size(tile);
331 }

336 template <template <int, class, class> class Impl, class Types, class Policy, class Field,
337 class Positions, class Values>
338 void dispatch(Field& field, const Positions& positions, const Values& values,
339 double rho_est) {
340 using memory_space = typename Types::memory_space;
341 using RealType = typename Types::RealType;
342 using grid_value_t = typename Field::value_type;
343
344 constexpr bool is_complex = std::is_same_v<grid_value_t, Kokkos::complex<RealType>>;
345
346 const int width = kernel_m.width();
347 const size_t n_particles = positions.getParticleCount();
348
349 Interpolation::WidthDispatcher<1, std::decay_t<decltype(kernel_m)>::max_width>::dispatch(width, [&]<int W>() {
350 // -- Step 1: Resolve config from cache (density-aware) ----------
351 auto tuned_config = resolve_config<Impl, W, Types, Policy, is_complex>(rho_est);
352
353 if constexpr (Impl<W, Types, Policy>::requires_binning) {
354 if (config_m.enable_tuning) {
355 Vector<int, Dim> tuned_tile =
356 get_tuned_tile_size<Impl, W, Types, Policy, is_complex>(
357 field, tuned_config.get_tile_size());
358 tuned_config.set_tile_size(tuned_tile);
359 }
360 }
361
362 // -- Step 1b: Safety clamp (asymmetric, dimension-aware) --------
363 clamp_tile_to_shmem<Impl, W, Types, Policy, is_complex>(tuned_config);
364
365 const Vector<int, Dim> tile_size = tuned_config.get_tile_size();
366
367 // -- Step 2: Binning --------------------------------------------
368 Interpolation::detail::BinningResult<Dim, memory_space> binning;
369 if constexpr (Impl<W, Types, Policy>::requires_binning) {
370 binning = performBinning<Types>(positions, field, tile_size);
371 } else if (config_m.do_binning()) {
372 binning = performBinning<Types>(positions, field, tile_size);
373 }
374
375 // -- Step 3: Run functor ----------------------------------------
376 auto args = Impl<W, Types, Policy>::Arguments::create(
377 field, positions, values, kernel_m, tuned_config, binning);
378 Impl<W, Types, Policy> functor{std::move(args)};
379
380 field = 0.0;
381
382 functor.run(n_particles);
383 Kokkos::fence();
384
385 // -- Step 4: End tuning context ---------------------------------
386 if constexpr (Impl<W, Types, Policy>::requires_binning) {
387 if (config_m.enable_tuning) {
388 auto& tuner = Interpolation::detail::get_scatter_tuner<Impl, Dim, RealType,
389 is_complex>();
390 tuner.end();
391 }
392 }
393
394 field.accumulateHalo();
395 });
396 }

400 Vector<int, Dim> get_tuned_tile_size(const Field& /*field*/,
401 const Vector<int, Dim>& default_tile) {
402 using RealType = typename Types::RealType;
403 using execution_space = typename Types::execution_space;
404 using team_policy = Kokkos::TeamPolicy<execution_space>;
405
406 auto& tuner =
407 Interpolation::detail::get_scatter_tuner<Impl, Dim, RealType, IsComplex>();
408
409 if (!tuner.is_initialized()) {
410 const size_t max_scratch = team_policy(1, config_m.team_size).scratch_size_max(0);
411 std::vector<int> candidates = {1, 2, 3, 4, 8, 16, 32};
412

414 return Impl<W, Types, Policy>::template compute_scratch_size<IsComplex>(
415 tile, config_m.team_size, config_m.z_batches);
416 };
417 tuner.initialize("Scatter_" + std::string(typeid(Impl<W, Types, Policy>).name()),
418 candidates, max_scratch, scratch_calc, default_tile);
419 }
420 return tuner.begin();
421 }
