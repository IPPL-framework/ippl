69 void transform(TransformDirection direction, Field& f) {
70 // FFTW scaling
71 if constexpr (is_available_v<FFTW>) {
72 if (direction == FORWARD)
73 f = f / fftw_trig_scale();
74 }
75
76 auto view = f.getView();
77 const int ng = f.getNghost();
78
79 ensureTemp(f);
80 fft::copyToTemp<ExecSpace, decltype(temp_), decltype(view)>(temp_, view, ng);
81
82 if (direction == FORWARD) {
83 backend_->forward(temp_.data(), temp_.data());
84 } else {
85 backend_->backward(temp_.data(), temp_.data());
86 }
87
88 fft::copyFromTemp<ExecSpace, decltype(view), decltype(temp_)>(view, temp_, ng);
89
90 if constexpr (is_available_v<FFTW>) {
91 if (direction == BACKWARD)
92 f = f * fftw_trig_scale();
93 }
94 }
95
96 private:
97 std::unique_ptr<Backend_t> backend_;
98 TempView_t temp_;

100 void ensureTemp(const Field& f) {
101 if (temp_.size() != f.getOwned().size()) {
102 temp_ = ippl::detail::shrinkView("fft_trig_temp", f.getView(), f.getNghost());
103 }
