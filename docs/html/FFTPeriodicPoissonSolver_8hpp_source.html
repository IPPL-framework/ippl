76  switch (this->params_m.template get<int>("output_type")) {

80  KOKKOS_LAMBDA(const index_array_type& args) {

82  for (unsigned d = 0; d < Dim; ++d) {

123  Vector<int, Dim> iVec = args - nghost;
124  for (unsigned d = 0; d < Dim; ++d) {

126  }
127

135  // https://math.mit.edu/~stevenj/fft-deriv.pdf Algorithm 1
