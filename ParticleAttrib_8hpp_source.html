1//
2// Class ParticleAttrib
3// Templated class for all particle attribute classes.
4//
5// This templated class is used to represent a single particle attribute.
6// An attribute is one data element within a particle object, and is
7// stored as a Kokkos::View. This class stores the type information for the
8// attribute, and provides methods to create and destroy new items, and
9// to perform operations involving this attribute with others.
10//
11// ParticleAttrib is the primary element involved in expressions for
12// particles (just as BareField is the primary element there). This file
13// defines the necessary templated classes and functions to make
14// ParticleAttrib a capable expression-template participant.
15//

43 dview_m(deleteIndex(i)) = dview_m(keepIndex(i));

62 template <typename T, class... Properties>
63 void ParticleAttrib<T, Properties...>::unpack(size_type nrecvs) {
64 auto size = dview_m.extent(0);
65 size_type required = *(this->localNum_mp) + nrecvs;
66 if (size < required) {
67 int overalloc = Comm->getDefaultOverallocation();
68 this->resize(required * overalloc);
69 }
70
71 size_type count = *(this->localNum_mp);
72 using policy_type = Kokkos::RangePolicy<execution_space>;

74 "ParticleAttrib::unpack()", policy_type(0, nrecvs),
75 KOKKOS_CLASS_LAMBDA(const size_t i) { dview_m(count + i) = buf_m(i); });
76 Kokkos::fence();

104 template <typename T, class... Properties>
105 template <typename Field, class PT>
106 void ParticleAttrib<T, Properties...>::scatter(
107 Field& f, const ParticleAttrib<Vector<PT, Field::dim>, Properties...>& pp) const {
108 constexpr unsigned Dim = Field::dim;
109 using PositionType = typename Field::Mesh_t::value_type;
110
111 static IpplTimings::TimerRef scatterTimer = IpplTimings::getTimer("scatter");
112 IpplTimings::startTimer(scatterTimer);
113 using view_type = typename Field::view_type;
114 view_type view = f.getView();
115
116 using mesh_type = typename Field::Mesh_t;
117 const mesh_type& mesh = f.get_mesh();
118
119 using vector_type = typename mesh_type::vector_type;
120 using value_type = typename ParticleAttrib<T, Properties...>::value_type;
121
122 const vector_type& dx = mesh.getMeshSpacing();
123 const vector_type& origin = mesh.getOrigin();
124 const vector_type invdx = 1.0 / dx;
125
126 const FieldLayout<Dim>& layout = f.getLayout();
127 const NDIndex<Dim>& lDom = layout.getLocalNDIndex();
128 const int nghost = f.getNghost();
129
130 using policy_type = Kokkos::RangePolicy<execution_space>;
131 Kokkos::parallel_for(
132 "ParticleAttrib::scatter", policy_type(0, *(this->localNum_mp)),
133 KOKKOS_CLASS_LAMBDA(const size_t idx) {
134 // find nearest grid point
135 vector_type l = (pp(idx) - origin) * invdx + 0.5;
136 Vector<int, Field::dim> index = l;
137 Vector<PositionType, Field::dim> whi = l - index;
138 Vector<PositionType, Field::dim> wlo = 1.0 - whi;
139
140 Vector<size_t, Field::dim> args = index - lDom.first() + nghost;
141
142 // scatter
143 const value_type& val = dview_m(idx);
144 detail::scatterToField(std::make_index_sequence<1 << Field::dim>{}, view, wlo, whi,
145 args, val);
146 });
147 IpplTimings::stopTimer(scatterTimer);
148
149 static IpplTimings::TimerRef accumulateHaloTimer = IpplTimings::getTimer("accumulateHalo");
150 IpplTimings::startTimer(accumulateHaloTimer);
151 f.accumulateHalo();
152 IpplTimings::stopTimer(accumulateHaloTimer);
153 }

182 const int nghost = f.getNghost();
183

185 Kokkos::parallel_for(
186 "ParticleAttrib::gather", policy_type(0, *(this->localNum_mp)),
187 KOKKOS_CLASS_LAMBDA(const size_t idx) {
