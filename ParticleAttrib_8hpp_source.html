1//
2// Class ParticleAttrib
3// Templated class for all particle attribute classes.
4//
5// This templated class is used to represent a single particle attribute.
6// An attribute is one data element within a particle object, and is
7// stored as a Kokkos::View. This class stores the type information for the
8// attribute, and provides methods to create and destroy new items, and
9// to perform operations involving this attribute with others.
10//
11// ParticleAttrib is the primary element involved in expressions for
12// particles (just as BareField is the primary element there). This file
13// defines the necessary templated classes and functions to make
14// ParticleAttrib a capable expression-template participant.
15//

132 template <typename Field, class PT, typename policy_type>
133 void ParticleAttrib<T, Properties...>::scatter(
134 Field& f, const ParticleAttrib<Vector<PT, Field::dim>, Properties...>& pp,
135 policy_type iteration_policy, hash_type hash_array) const {
136 constexpr unsigned Dim = Field::dim;
137 using PositionType = typename Field::Mesh_t::value_type;
138
139 static IpplTimings::TimerRef scatterTimer = IpplTimings::getTimer("scatter");
140 IpplTimings::startTimer(scatterTimer);
141 using view_type = typename Field::view_type;
142 view_type view = f.getView();
143
144 using mesh_type = typename Field::Mesh_t;
145 const mesh_type& mesh = f.get_mesh();
146
147 using vector_type = typename mesh_type::vector_type;
148 using value_type = typename ParticleAttrib<T, Properties...>::value_type;
149
150 const vector_type& dx = mesh.getMeshSpacing();
151 const vector_type& origin = mesh.getOrigin();
152 const vector_type invdx = 1.0 / dx;
153
154 const FieldLayout<Dim>& layout = f.getLayout();
155 const NDIndex<Dim>& lDom = layout.getLocalNDIndex();
156 const int nghost = f.getNghost();
157
158 // using policy_type = Kokkos::RangePolicy<execution_space>;
159 const bool useHashView = hash_array.extent(0) > 0;
160 if (useHashView && (iteration_policy.end() > hash_array.extent(0))) {
161 Inform m("scatter");
162 m << "Hash array was passed to scatter, but size does not match iteration policy."
163 << endl;
164 ippl::Comm->abort();
165 }
166 auto dview = dview_m;
167 auto ppview = pp.getView();
168 Kokkos::parallel_for(
169 "ParticleAttrib::scatter", iteration_policy, KOKKOS_LAMBDA(const size_t idx) {
170 // map index to possible hash_map
171 size_t mapped_idx = useHashView ? hash_array(idx) : idx;
172
173 vector_type l = (ppview(mapped_idx) - origin) * invdx + 0.5;
174 Vector<int, Field::dim> index = l;
175 Vector<PositionType, Field::dim> whi = l - index;
176 Vector<PositionType, Field::dim> wlo = 1.0 - whi;
177
178 Vector<size_t, Field::dim> args = index - lDom.first() + nghost;
179
180 const value_type& val = dview(mapped_idx);
181 detail::scatterToField(std::make_index_sequence<1 << Field::dim>{}, view, wlo, whi,
182 args, val);
183 });
184 IpplTimings::stopTimer(scatterTimer);
185
186 static IpplTimings::TimerRef accumulateHaloTimer = IpplTimings::getTimer("accumulateHalo");
187 IpplTimings::startTimer(accumulateHaloTimer);
188 f.accumulateHalo();
189 IpplTimings::stopTimer(accumulateHaloTimer);
190 }

193 template <typename Field, typename P2>
194 void ParticleAttrib<T, Properties...>::gather(
195 Field& f, const ParticleAttrib<Vector<P2, Field::dim>, Properties...>& pp,
196 const bool addToAttribute) {
197 constexpr unsigned Dim = Field::dim;
198 using PositionType = typename Field::Mesh_t::value_type;
199
200 static IpplTimings::TimerRef fillHaloTimer = IpplTimings::getTimer("fillHalo");
201 IpplTimings::startTimer(fillHaloTimer);
202 f.fillHalo();
203 IpplTimings::stopTimer(fillHaloTimer);
204
205 static IpplTimings::TimerRef gatherTimer = IpplTimings::getTimer("gather");
206 IpplTimings::startTimer(gatherTimer);
207 const typename Field::view_type view = f.getView();
208
209 using mesh_type = typename Field::Mesh_t;
210 const mesh_type& mesh = f.get_mesh();
211
212 using vector_type = typename mesh_type::vector_type;
213
214 const vector_type& dx = mesh.getMeshSpacing();
215 const vector_type& origin = mesh.getOrigin();
216 const vector_type invdx = 1.0 / dx;
217
218 const FieldLayout<Dim>& layout = f.getLayout();
219 const NDIndex<Dim>& lDom = layout.getLocalNDIndex();
220 const int nghost = f.getNghost();
221
222 auto dview = dview_m;
223 auto ppview = pp.getView();
224 using policy_type = Kokkos::RangePolicy<execution_space>;
225 Kokkos::parallel_for(
226 "ParticleAttrib::gather", policy_type(0, *(this->localNum_mp)),
227 KOKKOS_LAMBDA(const size_t idx) {
228 vector_type l = (ppview(idx) - origin) * invdx + 0.5;
229 Vector<int, Field::dim> index = l;
230 Vector<PositionType, Field::dim> whi = l - index;
231 Vector<PositionType, Field::dim> wlo = 1.0 - whi;
232
233 Vector<size_t, Field::dim> args = index - lDom.first() + nghost;
234
235 // gather
236 value_type gathered = detail::gatherFromField(
237 std::make_index_sequence<1 << Field::dim>{}, view, wlo, whi, args);
238 if (addToAttribute) {
239 dview(idx) += gathered;
240 } else {
241 dview(idx) = gathered;
242 }
243 });
244 IpplTimings::stopTimer(gatherTimer);
245 }

407 template <typename T, class... Properties>
408 template <unsigned Dim, class M, class C, class FT, class ST, class PT>
409 void ParticleAttrib<T, Properties...>::gatherPIFNUFFT(
410 Field<FT, Dim, M, C>& f, Field<ST, Dim, M, C>& Sk,
411 const ParticleAttrib<Vector<PT, Dim>, Properties...>& pp,
412 FFT<NUFFTransform, Field<ST, Dim, M, C>>* nufft, ParticleAttrib<PT, Properties...>& q) {
413 static IpplTimings::TimerRef gatherPIFNUFFTTimer =
414 IpplTimings::getTimer("GatherPIFNUFFT");
415 IpplTimings::startTimer(gatherPIFNUFFTTimer);
416
417 typename Field<FT, Dim, M, C>::uniform_type tempField;
418
419 FieldLayout<Dim>& layout = f.getLayout();
420 M& mesh = f.get_mesh();
421 const auto& lDom = layout.getLocalNDIndex();
422
423 tempField.initialize(mesh, layout);
424
425 using view_type = typename Field<FT, Dim, M, C>::view_type;
426 using vector_type = typename M::vector_type;
427 view_type fview = f.getView();
428 view_type tempview = tempField.getView();
429 auto qview = q.getView();
430 typename Field<ST, Dim, M, C>::view_type Skview = Sk.getView();
431 const int nghost = f.getNghost();
432 const vector_type& dx = mesh.getMeshSpacing();
433 const auto& domain = layout.getDomain();
434 vector_type Len;
435 Vector<int, Dim> N;
436
437 for (unsigned d = 0; d < Dim; ++d) {
438 N[d] = domain[d].length();
439 Len[d] = dx[d] * N[d];
440 }
441
442 constexpr double pi = Kokkos::numbers::pi_v<double>;
443 Kokkos::complex<double> imag = {0.0, 1.0};
444 size_t Np = *(this->localNum_mp);
445
446 using mdrange_type = Kokkos::MDRangePolicy<Kokkos::Rank<3>, execution_space>;
447
448 auto dview = dview_m;
449
450 for (size_t gd = 0; gd < Dim; ++gd) {
451 Kokkos::parallel_for(
452 "Gather NUFFT",
453 mdrange_type(
454 {nghost, nghost, nghost},
455 {fview.extent(0) - nghost, fview.extent(1) - nghost,
456 fview.extent(2) - nghost}),
457 KOKKOS_LAMBDA(const int i, const int j, const int k) {
458 Vector<int, 3> iVec = {i, j, k};
459 for (unsigned d = 0; d < Dim; ++d) {
460 iVec[d] = iVec[d] - nghost + lDom[d].first();
461 }
462 Vector<double, 3> kVec;
463
464 double Dr = 0.0;
465 for (size_t d = 0; d < Dim; ++d) {
466 bool shift = (iVec[d] > (N[d] / 2));
467 kVec[d] = 2 * pi / Len[d] * (iVec[d] - shift * N[d]);
468 Dr += kVec[d] * kVec[d];
469 }
470
471 tempview(i, j, k) = fview(i, j, k);
472
473 bool isNotZero = (Dr != 0.0);
474 double factor = isNotZero * (1.0 / (Dr + ((!isNotZero) * 1.0)));
475
476 tempview(i, j, k) *= -Skview(i, j, k) * (imag * kVec[gd] * factor);
477 });
478
479 nufft->transform(pp, q, tempField);
480
481 Kokkos::parallel_for(
482 "Assign E gather NUFFT", Np,
483 KOKKOS_LAMBDA(const size_t i) { dview(i)[gd] = qview(i); });
484 }
485
486 IpplTimings::stopTimer(gatherPIFNUFFTTimer);
487 }
