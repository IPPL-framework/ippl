1//
2// Class IpplTimings
3// IpplTimings - a simple singleton class which lets the user create and
4// timers that can be printed out at the end of the program.
5//
6// General usage
7// 1) create a timer:
8// IpplTimings::TimerRef val = IpplTimings::getTimer("timer name");
9// This will either create a new one, or return a ref to an existing one
10//
11// 2) start a timer:
12// IpplTimings::startTimer(val);
13// This will start the referenced timer running. If it is already running,
14// it will not change anything.
15//
16// 3) stop a timer:
17// IpplTimings::stopTimer(val);
18// This will stop the timer, assuming it was running, and add in the
19// time to the accumulating time for that timer.
20//
21// 4) print out the results:
22// IpplTimings::print();
23//
24// 5) reset all timers (for warmup):
25// IpplTimings::resetAllTimers();
26// Clears all accumulated times and individual measurements.
27//
28// 6) export to CSV for violin plots:
29// IpplTimings::dumpToCSV("output.csv");
30// Exports all individual timing measurements for statistical analysis.
31//

47class IpplTimerInfo {
48public:
49 // typedef for reference to a timer
50 typedef unsigned int TimerRef;
51
52 // constructor
53 IpplTimerInfo()
54 : name("")
55 , wallTime(0.0)
56 , indx(std::numeric_limits<TimerRef>::max())
57 , measurements() {
58 clear();
59 }
60
61 // destructor
62 ~IpplTimerInfo() {}
63
64 // timer operations
65 void start() {
66 if (!running) {
67 t.clear(); // running was false, no need to stop first
68 t.start();
69 running = true;
70 }
71 }
72
73 void stop() {
74 if (running) {
75 t.stop();
76 running = false;
77 const double elapsed = t.elapsed();
78 wallTime += elapsed;
79 measurements.push_back(elapsed);
80 }
81 }
82
83 void clear() {
84 t.stop();
85 t.clear();
86 running = false;
87 }
88
89 void clearAll() {
90 clear();
91 wallTime = 0.0;
92 measurements.clear();
93 }
94
95 // the IPPL timer that this object manages
96 Timer t;
97
98 // the name of this timer
99 std::string name;
100
101 // the accumulated time
102 double wallTime;
103
104 // is the timer turned on right now?
105 bool running;
106
107 // an index value for this timer
108 TimerRef indx;
109
110 std::vector<double> measurements;
111};
