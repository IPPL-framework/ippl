66 MPI_Datatype type = MPI_BYTE;
67 if (core::type_names.find(std::type_index(typeid(T))) == core::type_names.end()) {
68 if constexpr (vector_dim_type<T>::Dim > 0) {
69 MPI_Datatype tp;
70 MPI_Type_contiguous(
71 vector_dim_type<T>::Dim,
72 get_mpi_datatype<typename T::value_type>(typename T::value_type{}), &tp);
73 MPI_Type_commit(&tp);
74 core::type_names[std::type_index(typeid(T))] = tp;
75 }
76 }
77 try {
78 type = core::type_names.at(std::type_index(typeid(T)));
79 } catch (...) {
80 throw IpplException("ippl::mpi::get_mpi_datatype", "No such type available.");
81 }
