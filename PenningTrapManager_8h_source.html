213 void LeapFrogStep(){
214 // LeapFrog time stepping https://en.wikipedia.org/wiki/Leapfrog_integration
215 // Here, we assume a constant charge-to-mass ratio of -1 for
216 // all the particles hence eliminating the need to store mass as
217 // an attribute
218 static IpplTimings::TimerRef PTimer = IpplTimings::getTimer("pushVelocity");
219 static IpplTimings::TimerRef RTimer = IpplTimings::getTimer("pushPosition");
220 static IpplTimings::TimerRef updateTimer = IpplTimings::getTimer("update");
221 static IpplTimings::TimerRef domainDecomposition = IpplTimings::getTimer("loadBalance");
222 static IpplTimings::TimerRef SolveTimer = IpplTimings::getTimer("solve");
223
224 double alpha = this->alpha_m;
225 double Bext = this->Bext_m;
226 double DrInv = this->DrInv_m;
227 double V0 = 30 * this->length_m[2];
228 Vector_t<double, Dim> length = this->length_m;
229 Vector_t<double, Dim> origin = this->origin_m;
230 double dt = this->dt_m;
231 std::shared_ptr<ParticleContainer_t> pc = this->pcontainer_m;
232 std::shared_ptr<FieldContainer_t> fc = this->fcontainer_m;
233
234 IpplTimings::startTimer(PTimer);
235 auto Rview = pc->R.getView();
236 auto Pview = pc->P.getView();
237 auto Eview = pc->E.getView();
238 Kokkos::parallel_for(
239 "Kick1", pc->getLocalNum(), KOKKOS_LAMBDA(const size_t j) {
240 double Eext_x = -(Rview(j)[0] - origin[0] - 0.5 * length[0])
241 * (V0 / (2 * Kokkos::pow(length[2], 2)));
242 double Eext_y = -(Rview(j)[1] - origin[1] - 0.5 * length[1])
243 * (V0 / (2 * Kokkos::pow(length[2], 2)));
244 double Eext_z = (Rview(j)[2] - origin[2] - 0.5 * length[2])
245 * (V0 / (Kokkos::pow(length[2], 2)));
246
247 Eext_x += Eview(j)[0];
248 Eext_y += Eview(j)[1];
249 Eext_z += Eview(j)[2];
250
251 Pview(j)[0] += alpha * (Eext_x + Pview(j)[1] * Bext);
252 Pview(j)[1] += alpha * (Eext_y - Pview(j)[0] * Bext);
253 Pview(j)[2] += alpha * Eext_z;
254 });
255 Kokkos::fence();
256 ippl::Comm->barrier();
257 IpplTimings::stopTimer(PTimer);
258
259 // drift
260 IpplTimings::startTimer(RTimer);
261 pc->R = pc->R + dt * pc->P;
262 IpplTimings::stopTimer(RTimer);
263
264 // Since the particles have moved spatially update them to correct processors
265 IpplTimings::startTimer(updateTimer);
266 pc->update();
267 IpplTimings::stopTimer(updateTimer);
268
269 size_type totalP = this->totalP_m;
270 int it = this->it_m;
271 bool isFirstRepartition = false;
272 if (this->loadbalancer_m->balance(totalP, it + 1)) {
273 IpplTimings::startTimer(domainDecomposition);
274 auto* mesh = &fc->getRho().get_mesh();
275 auto* FL = &fc->getFL();
276 this->loadbalancer_m->repartition(FL, mesh, isFirstRepartition);
277 IpplTimings::stopTimer(domainDecomposition);
278 }
279
280 // scatter the charge onto the underlying grid
281 this->par2grid();
282
283 // Field solve
284 IpplTimings::startTimer(SolveTimer);
285 this->fsolver_m->runSolver();
286 IpplTimings::stopTimer(SolveTimer);
287
288 // gather E field
289 this->grid2par();
290
291 IpplTimings::startTimer(PTimer);
292 auto R2view = pc->R.getView();
293 auto P2view = pc->P.getView();
294 auto E2view = pc->E.getView();
295 Kokkos::parallel_for(
296 "Kick2", pc->getLocalNum(), KOKKOS_LAMBDA(const size_t j) {
297 double Eext_x = -(R2view(j)[0] - origin[0] - 0.5 * length[0])
298 * (V0 / (2 * Kokkos::pow(length[2], 2)));
299 double Eext_y = -(R2view(j)[1] - origin[1] - 0.5 * length[1])
300 * (V0 / (2 * Kokkos::pow(length[2], 2)));
301 double Eext_z = (R2view(j)[2] - origin[2] - 0.5 * length[2])
302 * (V0 / (Kokkos::pow(length[2], 2)));
303
304 Eext_x += E2view(j)[0];
305 Eext_y += E2view(j)[1];
306 Eext_z += E2view(j)[2];
307
308 P2view(j)[0] = DrInv * (P2view(j)[0] + alpha * (Eext_x + P2view(j)[1] * Bext + alpha * Bext * Eext_y));
309 P2view(j)[1] = DrInv * (P2view(j)[1] + alpha * (Eext_y - P2view(j)[0] * Bext - alpha * Bext * Eext_x));
310 P2view(j)[2] += alpha * Eext_z;
311 });
312 Kokkos::fence();
313 ippl::Comm->barrier();
314 IpplTimings::stopTimer(PTimer);
315 }

324 void dumpData() {
325 auto Pview = this->pcontainer_m->P.getView();
326 double kinEnergy = 0.0;
327 double potEnergy = 0.0;
328 this->fcontainer_m->getRho() = dot(this->fcontainer_m->getE(), this->fcontainer_m->getE());
329 potEnergy = 0.5 * this->hr_m[0] * this->hr_m[1] * this->hr_m[2] * this->fcontainer_m->getRho().sum();
330
331 Kokkos::parallel_reduce(
332 "Particle Kinetic Energy", this->pcontainer_m->getLocalNum(),
333 KOKKOS_LAMBDA(const int i, double& valL) {
334 double myVal = dot(Pview(i), Pview(i)).apply();
335 valL += myVal;
336 },
337 Kokkos::Sum<double>(kinEnergy));
338
339 kinEnergy *= 0.5;
340 double gkinEnergy = 0.0;
341
342 ippl::Comm->reduce(kinEnergy, gkinEnergy, 1, std::plus<double>());
343
344 const int nghostE = this->fcontainer_m->getE().getNghost();
345 auto Eview = this->fcontainer_m->getE().getView();
346 Vector_t<T, Dim> normE;
347
348 using index_array_type = typename ippl::RangePolicy<Dim>::index_array_type;
349 for (unsigned d = 0; d < Dim; ++d) {
350 T temp = 0.0;
351 ippl::parallel_reduce(
352 "Vector E reduce", ippl::getRangePolicy(Eview, nghostE),
353 KOKKOS_LAMBDA(const index_array_type& args, T& valL) {
354 // ippl::apply accesses the view at the given indices and obtains a
355 // reference; see src/Expression/IpplOperations.h
356 T myVal = std::pow(ippl::apply(Eview, args)[d], 2);
357 valL += myVal;
358 },
359 Kokkos::Sum<T>(temp));
360 Kokkos::fence();
361 T globaltemp = 0.0;
362 ippl::Comm->reduce(temp, globaltemp, 1, std::plus<double>());
363
364 normE[d] = std::sqrt(globaltemp);
365 ippl::Comm->barrier();
366 }
367
368 if (ippl::Comm->rank() == 0) {
369 std::stringstream fname;
370 fname << "data/ParticleField_";
371 fname << ippl::Comm->size();
372 fname << "_manager";
373 fname << ".csv";
374
375 Inform csvout(NULL, fname.str().c_str(), Inform::APPEND);
376 csvout.precision(10);
377 csvout.setf(std::ios::scientific, std::ios::floatfield);
378
379 if ( std::fabs(this->time_m) < 1e-14 ) {
380 csvout << "time, Potential energy, Kinetic energy, Total energy, Rho_norm2";
381 for (unsigned d = 0; d < Dim; d++) {
382 csvout << ", E" << static_cast<char>((Dim <= 3 ? 'x' : '1') + d) << "_norm2";
383 }
384 csvout << endl;
385 }
386
387 csvout << this->time_m << " " << potEnergy << " " << gkinEnergy << " "
388 << potEnergy + gkinEnergy << " " << this->rhoNorm_m << " ";
389 for (unsigned d = 0; d < Dim; d++) {
390 csvout << normE[d] << " ";
391 }
392 csvout << endl;
393 csvout.flush();
394 }
395 ippl::Comm->barrier();
396 }
