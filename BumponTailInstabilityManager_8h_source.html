23struct CustomDistributionFunctions {
24 struct CDF {
25 KOKKOS_INLINE_FUNCTION double operator()(double x, unsigned int d,
26 const double* params_p) const {
27 if (d == Dim - 1)
28 return x
29 + (params_p[d * 2 + 0] / params_p[d * 2 + 1])
30 * Kokkos::sin(params_p[d * 2 + 1] * x);
31 else
32 return ippl::random::uniform_cdf_func<double>(x);
33 }
34 };
35
36 struct PDF {
37 KOKKOS_INLINE_FUNCTION double operator()(double x, unsigned int d,
38 double const* params_p) const {
39 if (d == Dim - 1)
40 return (1.0 + params_p[d * 2 + 0] * Kokkos::cos(params_p[d * 2 + 1] * x));
41 else
42 return ippl::random::uniform_pdf_func<double>();
43 }
44 };
45
46 struct Estimate {
47 KOKKOS_INLINE_FUNCTION double operator()(double u, unsigned int d,
48 double const* params_p) const {
49 return u + params_p[d] * 0.;
50 }
51 };

312 void LeapFrogStep() {
313 // LeapFrog time stepping https://en.wikipedia.org/wiki/Leapfrog_integration
314 // Here, we assume a constant charge-to-mass ratio of -1 for
315 // all the particles hence eliminating the need to store mass as
316 // an attribute
317 static IpplTimings::TimerRef PTimer = IpplTimings::getTimer("pushVelocity");
318 static IpplTimings::TimerRef RTimer = IpplTimings::getTimer("pushPosition");
319 static IpplTimings::TimerRef updateTimer = IpplTimings::getTimer("update");
320 static IpplTimings::TimerRef domainDecomposition = IpplTimings::getTimer("loadBalance");
321 static IpplTimings::TimerRef SolveTimer = IpplTimings::getTimer("solve");
322
323 double dt = this->dt_m;
324 std::shared_ptr<ParticleContainer_t> pc = this->pcontainer_m;
325 std::shared_ptr<FieldContainer_t> fc = this->fcontainer_m;
326
327 IpplTimings::startTimer(PTimer);
328 pc->P = pc->P - 0.5 * dt * pc->E;
329 IpplTimings::stopTimer(PTimer);
330
331 // drift
332 IpplTimings::startTimer(RTimer);
333 pc->R = pc->R + dt * pc->P;
334 IpplTimings::stopTimer(RTimer);
335
336 // Since the particles have moved spatially update them to correct processors
337 IpplTimings::startTimer(updateTimer);
338 pc->update();
339 IpplTimings::stopTimer(updateTimer);
340
341 size_type totalP = this->totalP_m;
342 int it = this->it_m;
343 bool isFirstRepartition = false;
344 if (this->loadbalancer_m->balance(totalP, it + 1)) {
345 IpplTimings::startTimer(domainDecomposition);
346 auto* mesh = &fc->getRho().get_mesh();
347 auto* FL = &fc->getFL();
348 this->loadbalancer_m->repartition(FL, mesh, isFirstRepartition);
349 IpplTimings::stopTimer(domainDecomposition);
350 }
351
352 // scatter the charge onto the underlying grid
353 this->par2grid();
354
355 // Field solve
356 IpplTimings::startTimer(SolveTimer);
357 this->fsolver_m->runSolver();
358 IpplTimings::stopTimer(SolveTimer);
359
360 // gather E field
361 this->grid2par();
362
363 // kick
364 IpplTimings::startTimer(PTimer);
365 pc->P = pc->P - 0.5 * dt * pc->E;
366 IpplTimings::stopTimer(PTimer);
367 }

386 void dump(int it_, std::shared_ptr<ParticleContainer_t> pc, bool allDims = false) {
387 const auto pcount = pc->getLocalNum();
388 phase.realloc(pcount);
389 auto& Ri = pc->R;
390 auto& Pi = pc->P;
391 for (unsigned d = allDims ? 0 : Dim - 1; d < Dim; d++) {
392 Kokkos::parallel_for(
393 "Copy phase space", pcount, KOKKOS_CLASS_LAMBDA(const size_t i) {
394 phase(i) = {Ri(i)[d], Pi(i)[d]};
395 });
396 phaseSpace = 0;
397 Kokkos::fence();
398 scatter(pc->q, phaseSpace, phase);
399 auto& view = phaseSpace.getView();
400 MPI_Reduce(view.data(), phaseSpaceBuf.getView().data(), view.size(), MPI_DOUBLE,
401 MPI_SUM, 0, ippl::Comm->getCommunicator());
402 if (ippl::Comm->rank() == 0) {
403 std::stringstream fname;
404 fname << "PhaseSpace_t=" << it_ << "_d=" << d << ".csv";
405
406 Inform out("Phase Dump", fname.str().c_str(), Inform::OVERWRITE, 0);
407 phaseSpaceBuf.write(out);
408
409 auto max = phaseSpaceBuf.max();
410 auto min = phaseSpaceBuf.min();
411 if (max > maxValue) {
412 maxValue = max;
413 }
414 if (min < minValue) {
415 minValue = min;
416 }
417 }
418 ippl::Comm->barrier();
419 }
420 MPI_Bcast(&maxValue, 1, MPI_DOUBLE, 0, ippl::Comm->getCommunicator());
421 MPI_Bcast(&minValue, 1, MPI_DOUBLE, 0, ippl::Comm->getCommunicator());
422 }
