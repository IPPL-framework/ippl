1//
2// Class ParticleSpatialOverlapLayout
3// Particle layout based on spatial decomposition.
4//
5// This is a specialized version of ParticleSpatialLayout, which allows
6// particles to be on multiple processes if they are close to the respective
7// region.
8//
9// After each 'time step' in a calculation, which is defined as a period
10// in which the particle positions may change enough to affect the global
11// layout, the user must call the 'update' routine, which will move
12// particles between processors, etc. After the Nth call to update, a
13// load balancing routine will be called instead. The user may set the
14// frequency of load balancing (N), or may supply a function to
15// determine if load balancing should be done or not.
16//

42 template <typename T, unsigned Dim, class Mesh, typename... Properties>
43 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::initializeCells() {
44 const auto rank = Comm->rank();
45 const auto hLocalRegions = this->rlayout_m.gethLocalRegions();
46 for (unsigned d = 0; d < Dim; ++d) {
47 PAssert(rcutoff_m <= hLocalRegions(rank)[d].length() / 2 &&
48 "Cutoff is too big with respect to region. "
49 "Particle could be on 3 or more ranks ins one dimension");
50 }
51
52 /* precompute information of cell structure. dividing the region into cells of at least
53 * rcutoff_m length, the length of the overlap. Use std::floor to make sure the boundary
54 * cells are big enough as well.
55 */
56 totalCells_m = 1;
57 numLocalCells_m = 1;
58 for (unsigned d = 0; d < Dim; ++d) {
59 const T length = hLocalRegions(rank)[d].length();
60 const size_type nLocalCells = std::floor(length / rcutoff_m);
61 // two ghost cells, one in each direction
62 numCells_m[d] = nLocalCells + 2 * numGhostCellsPerDim_m;
63 cellWidth_m[d] = length / nLocalCells;
64 totalCells_m *= numCells_m[d];
65 numLocalCells_m *= nLocalCells;
66 }
67 numGhostCells_m = totalCells_m - numLocalCells_m;
68
69 /* calculate cell strides to compute flat index from nd-index
70 * idx = strides[0] * idx0 + strides[1] * idx1 + ...
71 */
72 std::exclusive_scan(numCells_m.begin(), numCells_m.end(), cellStrides_m.begin(), 1,
73 std::multiplies());
74
75 cellParticleCount_m = hash_type("cellParticleCount", totalCells_m);
76 cellStartingIdx_m = hash_type("cellStartingIdx", totalCells_m + 1);
77
78 // Compute cell permutation need to place ghost cells at the end to make sure only local
79 // particles are at indices 0 to numLocalParticles
80 hash_type cellPermutationForward("cell permutation forward", totalCells_m);
81 hash_type cellPermutationBackward("cell permutation backward", totalCells_m);
82
83 /* Step 1. compute prefix sum to determine where the local/ghost cells go */
84 hash_type localPrefixSum("local prefix sum", totalCells_m);
85 hash_type ghostPrefixSum("ghost prefix sum", totalCells_m);
86 const auto& numCells = numCells_m;
87 constexpr auto is = std::make_index_sequence<Dim>();
88
89 Kokkos::parallel_scan(
90 "scan_local", Kokkos::RangePolicy(0, totalCells_m),
91 KOKKOS_LAMBDA(const size_type i, size_type& update, const bool final) {
92 const size_type val = isLocalCellIndex(is, toCellIndex(i, numCells), numCells);
93 if (final) {
94 localPrefixSum(i) = update;
95 }
96 update += val;
97 });
98 Kokkos::parallel_scan(
99 "scan_ghost", Kokkos::RangePolicy(0, totalCells_m),
100 KOKKOS_LAMBDA(const size_type i, size_type& update, const bool final) {
101 const size_type val = !isLocalCellIndex(is, toCellIndex(i, numCells), numCells);
102 if (final) {
103 ghostPrefixSum(i) = update;
104 }
105 update += val;
106 });
107
108 /* Step 2. assign the cells at the correct locations of the permutations */
109 const auto numLocalCells = numLocalCells_m;
110 Kokkos::parallel_for(
111 "assign_permutations", Kokkos::RangePolicy(0, totalCells_m),
112 KOKKOS_LAMBDA(const size_type i) {
113 if (const auto cellIdx = toCellIndex(i, numCells);
114 isLocalCellIndex(is, cellIdx, numCells)) {
115 const size_type localIdx = localPrefixSum(i);
116 cellPermutationForward(i) = localIdx;
117 cellPermutationBackward(localIdx) = i;
118 } else {
119 const size_type ghostIdx = numLocalCells + ghostPrefixSum(i);
120 cellPermutationForward(i) = ghostIdx;
121 cellPermutationBackward(ghostIdx) = i;
122 }
123 });
124
125 cellPermutationForward_m = cellPermutationForward;
126 cellPermutationBackward_m = cellPermutationBackward;
127 }

143 template <typename ParticleContainer, typename index_type>
144 inline void copyAttributes(ParticleContainer& pc, const index_type& boundaryIndices) {
145 const auto numLoc = pc.getLocalNum();
146 const auto numBoundaryParticles = boundaryIndices.size();
147 detail::runForAllSpaces([&]<typename MemorySpace>() {
148 size_t numAttributesInSpace = 0;
149 pc.template forAllAttributes<MemorySpace>(
150 [&numAttributesInSpace]<typename Attribute>(Attribute&) {
151 ++numAttributesInSpace;
152 });
153 if (numAttributesInSpace == 0) {
154 return;
155 }
156
157 pc.template forAllAttributes<MemorySpace>(
158 [boundaryIndicesMirror = Kokkos::create_mirror_view_and_copy(
159 MemorySpace(), boundaryIndices)]<typename Attribute>(Attribute& att) {
160 att->internalCopy(boundaryIndicesMirror);
161 });
162 });
163 Kokkos::fence();
164 /* make sure other functions (particleExchange and buildCells) know about the ghost
165 * particles. They will not stay pc's particle as their positions are outside the global
166 * domain but are needed as ghost particles.
167 */
168 pc.setLocalNum(numLoc + numBoundaryParticles);
169 }

172 template <typename T, unsigned Dim, class Mesh, typename... Properties>
173 template <class ParticleContainer>
174 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::createPeriodicGhostParticles(
175 ParticleContainer& pc) {
176 static IpplTimings::TimerRef timer = IpplTimings::getTimer("createPeriodicGhostParticles");
177 IpplTimings::startTimer(timer);
178 /* periodic boundary conditions come in pairs. Thus collect whether each dimension is
179 * subject to periodic boundary conditions
180 */
181 Vector<bool, Dim> periodic;
182 for (unsigned d = 0; d < Dim; ++d) {
183 periodic[d] = this->getParticleBC()[2 * d];
184 }
185 // no need to create periodic ghost particles if no dimension is periodic
186 if (!std::any_of(periodic.begin(), periodic.end(), [](auto bc) {
187 return bc == BC::PERIODIC;
188 })) {
189 return;
190 }
191
192 const auto& globalRegion = this->rlayout_m.getDomain();
193 const auto overlap = rcutoff_m;
194 const auto numLoc = pc.getLocalNum();
195 const auto positions = pc.R.getView();
196
197 constexpr auto is = std::make_index_sequence<Dim>();
198 /* Step 1. Determine all particles which are close to the global domain boundary */
199 size_type numBoundaryParticles = 0;
200 Kokkos::parallel_reduce(
201 "count boundary particles", numLoc,
202 KOKKOS_LAMBDA(const size_t& i, size_type& sum) {
203 if (isCloseToBoundary(is, positions(i), globalRegion, periodic, overlap)) {
204 ++sum;
205 }
206 },
207 Kokkos::Sum<size_type>(numBoundaryParticles));
208 if (numBoundaryParticles == 0) {
209 return;
210 }
211
212 /* Step 2. compute prefix sum to collect boundary particle indices */
213 hash_type boundaryIndices("boundaryIndices", numBoundaryParticles);
214 Kokkos::parallel_scan(
215 "count boundary particles", numLoc,
216 KOKKOS_LAMBDA(const size_t& i, size_type& sum, bool final) {
217 if (isCloseToBoundary(is, positions(i), globalRegion, periodic, overlap)) {
218 if (final) {
219 boundaryIndices(sum) = i;
220 }
221 ++sum;
222 }
223 });
224
225 /* Step 3. copy given particles and all its attributes. A separate function is needed as
226 * lambdas with captures do not work with nvcc and template default argument ot the layout
227 * somehow. NOTE numLoc will change after this call.
228 */
229 detail::copyAttributes(pc, boundaryIndices);
230
231 /* Step 4. set the position of the copied particles to their periodic image */
232 for (unsigned d = 0; d < Dim; ++d) {
233 if (!periodic[d]) {
234 continue;
235 }
236 const auto min = globalRegion[d].min();
237 const auto length = globalRegion[d].length();
238 const auto middle = min + length / 2;
239 Kokkos::parallel_for(
240 "correct positions",
241 // the copied particles are appended at the end of the attributes
242 Kokkos::RangePolicy<position_execution_space>(numLoc,
243 numLoc + numBoundaryParticles),
244 KOKKOS_LAMBDA(const size_t& i) {
245 positions(i)[d] += positions(i)[d] > middle ? -length : length;
246 });
247 }
248
249 IpplTimings::stopTimer(timer);
250 }

252 template <typename T, unsigned Dim, class Mesh, typename... Properties>
253 template <class ParticleContainer>
254 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::particleExchange(
255 ParticleContainer& pc) {
256 /* Apply Boundary Conditions */
257 static IpplTimings::TimerRef ParticleBCTimer = IpplTimings::getTimer("particleBC");
258 IpplTimings::startTimer(ParticleBCTimer);
259 this->applyBC(pc.R, this->rlayout_m.getDomain());
260 createPeriodicGhostParticles(pc);
261 IpplTimings::stopTimer(ParticleBCTimer);
262
263 /* Update Timer for the rest of the function */
264 static IpplTimings::TimerRef ParticleUpdateTimer = IpplTimings::getTimer("updateParticle");
265 IpplTimings::startTimer(ParticleUpdateTimer);
266
267 int nRanks = Comm->size();
268 if (nRanks < 2) {
269 return;
270 }
271
272 /* particle MPI exchange:
273 * 1. figure out which particles need to go where -> locateParticles(...)
274 * 2. fill send buffer and send particles
275 * 3. delete invalidated particles
276 * 4. receive particles
277 */
278
279 // 1. figure out which particles need to go where -> locateParticles(...) ============= //
280
281 static IpplTimings::TimerRef locateTimer = IpplTimings::getTimer("locateParticles");
282 IpplTimings::startTimer(locateTimer);
283
284 /* Rank-local number of particles */
285 size_type localnum = pc.getLocalNum();
286
287 /* particleRanks are the indices correspond to the indices of the local particles,
288 * the values correspond to the ranks to which the particles need to be sent note that the
289 * size is not known yet as particles may belong to multiple ranks
290 * particleRankOffsets are the offsets of each particle as a particle
291 * can be sent to multiple ranks
292 */
293 locate_type particleRanks("particles' MPI ranks");
294 locate_type particleRankOffsets("particles' MPI rank offsets", localnum + 1);
295
296 /* The indices are the indices of the particles,
297 * the boolean values describe whether the particle has left the current rank
298 * 0 --> particle valid (inside current rank)
299 * 1 --> particle invalid (left rank)
300 */
301 bool_type invalidParticles("validity of particles", localnum);
302
303 /* The indices are the MPI ranks,
304 * the values are the number of particles are sent to that rank from myrank
305 */
306 locate_type rankSendCount_dview("rankSendCount Device", nRanks);
307
308 /* The indices have no particular meaning,
309 * the values are the MPI ranks to which we need to send
310 */
311 locate_type destinationRanks_dview("destinationRanks Device", nRanks);
312
313 /* nInvalid is the number of invalid particles
314 * nDestinationRanks is the number of MPI ranks we need to send to
315 */
316 const auto [nInvalid, nDestinationRanks] =
317 locateParticles(pc, particleRanks, particleRankOffsets, invalidParticles,
318 rankSendCount_dview, destinationRanks_dview);
319
320 /* Host space copy of rankSendCount_dview */
321 auto rankSendCount_hview =
322 Kokkos::create_mirror_view_and_copy(Kokkos::HostSpace(), rankSendCount_dview);
323
324 /* Host Space copy of destinationRanks_dview */
325 auto destinationRanks_hview =
326 Kokkos::create_mirror_view_and_copy(Kokkos::HostSpace(), destinationRanks_dview);
327 Kokkos::fence();
328
329 IpplTimings::stopTimer(locateTimer);
330
331 // 2. fill send buffer and send particles =============================================== //
332
333 // 2.1 Remote Memory Access window for one-sided communication
334
335 static IpplTimings::TimerRef preprocTimer = IpplTimings::getTimer("sendPreprocess");
336 IpplTimings::startTimer(preprocTimer);
337
338 std::fill(this->nRecvs_m.begin(), this->nRecvs_m.end(), 0);
339
340 this->window_m.fence(0);
341
342 // Prepare RMA window for the ranks we need to send to
343 for (size_t ridx = 0; ridx < nDestinationRanks; ridx++) {
344 int rank = destinationRanks_hview[ridx];
345 if (rank == Comm->rank()) {
346 // we do not need to send to ourselves
347 continue;
348 }
349 this->window_m.template put<size_type>(rankSendCount_hview(rank), rank, Comm->rank());
350 }
351 this->window_m.fence(0);
352
353 IpplTimings::stopTimer(preprocTimer);
354
355 // 2.2 Particle Sends
356
357 static IpplTimings::TimerRef sendTimer = IpplTimings::getTimer("particleSend");
358 IpplTimings::startTimer(sendTimer);
359
360 std::vector<MPI_Request> requests(0);
361
362 int tag = Comm->next_tag(mpi::tag::P_SPATIAL_LAYOUT, mpi::tag::P_LAYOUT_CYCLE);
363
364 for (size_t ridx = 0; ridx < nDestinationRanks; ridx++) {
365 int rank = destinationRanks_hview[ridx];
366 if (rank == Comm->rank()) {
367 continue;
368 }
369 hash_type hash("hash", rankSendCount_hview(rank));
370 fillHash(rank, particleRanks, particleRankOffsets, hash);
371 pc.sendToRank(rank, tag, requests, hash);
372 }
373
374 IpplTimings::stopTimer(sendTimer);
375
376 // 3. Internal destruction of invalid particles ======================================= //
377
378 static IpplTimings::TimerRef destroyTimer = IpplTimings::getTimer("particleDestroy");
379 IpplTimings::startTimer(destroyTimer);
380
381 pc.internalDestroy(invalidParticles, nInvalid);
382 Kokkos::fence();
383
384 IpplTimings::stopTimer(destroyTimer);
385
386 // 4. Receive Particles ================================================================ //
387
388 static IpplTimings::TimerRef recvTimer = IpplTimings::getTimer("particleRecv");
389 IpplTimings::startTimer(recvTimer);
390
391 for (int rank = 0; rank < nRanks; ++rank) {
392 if (this->nRecvs_m[rank] > 0) {
393 pc.recvFromRank(rank, tag, this->nRecvs_m[rank]);
394 }
395 }
396 IpplTimings::stopTimer(recvTimer);
397
398 IpplTimings::startTimer(sendTimer);
399
400 if (requests.size() > 0) {
401 MPI_Waitall(requests.size(), requests.data(), MPI_STATUSES_IGNORE);
402 }
403 IpplTimings::stopTimer(sendTimer);
404
405 IpplTimings::stopTimer(ParticleUpdateTimer);
406 }

433 template <typename T, unsigned Dim, class Mesh, typename... Properties>
434 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::fillHash(
435 int rank, const locate_type& ranks, const locate_type& offsets, hash_type& hash) {
436 /* Compute the prefix sum and fill the hash
437 */
438 using policy_type = Kokkos::RangePolicy<position_execution_space>;
439 Kokkos::parallel_scan(
440 "ParticleSpatialLayout::fillHash()", policy_type(0, offsets.extent(0) - 1),
441 KOKKOS_LAMBDA(const size_t i, int& idx, const bool final) {
442 /* Check if this particle belongs to our target rank. Ranks of particle i are stored
443 * from offsets(i) to offsets(i + 1)
444 */
445 bool belongs_to_rank = false;
446 const size_t start_rank_idx = offsets(i);
447 const size_t end_rank_idx = offsets(i + 1);
448
449 for (size_t rank_idx = start_rank_idx; rank_idx < end_rank_idx; ++rank_idx) {
450 if (ranks(rank_idx) == rank) {
451 belongs_to_rank = true;
452 break; // Found it, no need to continue
453 }
454 }
455
456 if (final && belongs_to_rank) {
457 hash(idx) = i;
458 }
459
460 if (belongs_to_rank) {
461 idx += 1;
462 }
463 });
464 Kokkos::fence();
465 }

468 */
469 template <typename T, unsigned Dim, class Mesh, typename... Properties>
470 typename ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::locate_type
471 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::getFlatNeighbors(
472 const neighbor_list& neighbors) const {
473 /* collect all neighbors in a set. set instead of vector as with small number of ranks and
474 * periodic boundary conditions the same rank can be in the neighbor list multiple times.
475 */
476 std::set<int> neighborSet;
477 for (const auto& componentNeighbors : neighbors) {
478 for (const auto& nrank : componentNeighbors) {
479 neighborSet.insert(nrank);
480 }
481 }
482
483 // copy neighbors into view
484 locate_type flatNeighbors("Nearest neighbors IDs", neighborSet.size());
485 auto hostMirror = Kokkos::create_mirror_view(flatNeighbors);
486
487 size_type i = 0;
488 for (const auto& neighbor : neighborSet) {
489 hostMirror(i) = neighbor;
490 ++i;
491 }
492
493 Kokkos::deep_copy(flatNeighbors, hostMirror);
494 Kokkos::fence();
495 return flatNeighbors;
496 }

498 /* Helper function to get non-neighboring ranks
499 */
500 template <typename T, unsigned Dim, class Mesh, typename... Properties>
501 typename ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::locate_type
502 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::getNonNeighborRanks(
503 const locate_type& neighbors_view) const {
504 // Create a view of all non-neighbor ranks. make sure to exclude own its rank
505 const auto numNonNeighborRanks = Comm->size() - neighbors_view.extent(0) - 1;
506 locate_type nonNeighborRanks("Non Neighbor Ranks", numNonNeighborRanks);
507 if (numNonNeighborRanks == 0) {
508 return nonNeighborRanks;
509 }
510
511 // Step 1. Mark all non-neighbor ranks, by removing all neighbors and self
512 const auto total_ranks = Comm->rank();
513 bool_type is_remaining("is_remaining", total_ranks);
514 Kokkos::deep_copy(is_remaining, true);
515 Kokkos::fence();
516
517 const auto myRank = Comm->rank();
518 Kokkos::parallel_for(
519 "mark_comm_ranks", Kokkos::RangePolicy(myRank, myRank + 1),
520 KOKKOS_LAMBDA(const size_t& i) { is_remaining(i) = false; });
521 Kokkos::parallel_for(
522 "mark_comm_ranks", neighbors_view.extent(0),
523 KOKKOS_LAMBDA(const size_t& i) { is_remaining(neighbors_view(i)) = false; });
524 Kokkos::fence();
525
526 // Step 2. Fill remaining ranks
527 Kokkos::View<size_type> counter("counter");
528 Kokkos::parallel_for(
529 "fill_remaining", total_ranks, KOKKOS_LAMBDA(const size_t& i) {
530 if (is_remaining(i)) {
531 const size_type idx = Kokkos::atomic_fetch_inc(&counter());
532 nonNeighborRanks(idx) = i;
533 }
534 });
535 return nonNeighborRanks;
536 }

538 template <typename T, unsigned Dim, class Mesh, typename... Properties>
539 template <typename ParticleContainer>
540 std::pair<detail::size_type, detail::size_type>
541 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::locateParticles(
542 const ParticleContainer& pc, locate_type& ranks, locate_type& rankOffsets,
543 bool_type& invalid, locate_type& nSends_dview, locate_type& sends_dview) const {
544 const auto positions = pc.R.getView();
545 const auto regions = this->rlayout_m.getdLocalRegions();
546 const auto myRank = Comm->rank();
547 const auto localNum = pc.getLocalNum();
548 const T overlap = rcutoff_m;
549 constexpr auto is = std::make_index_sequence<Dim>();
550
553 locate_type counts("counts", localNum);
554 locate_type outsideIds("Particles outside of neighborhood", localNum);
555
557 size_type outsideCount = 0;
559 size_type invalidCount = 0;
560
562 locate_type neighbors_view = getFlatNeighbors(this->flayout_m.getNeighbors());
563
564 /* red_val: Used to reduce both the number of invalid particles and the number of particles
565 * outside the neighborhood (Kokkos::parallel_scan doesn't allow multiple reduction
566 * values, so we use the helper class increment_type). First element updates InvalidCount,
567 * second one updates outsideCount.
568 */
569 increment_type red_val;
570 red_val.init();
571
578 static IpplTimings::TimerRef neighborSearch = IpplTimings::getTimer("neighborSearch");
579 IpplTimings::startTimer(neighborSearch);
580
581 /* First Pass: count the numbers of neighbor ranks (including self) a particle belongs to */
582 Kokkos::parallel_for(
583 "ParticleSpatialLayout::locateParticles()", localNum, KOKKOS_LAMBDA(const size_t& i) {
584 const bool inCurr = positionInRegion(is, positions(i), regions(myRank), overlap);
585
586 size_type count = inCurr;
587 invalid(i) = !inCurr;
588 // Count neighboring regions
589 for (size_type j = 0; j < neighbors_view.extent(0); ++j) {
590 const size_type rank = neighbors_view(j);
591 if (positionInRegion(is, positions(i), regions(rank), overlap)) {
592 ++count;
593 }
594 }
595
596 counts(i) = count;
597 });
598 Kokkos::fence();
599
600 /* Second Pass: collect number of particles outside this ranks region and the indices of the
601 * respective particles. Note that in comparison to ParticleSpatialLayout::locateParticles()
602 * particle can be in multiple ranks. For this reason if a particle is in a neighbor but not
603 * here, then we still need to search in all ranks as there is no way to get second
604 * neighbors.
605 */
606 Kokkos::parallel_scan(
607 "count_outside", localNum,
608 KOKKOS_LAMBDA(const size_type i, increment_type& val, const bool final) {
609 const bool inCurr = !invalid(i);
610 if (final && !inCurr) {
611 outsideIds(val.count[1]) = i;
612 }
613 bool increment[2] = {invalid(i), !inCurr};
614 val += increment;
615 },
616 red_val);
617 Kokkos::fence();
618
619 invalidCount = red_val.count[0];
620 outsideCount = red_val.count[1];
621
622 IpplTimings::stopTimer(neighborSearch);
623
625 static IpplTimings::TimerRef nonNeighboringParticles =
626 IpplTimings::getTimer("nonNeighboringParticles");
627 IpplTimings::startTimer(nonNeighboringParticles);
628
629 using mdrange_type = Kokkos::MDRangePolicy<Kokkos::Rank<2>, position_execution_space>;
630
631 /* Count the number of non-neighbor ranks each particle belongs to */
632 locate_type outsideCounts("counts of outside neighbors", outsideCount);
633 locate_type nonNeighborsView = getNonNeighborRanks(neighbors_view);
634 if (outsideCount > 0 && nonNeighborsView.extent(0) > 0) {
635 Kokkos::deep_copy(outsideCounts, 0);
636 Kokkos::parallel_for(
637 "ParticleSpatialLayout::leftParticles()",
638 mdrange_type({0, 0}, {outsideCount, nonNeighborsView.extent(0)}),
639 KOKKOS_LAMBDA(const size_t i, const size_type j) {
641 const size_type pId = outsideIds(i);
642 const auto rank = nonNeighborsView(j);
643
645 if (positionInRegion(is, positions(pId), regions(rank), overlap)) {
646 Kokkos::atomic_increment(&outsideCounts(i));
647 }
648 });
649 Kokkos::fence();
650 Kokkos::parallel_for(
651 "ParticleSpatialLayout::leftParticles()", outsideCount,
652 KOKKOS_LAMBDA(const size_t& i) { counts(outsideIds(i)) += outsideCounts(i); });
653 Kokkos::fence();
654 }
655 IpplTimings::stopTimer(nonNeighboringParticles);
656
657 IpplTimings::startTimer(neighborSearch);
658 /* prefix sum for particle rank offsets */
659 Kokkos::deep_copy(Kokkos::subview(rankOffsets, 0), 0);
660 Kokkos::parallel_scan(
661 "ParticleSpatialLayout::locateParticles()", localNum,
662 KOKKOS_LAMBDA(const size_t i, size_type& localSum, const bool final) {
663 const auto count_i = counts(i);
664 if (final) {
665 rankOffsets(i + 1) = localSum + count_i;
666 }
667 localSum += count_i;
668 });
669 Kokkos::fence();
670
671 /* Get total number of assignments for allocation from the last entry of offsets */
672 auto total_assignments = Kokkos::create_mirror_view(Kokkos::subview(rankOffsets, localNum));
673 Kokkos::deep_copy(total_assignments, Kokkos::subview(rankOffsets, localNum));
674 Kokkos::fence();
675 Kokkos::resize(ranks, total_assignments());
676
677 /* Last Pass: fill the rank data */
678 Kokkos::parallel_for(
679 "ParticleSpatialLayout::locateParticles()", localNum, KOKKOS_LAMBDA(const size_t& i) {
680 const size_t offset = rankOffsets(i);
681 size_type local_count = 0;
682 if (positionInRegion(is, positions(i), regions(myRank), overlap)) {
683 ranks(offset) = myRank;
684 local_count++;
685 }
686 for (size_t j = 0; j < neighbors_view.extent(0); ++j) {
687 const auto nRank = neighbors_view(j);
688 if (positionInRegion(is, positions(i), regions(nRank), overlap)) {
689 ranks(offset + local_count) = nRank;
690 local_count++;
691 }
692 }
693 });
694 Kokkos::fence();
695
696 IpplTimings::stopTimer(neighborSearch);
697
698 /* Last Pass: add the data of the outside particles to the ranks */
699 IpplTimings::startTimer(nonNeighboringParticles);
700 if (outsideCount > 0) {
701 Kokkos::parallel_for(
702 "ParticleSpatialLayout::leftParticles()", outsideCount,
703 KOKKOS_LAMBDA(const size_t& i) {
705 const size_type pId = outsideIds(i);
706 const size_type offset = rankOffsets(pId) + counts(pId);
707 for (size_t local_count = 0, j = 0; j < nonNeighborsView.extent(0); ++j) {
708 const auto rank = nonNeighborsView(j);
709 if (positionInRegion(is, positions(i), regions(rank), overlap)) {
710 ranks(offset + local_count) = rank;
711 local_count++;
712 }
713 }
714 });
715 Kokkos::fence();
716 }
717 IpplTimings::stopTimer(nonNeighboringParticles);
718
719 /* compute the number of sends to all ranks */
720 Kokkos::deep_copy(nSends_dview, 0);
721 Kokkos::parallel_for(
722 "Calculate nSends", Kokkos::RangePolicy<size_t>(0, ranks.extent(0)),
723 KOKKOS_LAMBDA(const size_t i) {
724 size_type rank = ranks(i);
725 Kokkos::atomic_increment(&nSends_dview(rank));
726 });
727 Kokkos::fence();
728
729 /* compute the ranks to send to and the number of ranks to send to*/
730 Kokkos::View<size_type> rankSends("Number of Ranks we need to send to");
731 Kokkos::parallel_for(
732 "Calculate sends", Kokkos::RangePolicy<size_t>(0, nSends_dview.extent(0)),
733 KOKKOS_LAMBDA(const size_t rank) {
734 if (nSends_dview(rank) != 0) {
735 size_type index = Kokkos::atomic_fetch_inc(&rankSends());
736 sends_dview(index) = rank;
737 }
738 });
739 Kokkos::fence();
740 size_type temp;
741 Kokkos::deep_copy(temp, rankSends);
742 Kokkos::fence();
743
744 return {invalidCount, temp};
745 }

823 template <typename T, unsigned Dim, class Mesh, typename... Properties>
824 template <class ParticleContainer>
825 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::buildCells(
826 ParticleContainer& pc) {
827 static IpplTimings::TimerRef cellBuildTimer = IpplTimings::getTimer("cellBuildTimer");
828 IpplTimings::startTimer(cellBuildTimer);
829
830 // get local variables of all necessary data as needed for the Kokkos parallel loops
831 const auto rank = Comm->rank();
832 const size_type numLoc = pc.getLocalNum();
833 const auto positions = pc.R.getView();
834 const auto totalCells = totalCells_m;
835 const auto numLocalCells = numLocalCells_m;
836 const auto localRegion = this->rlayout_m.gethLocalRegions()(rank);
837 const auto& cellWidth = cellWidth_m;
838 const auto cellStrides = cellStrides_m;
839 const auto cellPermutationForward = cellPermutationForward_m;
840
841 using int_type = typename particle_neighbor_list_type::value_type;
842
843 // allocate required (temporary) Kokkos views
844 hash_type cellIndex("cellIndex", numLoc);
845 hash_type cellParticleCount = cellParticleCount_m;
846 hash_type cellStartingIdx = cellStartingIdx_m;
847 hash_type cellCurrentIdx("cellCurrentIdx", totalCells + 1);
848
849 using range_policy = Kokkos::RangePolicy<position_execution_space>;
850
851 /* Step 1. calculate cell index for each particle and keep track of how many particles are
852 * in each cell
853 */
854 Kokkos::deep_copy(cellParticleCount, 0);
855 Kokkos::parallel_for(
856 "CalcCellIndices", range_policy(0, numLoc), KOKKOS_LAMBDA(const size_t& i) {
857 const auto locCellIndex = getCellIndex(positions(i), localRegion, cellWidth);
858 const auto locCellIndexFlat =
859 toFlatCellIndex(locCellIndex, cellStrides, cellPermutationForward);
860 Kokkos::atomic_increment(&cellParticleCount(locCellIndexFlat));
861 cellIndex(i) = locCellIndexFlat;
862 });
863 Kokkos::fence();
864
865 /* Step 2. compute starting indices for each cell from the counts */
866 Kokkos::parallel_scan(
867 "CalcStartingIndices", range_policy(0, totalCells),
868 KOKKOS_LAMBDA(const size_t i, int_type& localSum, bool isFinal) {
869 if (isFinal) {
870 cellStartingIdx(i) = localSum;
871 }
872 localSum += cellParticleCount(i);
873 });
874 /* set last position of cell staring index to numLoc*/
875 Kokkos::deep_copy(
876 Kokkos::subview(cellStartingIdx, Kokkos::make_pair(totalCells, totalCells + 1)),
877 numLoc);
878 Kokkos::fence();
879
880 Kokkos::deep_copy(cellCurrentIdx, cellStartingIdx);
881 Kokkos::fence();
882
883 hash_type newIndex("newIndex", numLoc);
884 hash_type newCellIndex("cellIndex", numLoc);
885
886 /* Step 3. compute new indices for the particles such that they are sorted according to
887 * cellStaringIdx and sort cell indices of the particles in tandem
888 */
889 Kokkos::parallel_for(
890 "Calculate new Indices", range_policy(0, numLoc), KOKKOS_LAMBDA(const size_type& i) {
891 const auto locCellIndex = cellIndex(i);
892 const size_type newIdx = Kokkos::atomic_fetch_inc(&cellCurrentIdx(locCellIndex));
893 newIndex(i) = newIdx;
894 newCellIndex(newIdx) = locCellIndex;
895 });
896 Kokkos::fence();
897
898 /* Step 4. Sort all particles (and all their attributes) according to then new indices
899 * (maybe there is a better solution). A separate function is needed as lambdas with
900 * captures do not work with nvcc and template default argument ot the layout somehow.
901 */
902 detail::sortParticles(pc, newIndex);
903
904 /* Step 5. set local number of particles (excluding ghost particles) is the value of
905 * cellStartingIdx at index numLocalCells*/
906 auto numLocalParticles =
907 Kokkos::create_mirror_view(Kokkos::subview(cellStartingIdx, numLocalCells));
908 Kokkos::deep_copy(numLocalParticles, Kokkos::subview(cellStartingIdx, numLocalCells));
909
910 numLocalParticles_m = numLocalParticles();
911 pc.setLocalNum(numLocalParticles_m);
912
913 /* store the new cell indices */
914 cellIndex_m = newCellIndex;
915
916 IpplTimings::stopTimer(cellBuildTimer);
917 }

919 template <typename T, unsigned Dim, class Mesh, typename... Properties>
920 KOKKOS_INLINE_FUNCTION constexpr
921 typename ParticleSpatialOverlapLayout<T, Dim, Mesh,
922 Properties...>::cell_particle_neighbor_list_type
923 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::getCellNeighbors(
924 const CellIndex_t& cellIndex, const Vector<size_type, Dim>& cellStrides,
925 const hash_type& cellPermutationForward) {
926 /* Generate all 3^Dim combinations of offsets (-1, 0, +1) for each dimension by using
927 * "base-3" representation of neighbor index. in base-3 each digit is 0, 1, 2 subtracting
928 * one leads to the desired offsets -1, 0, +1.
929 */
930 constexpr size_type numNeighbors = detail::countHypercubes(Dim);
931 cell_particle_neighbor_list_type neighborIndices;
932 for (size_type neighborIdx = 0; neighborIdx < numNeighbors; ++neighborIdx) {
933 index_t temp = neighborIdx;
934
935 /* extract the offsets from the base-3 representation of the neighborIdx */
936 auto neighborCellIndex = cellIndex;
937 for (unsigned d = 0; d < Dim; ++d) {
938 neighborCellIndex(d) += (temp % 3) - 1;
939 temp /= 3;
940 }
941
942 neighborIndices[neighborIdx] =
943 toFlatCellIndex(neighborCellIndex, cellStrides, cellPermutationForward);
944 }
945 return neighborIndices;
946 }

960 Properties...>::particle_neighbor_list_type
961 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::getParticleNeighbors(
962 const vector_type& pos, const ParticleNeighborData& particleNeighborData) {
963 /* get the cell index corresponding to pos */
964 const auto locCellIndex =
965 getCellIndex(pos, particleNeighborData.region, particleNeighborData.cellStrides,
966 particleNeighborData.cellWidth);
967 constexpr size_type numNeighbors = detail::countHypercubes(Dim);
968
969 /* cell neighbors */
970 const auto neighbors = getCellNeighbors(locCellIndex, particleNeighborData.cellStrides,
971 particleNeighborData.cellPermutationForward);
972
973 /* Get sizes of the cell neighbors and total particle neighbors */
974 size_type totalParticleInNeighbors = 0;
975 size_type maxParticleInNeighbors = 0;
976
977 Kokkos::Array<typename hash_type::value_type, numNeighbors> neighborSizes;
978 Kokkos::Array<typename hash_type::value_type, numNeighbors> neighborOffsets;
979 // #pragma unroll
980 for (size_type neighborIdx = 0; neighborIdx < numNeighbors; ++neighborIdx) {
981 auto n = particleNeighborData.cellParticleCount(neighbors[neighborIdx]);
982 neighborSizes[neighborIdx] = n;
983 maxParticleInNeighbors = std::max<size_type>(n, maxParticleInNeighbors);
984 neighborOffsets[neighborIdx] = totalParticleInNeighbors;
985 totalParticleInNeighbors += n;
986 }
987
988 /* Collect the neighbor particles from all cell neighbors */
989 particle_neighbor_list_type neighborList("Neighbor list", totalParticleInNeighbors);
990
991 using mdrange_policy = Kokkos::MDRangePolicy<Kokkos::Rank<2>, position_execution_space>;
992 Kokkos::parallel_for(
993 "collect neighbors", mdrange_policy({0, 0}, {numNeighbors, maxParticleInNeighbors}),
994 KOKKOS_LAMBDA(const size_type& i, const size_type& j) {
995 if (j < neighborSizes[i]) {
996 neighborList(neighborOffsets[i] + j) =
997 particleNeighborData.cellStartingIdx(neighbors[i]) + j;
998 }
999 });
1000 Kokkos::fence();
1001
1002 return neighborList;

1058 template <typename T, unsigned Dim, class Mesh, typename... Properties>
1059 template <typename ExecutionSpace, typename Functor>
1060 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::forEachPair(Functor&& f) const {
1061 static IpplTimings::TimerRef interactionTimer = IpplTimings::getTimer("PPInteractionTimer");
1062 IpplTimings::startTimer(interactionTimer);
1063
1064 /* get local variables necessary for Kokkos parallel regions */
1065 const auto cellStartingIdx = cellStartingIdx_m;
1066 const auto cellParticleCount = cellParticleCount_m;
1067 const auto cellPermutationForward = cellPermutationForward_m;
1068 const auto cellPermutationBackward = cellPermutationBackward_m;
1069 const auto& cellStrides = cellStrides_m;
1070 const auto& numCells = numCells_m;
1071
1072 constexpr auto numCellNeighbors = detail::countHypercubes(Dim);
1073
1074 /* Iterate over all local cells (non-ghost cells) and let all particles of this cell
1075 * interact with all particles from all particles with all cell neighbors
1076 */
1077 using team_policy_t = Kokkos::TeamPolicy<ExecutionSpace>;
1078 using team_t = typename team_policy_t::member_type;
1079 Kokkos::parallel_for(
1080 "ParticleSpatialOverlapLayout::forEachPair()",
1081 team_policy_t(numLocalCells_m, Kokkos::AUTO()), KOKKOS_LAMBDA(const team_t& team) {
1082 const size_type cellIdxFlat = team.league_rank();
1083 if (cellParticleCount(cellIdxFlat) == 0) {
1084 return;
1085 }
1086
1087 const auto cellParticleOffset = cellStartingIdx(cellIdxFlat);
1088 const auto numCellParticles = cellParticleCount(cellIdxFlat);
1089
1090 /* get nd-cell-index and its neighbors */
1091 const auto cellIdx = toCellIndex(cellPermutationBackward(cellIdxFlat), numCells);
1092 const auto cellNeighbors =
1093 getCellNeighbors(cellIdx, cellStrides, cellPermutationForward);
1094
1095 /* iterate over all cell neighbors */
1096 Kokkos::parallel_for(
1097 Kokkos::TeamThreadRange(team, numCellNeighbors), [&](const size_t& n) {
1098 const auto neighborCellIdx = cellNeighbors[n];
1099 const auto neighborCellParticleOffset = cellStartingIdx(neighborCellIdx);
1100 const auto numNeighborCellParticles = cellParticleCount(neighborCellIdx);
1101
1102 /* iterate over all combinations of this cell and neighboring cells
1103 * particles
1104 */
1105 Kokkos::parallel_for(Kokkos::ThreadVectorMDRange<Kokkos::Rank<2>, team_t>(
1106 team, numCellParticles, numNeighborCellParticles),
1107 [&](const size_t& i, const size_t& j) {
1108 const auto particleIdx = cellParticleOffset + i;
1109 const auto neighborIdx =
1110 neighborCellParticleOffset + j;
1111
1112 if (neighborIdx == particleIdx) {
1113 return;
1114 }
1115
1116 f(particleIdx, neighborIdx);
1117 });
1118 });
1119 });
1120 Kokkos::fence();
1121
1122 IpplTimings::stopTimer(interactionTimer);
1123 }
