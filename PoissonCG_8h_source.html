50 void setSolver(lhs_type lhs) {
51 std::string solver_type = this->params_m.template get<std::string>("solver");
52 typename lhs_type::Mesh_t mesh = lhs.get_mesh();
53 typename lhs_type::Layout_t layout = lhs.getLayout();
54 double beta = 0;
55 double alpha = 0;
56 if (solver_type == "preconditioned") {
57 algo_m = std::move(
58 std::make_unique<PCG<OperatorRet, LowerRet, UpperRet, UpperAndLowerRet,
59 InverseDiagonalRet, FieldLHS, FieldRHS>>());
60 std::string preconditioner_type =
61 this->params_m.template get<std::string>("preconditioner_type");
62 int level = this->params_m.template get<int>("newton_level");
63 int degree = this->params_m.template get<int>("chebyshev_degree");
64 int inner = this->params_m.template get<int>("gauss_seidel_inner_iterations");
65 int outer = this->params_m.template get<int>("gauss_seidel_outer_iterations");
66 int richardson_iterations =
67 this->params_m.template get<int>("richardson_iterations");
68 int communication = this->params_m.template get<int>("communication");
69 // Analytical eigenvalues for the d dimensional laplace operator
70 // Going brute force through all possible eigenvalues seems to be the only way to
71 // find max and min
72
73 unsigned long n;
74 double h;
75 for (unsigned int d = 0; d < Dim; ++d) {
76 n = mesh.getGridsize(d);
77 h = mesh.getMeshSpacing(d);
78 double local_min = 4 / std::pow(h, 2); // theoretical maximum
79 double local_max = 0;
80 double test;
81 for (unsigned int i = 1; i < n; ++i) {
82 test = 4. / std::pow(h, 2) * std::pow(std::sin(i * M_PI * h / 2.), 2);
83 if (test > local_max) {
84 local_max = test;
85 }
86 if (test < local_min) {
87 local_min = test;
88 }
89 }
90 beta += local_max;
91 alpha += local_min;
92 }
93 if (communication) {
94 algo_m->setPreconditioner(
95 IPPL_SOLVER_OPERATOR_WRAPPER(-laplace, lhs_type),
96 IPPL_SOLVER_OPERATOR_WRAPPER(-lower_laplace, lhs_type),
97 IPPL_SOLVER_OPERATOR_WRAPPER(-upper_laplace, lhs_type),
98 IPPL_SOLVER_OPERATOR_WRAPPER(-upper_and_lower_laplace, lhs_type),
99 IPPL_SOLVER_OPERATOR_WRAPPER(negative_inverse_diagonal_laplace, lhs_type),
100 alpha, beta, preconditioner_type, level, degree, richardson_iterations,
101 inner, outer);
102 } else {
103 algo_m->setPreconditioner(
104 IPPL_SOLVER_OPERATOR_WRAPPER(-laplace, lhs_type),
105 IPPL_SOLVER_OPERATOR_WRAPPER(-lower_laplace_no_comm, lhs_type),
106 IPPL_SOLVER_OPERATOR_WRAPPER(-upper_laplace_no_comm, lhs_type),
107 IPPL_SOLVER_OPERATOR_WRAPPER(-upper_and_lower_laplace_no_comm, lhs_type),
108 IPPL_SOLVER_OPERATOR_WRAPPER(negative_inverse_diagonal_laplace, lhs_type),
109 alpha, beta, preconditioner_type, level, degree, richardson_iterations,
110 inner, outer);
111 }
112 } else {
113 algo_m =
114 std::move(std::make_unique<CG<OperatorRet, LowerRet, UpperRet, UpperAndLowerRet,
115 InverseDiagonalRet, FieldLHS, FieldRHS>>());
116 }
117 }
