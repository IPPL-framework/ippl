21 template <unsigned Dim>
22 int FieldLayout<Dim>::getMatchingIndex(int index) {
23 // If we are touching another boundary component, that component must be parallel to us,
24 // so any 2s are unchanged. All other digits must be swapped because otherwise we would
25 // share a higher dimension boundary region with that other component and the digit
26 // would be 2
27 static const int digit_swap[3] = {1, 0, 2};
28 int match = 0;
29 for (unsigned d = 1; d < detail::countHypercubes(Dim); d *= 3) {
30 match += digit_swap[index % 3] * d;
31 index /= 3;
32 }
33 return match;
34 }

70 template <unsigned Dim>
71 void FieldLayout<Dim>::initialize(const NDIndex<Dim>& domain, std::array<bool, Dim> isParallel,
72 bool isAllPeriodic) {
73 int nRanks = comm.size();
74
75 gDomain_m = domain;
76
77 isAllPeriodic_m = isAllPeriodic;
78
79 isParallelDim_m = isParallel;
80
81 if (nRanks < 2) {
82 Kokkos::resize(dLocalDomains_m, nRanks);
83 Kokkos::resize(hLocalDomains_m, nRanks);
84 hLocalDomains_m(0) = domain;
85 Kokkos::deep_copy(dLocalDomains_m, hLocalDomains_m);
86 return;
87 }
88
89 /* Check to see if we have too few elements to partition. If so, reduce
90 * the number of ranks (if necessary) to just the number of elements along
91 * parallel dims.
92 */
93 long totparelems = 1;
94 for (unsigned d = 0; d < Dim; ++d) {
95 totparelems *= domain[d].length();
96 }
97
98 if (totparelems < nRanks) {
99 nRanks = totparelems;
100 }
101
102 Kokkos::resize(dLocalDomains_m, nRanks);
103 Kokkos::resize(hLocalDomains_m, nRanks);
104
105 detail::Partitioner<Dim> partitioner;
106 partitioner.split(domain, hLocalDomains_m, isParallel, nRanks);
107
108 findNeighbors();
109
110 Kokkos::deep_copy(dLocalDomains_m, hLocalDomains_m);
111
112 calcWidths();
113 }

163 template <unsigned Dim>
164 void FieldLayout<Dim>::calcWidths() {
165 // initialize widths first
166 for (unsigned int d = 0; d < Dim; ++d) {
167 minWidth_m[d] = gDomain_m[d].length();
168 }
169
170 using size_type = typename host_mirror_type::size_type;

172 const NDIndex_t& dom = hLocalDomains_m(i);
173 for (unsigned int d = 0; d < Dim; ++d) {
174 if ((unsigned int)dom[d].length() < minWidth_m[d]) {

181 template <unsigned Dim>
182 void FieldLayout<Dim>::findPeriodicNeighbors(const int nghost, const NDIndex<Dim>& localDomain,
183 NDIndex<Dim>& grown, NDIndex<Dim>& neighborDomain,
184 const int rank,
185 std::map<unsigned int, int>& offsets, unsigned d0,
186 unsigned codim) {
187 for (unsigned int d = d0; d < Dim; ++d) {
188 // 0 - check upper boundary
189 // 1 - check lower boundary
190 for (int k = 0; k < 2; ++k) {
191 auto offset = offsets[d] = getPeriodicOffset(localDomain, d, k);
192 if (offset == 0) {
193 continue;
194 }
195
196 grown[d] += offset;
197
198 if (grown.touches(neighborDomain)) {
199 auto intersect = grown.intersect(neighborDomain);
200 for (auto& [d, offset] : offsets) {
201 neighborDomain[d] -= offset;
202 }
203 addNeighbors(grown, localDomain, neighborDomain, intersect, nghost, rank);
204 for (auto& [d, offset] : offsets) {
205 neighborDomain[d] += offset;
206 }
207 }
208 if (codim + 1 < Dim) {
209 findPeriodicNeighbors(nghost, localDomain, grown, neighborDomain, rank, offsets,
210 d + 1, codim + 1);
211 }
212
213 grown[d] -= offset;
214 offsets.erase(d);
215 }
216 }

219 template <unsigned Dim>
220 void FieldLayout<Dim>::findNeighbors(int nghost) {
221 /* We need to reset the neighbor list
222 * and its ranges because of the repartitioner.
223 */
224 for (size_t i = 0; i < detail::countHypercubes(Dim) - 1; i++) {
225 neighbors_m[i].clear();
226 neighborsSendRange_m[i].clear();
227 neighborsRecvRange_m[i].clear();
228 }
229
230 int myRank = comm.rank();
231
232 // get my local box
233 auto& nd = hLocalDomains_m[myRank];
234
235 // grow the box by nghost cells in each dimension
236 auto gnd = nd.grow(nghost);
237
238 static IpplTimings::TimerRef findInternalNeighborsTimer =
239 IpplTimings::getTimer("findInternal");
240 static IpplTimings::TimerRef findPeriodicNeighborsTimer =
241 IpplTimings::getTimer("findPeriodic");
242 for (int rank = 0; rank < comm.size(); ++rank) {
243 if (rank == myRank) {
244 // do not compare with my domain
245 continue;
246 }
247
248 auto& ndNeighbor = hLocalDomains_m[rank];
249 IpplTimings::startTimer(findInternalNeighborsTimer);
250 // For inter-processor neighbors
251 if (gnd.touches(ndNeighbor)) {
252 auto intersect = gnd.intersect(ndNeighbor);
253 addNeighbors(gnd, nd, ndNeighbor, intersect, nghost, rank);
254 }
255 IpplTimings::stopTimer(findInternalNeighborsTimer);
256
257 IpplTimings::startTimer(findPeriodicNeighborsTimer);
258 if (isAllPeriodic_m) {
259 std::map<unsigned int, int> offsets;
260 findPeriodicNeighbors(nghost, nd, gnd, ndNeighbor, rank, offsets);
261 }
262 IpplTimings::stopTimer(findPeriodicNeighborsTimer);
263 }

266 template <unsigned Dim>
267 void FieldLayout<Dim>::addNeighbors(const NDIndex_t& gnd, const NDIndex_t& nd,
268 const NDIndex_t& ndNeighbor, const NDIndex_t& intersect,
269 int nghost, int rank) {
270 bound_type rangeSend, rangeRecv;
271 rangeSend = getBounds(nd, ndNeighbor, nd, nghost);
272
273 rangeRecv = getBounds(ndNeighbor, nd, nd, nghost);
274
275 int index = 0;
276 for (unsigned d = 0, digit = 1; d < Dim; d++, digit *= 3) {
277 // For each dimension, check what kind of intersection we have and construct
278 // an index for the component based on its base-3 representation with the
279 // following properties for each digit:
280 // 0 - touching the lower axis value
281 // 1 - touching the upper axis value
282 // 2 - parallel to the axis
283 if (intersect[d].length() == 1) {
284 if (gnd[d].first() != intersect[d].first()) {
285 index += digit;
286 }
287 } else {
288 index += 2 * digit;
289 }
290 }
291 neighbors_m[index].push_back(rank);
292 neighborsSendRange_m[index].push_back(rangeSend);
293 neighborsRecvRange_m[index].push_back(rangeRecv);
294 }

296 template <unsigned Dim>
297 typename FieldLayout<Dim>::bound_type FieldLayout<Dim>::getBounds(const NDIndex_t& nd1,
298 const NDIndex_t& nd2,
299 const NDIndex_t& offset,
300 int nghost) {
301 NDIndex<Dim> gnd = nd2.grow(nghost);
302
303 NDIndex<Dim> overlap = gnd.intersect(nd1);
304
305 bound_type intersect;
306
307 /* Obtain the intersection bounds with local ranges of the view.
308 * Add "+1" to the upper bound since Kokkos loops always to "< extent".
309 */
310 for (size_t i = 0; i < Dim; ++i) {
311 intersect.lo[i] = overlap[i].first() - offset[i].first() /*offset*/ + nghost;
312 intersect.hi[i] = overlap[i].last() - offset[i].first() /*offset*/ + nghost + 1;
313 }
314
315 return intersect;
316 }

318 template <unsigned Dim>
319 int FieldLayout<Dim>::getPeriodicOffset(const NDIndex_t& nd, const unsigned int d,
320 const int k) {
321 switch (k) {
322 case 0:
323 if (nd[d].max() == gDomain_m[d].max()) {
324 return -gDomain_m[d].length();
325 }
326 break;
327 case 1:
328 if (nd[d].min() == gDomain_m[d].min()) {
329 return gDomain_m[d].length();
330 }
331 break;
332 default:
333 throw IpplException("FieldLayout:getPeriodicOffset", "k has to be either 0 or 1");
334 }
335 return 0;
336 }
