413 template <typename T = double>
414 class ESKernel {
415 public:
416 static constexpr bool has_width_template = true;
417 // Upper bound on the runtime width; matches the precomputed
418 // ES-kernel polynomial expansions (w = 4..15) used by NativeNUFFT.
419 static constexpr int max_width = 15;
420 using value_type = T;
421
422 static constexpr T default_tol = T(1e-10);
423 static constexpr T beta_factor = T(2.30);
424
429 KOKKOS_INLINE_FUNCTION explicit ESKernel(T tol = default_tol)
430 : w_(static_cast<int>(Kokkos::ceil(Kokkos::log10(T(1.0) / tol))) + 1)
431 , beta_(beta_factor * w_)
432 , tol_(tol) {}
433
439 KOKKOS_INLINE_FUNCTION ESKernel(int width, T beta)
440 : w_(width)
441 , beta_(beta)
442 , tol_(default_tol) {}
443
449 KOKKOS_INLINE_FUNCTION T operator()(T x) const {
450 x = Kokkos::abs(x);
451 return x >= T(1.0) ? T(0.0) : es_kernel_eval(x, w_);
452 }
453
458 template<int W>
459 KOKKOS_INLINE_FUNCTION T eval(T x) const {
460 return x >= T(1.0) ? T(0.0) : es_kernel_eval<T, W>(x);
461 }
462
464 KOKKOS_INLINE_FUNCTION int width() const { return w_; }
466 KOKKOS_INLINE_FUNCTION T beta() const { return beta_; }
468 KOKKOS_INLINE_FUNCTION T tol() const { return tol_; }
469
470 private:
471 int w_;
472 T beta_;
473 T tol_;
474 };
