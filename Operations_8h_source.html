83 MPI_Op operator()() {
84 std::pair<std::type_index, std::type_index> pear{std::type_index(typeid(Op)), std::type_index(typeid(Type))};
85 if(mpiOperations.contains(pear)){
86 return mpiOperations.at(pear);

88 constexpr binaryOperationKind opKind = extractBinaryOperationKind<Op>::value;
89 MPI_Op ret;
90 MPI_Op_create(
98 [](void* inputBuffer, void* outputBuffer, int* len, MPI_Datatype*) {
99 Type* input = (Type*)inputBuffer;
100
101 Type* output = (Type*)outputBuffer;
102
103 for (int i = 0; i < *len; i++) {
104 if constexpr (opKind == binaryOperationKind::SUM) {
105 output[i] += input[i];
106 }
107 if constexpr (opKind == binaryOperationKind::MIN) {
108 output[i] = min(output[i], input[i]);
109 }
110 if constexpr (opKind == binaryOperationKind::MAX) {
111 output[i] = max(output[i], input[i]);
112 }
113 if constexpr (opKind == binaryOperationKind::MULTIPLICATION) {
114 output[i] *= input[i];
115 }
116 }
117 },
118 1, &ret);
119 mpiOperations[pear] = ret;
120 return ret;
121 }
122 };
