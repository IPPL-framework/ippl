36 // ====================================================================
37 // Cell-centered phase convention
38 // ====================================================================
39 //
40 // Scatter/gather use cell-centered DOFs at x_j = (j + 1/2) * h.
41 // The DFT of the scattered field is:
42 //
43 // G_hat_k = phi_hat_k * exp(+i pi k / N) * conj(f_k)
44 //
45 // where the exp(+i pi k / N) arises because the kernel is evaluated
46 // at (x_p / h - j - 1/2) instead of (x_p / h - j).
47 //
48 // The shared `factor` stored in the `factors` views is
49 // factor = deconv * exp(-i pi k / N) [negative phase]
50 //
51 // Type 1 (post-FFT): f_k = conj(G_hat_k * factor)
52 // Type 2 (pre-IFFT): G_hat_k = f_k * factor
53 // The +i pi phase needed by the cell-centered gather emerges from the
54 // IFFT of the conjugate-symmetric mode field, so no explicit conj() is
55 // applied here.
56 // ====================================================================

61 template <class ExecSpace, typename RealType = double>
62 void compute_deconvolution_factors(
63 Kokkos::View<Kokkos::complex<RealType>*, typename ExecSpace::memory_space> factors,
64 int64_t n_modes, int64_t n_grid, const ESKernel<RealType>& kernel) {
65 using complex_type = Kokkos::complex<RealType>;
66
67 // Set up quadrature
68 constexpr int q = 100;
69 auto nodes = Kokkos::View<RealType*, typename ExecSpace::memory_space>("nodes", q);
70 auto weights = Kokkos::View<RealType*, typename ExecSpace::memory_space>("weights", q);
71
72 gauss_legendre<ExecSpace>(q, nodes, weights);
73
74 const RealType alpha = Kokkos::numbers::pi_v<RealType> * kernel.width() / n_grid;
75 const RealType beta = kernel.beta();
76
77 const int64_t l_n_modes = n_modes;
78 constexpr int l_q = q;
79
80 Kokkos::parallel_for(
81 "compute_deconv_factors", Kokkos::RangePolicy<ExecSpace>(0, n_modes),
82 KOKKOS_LAMBDA(const int64_t k) {
83 int freq = (k < l_n_modes / 2) ? k : k - l_n_modes;
84 RealType ft = 0.0;
85
86 for (int i = 0; i < l_q; ++i) {
87 const RealType x = nodes(i);
88 const RealType w = weights(i);
89 const RealType ker = Kokkos::exp(beta * (Kokkos::sqrt(1.0 - x * x) - 1.0));
90 ft += w * ker * Kokkos::cos(freq * alpha * x);
91 }
92
93 const RealType deconv = 2.0 / (kernel.width() * ft);
94
95 // Phase correction for cell-centered DOFs: exp(-i pi freq / N_grid)
96 // Type-1 uses this factor directly; Type-2 applies conj(factor).
97 const RealType phase = Kokkos::numbers::pi_v<RealType>
98 * static_cast<RealType>(freq)
99 / static_cast<RealType>(n_grid);
100 factors(k) =
101 complex_type(deconv * Kokkos::cos(phase), deconv * Kokkos::sin(phase));
102 });
103 }

122 template <typename FieldIn, typename FieldOut, typename ExecSpace, typename T,
123 unsigned Dim>
124 void applyDeconvolutionType1(
125 FieldIn& input,
126 const std::array<Kokkos::View<Kokkos::complex<T>*, typename ExecSpace::memory_space>,
127 Dim>& factors,
128 FieldOut& output, const Vector<size_t, Dim>& n_modes,
129 const Vector<size_t, Dim>& /*n_grid*/) {
130 using complex_type = Kokkos::complex<T>;
131 using factor_view_t = Kokkos::View<Kokkos::complex<T>*,
132 typename ExecSpace::memory_space>;
133 using index_array_type = typename ippl::RangePolicy<Dim, ExecSpace>::index_array_type;
134
135 auto input_view = input.getView();
136 auto output_view = output.getView();
137
138 const auto& layout = input.getLayout();
139 const auto& lDom = layout.getLocalNDIndex();
140
141 const int nghost_in = input.getNghost();
142 const int nghost_out = output.getNghost();
143
144 // Capture-by-value device-friendly arrays.
145 Vector<int, Dim> local_first;
146 Kokkos::Array<int64_t, Dim> begin, end;
147 Kokkos::Array<factor_view_t, Dim> f_arr;
148 Kokkos::Array<int, Dim> n_modes_arr;
149 for (unsigned d = 0; d < Dim; ++d) {
150 local_first[d] = lDom[d].first();
151 begin[d] = lDom[d].first();
152 end[d] = lDom[d].last() + 1;
153 f_arr[d] = factors[d];
154 n_modes_arr[d] = static_cast<int>(n_modes[d]);
155 }
156
157 ippl::parallel_for(
158 "deconv_type1_local",
159 ippl::createRangePolicy<Dim, ExecSpace>(begin, end),
160 KOKKOS_LAMBDA(const index_array_type& g) {
161 bool inside = true;
162 for (unsigned d = 0; d < Dim; ++d) {
163 const int gi = static_cast<int>(g[d]);
164 const int n = n_modes_arr[d];
165 if (!((gi >= 0 && gi < n / 2) || (gi >= n + n / 2 && gi < 2 * n))) {
166 inside = false;
167 break;
168 }
169 }
170
171 index_array_type idx_in, idx_out;
172 for (unsigned d = 0; d < Dim; ++d) {
173 idx_in[d] = static_cast<int>(g[d]) - local_first[d] + nghost_in;
174 idx_out[d] = static_cast<int>(g[d]) - local_first[d] + nghost_out;
175 }
176
177 if (inside) {
178 complex_type factor(T(1), T(0));
179 for (unsigned d = 0; d < Dim; ++d) {
180 const int gi = static_cast<int>(g[d]);
181 const int n = n_modes_arr[d];
182 const int rescaled = (gi < n) ? gi : gi - n;
183 factor *= f_arr[d](rescaled);
184 }
185 ippl::apply(output_view, idx_out) =
186 Kokkos::conj(ippl::apply(input_view, idx_in) * factor);
187 } else {
188 ippl::apply(output_view, idx_out) = complex_type(0, 0);
189 }
190 });
191
192 Kokkos::fence();
193 }
