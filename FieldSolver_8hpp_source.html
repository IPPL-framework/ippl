62 void setPotentialBCs() {
63 // CG requires explicit periodic boundary conditions while the periodic Poisson solver
64 // simply assumes them
65 typedef ippl::BConds<Field<T, Dim>, Dim> bc_type;
66 if ((this->getStype() == "CG") || (this->getStype() == "PCG") || (this->getStype() == "FEM") ||
67 (this->getStype() == "FEM_PRECON")) {
68 bc_type allPeriodic;
69 for (unsigned int i = 0; i < 2 * Dim; ++i) {
70 allPeriodic[i] = std::make_shared<ippl::PeriodicFace<Field<T, Dim>>>(i);
71 }
72 phi_m->setFieldBC(allPeriodic);
73 if ((this->getStype() == "FEM") || (this->getStype() == "FEM_PRECON")) {
74 rho_m->setFieldBC(allPeriodic);

79 void runSolver() override {
80 if ((this->getStype() == "CG") || (this->getStype() == "PCG") || (this->getStype() == "FEM") ||
81 (this->getStype() == "FEM_PRECON")) {
82 int iterations = 0;
83 int residue = 0;
84
85 if (this->getStype() == "FEM") {
86 FEMSolver_t<T, Dim>& solver = std::get<FEMSolver_t<T, Dim>>(this->getSolver());
87 solver.solve();
88
89 iterations = solver.getIterationCount();
90 residue = solver.getResidue();
91 } else if (this->getStype() == "FEM_PRECON") {
92 FEMPreconSolver_t<T, Dim>& solver = std::get<FEMPreconSolver_t<T, Dim>>(this->getSolver());
93 solver.solve();
94
95 iterations = solver.getIterationCount();
96 residue = solver.getResidue();
97 } else {
98 CGSolver_t<T, Dim>& solver = std::get<CGSolver_t<T, Dim>>(this->getSolver());
99 solver.solve();
100
101 iterations = solver.getIterationCount();
102 residue = solver.getResidue();
103 }
104
105 if (ippl::Comm->rank() == 0) {
106 std::filesystem::create_directory("data_CG");
107 std::stringstream fname;
108 if ((this->getStype() == "CG") || (this->getStype() == "FEM") ||
109 (this->getStype() == "FEM_PRECON")) {
110 fname << "data_CG/CG_";
111 } else {
112 fname << "data_CG/";
113 fname << preconditioner_params_m[0];
114 fname << "_";
115 }
116 fname << ippl::Comm->size();
117 fname << ".csv";
118
119 Inform log(NULL, fname.str().c_str(), Inform::APPEND);
120 // Assume the dummy solve is the first call
121 if (iterations == 0) {
122 log << "residue,iterations" << endl;
123 }
124 // Don't print the dummy solve
125 if (iterations > 0) {
126 log << residue << "," << iterations << endl;
127 }
128 }
129 ippl::Comm->barrier();
130 } else if (this->getStype() == "FFT") {
131 if constexpr (Dim == 2 || Dim == 3) {
132 std::get<FFTSolver_t<T, Dim>>(this->getSolver()).solve();
133 }
134 } else if (this->getStype() == "TG") {
135 if constexpr (Dim == 3) {
136 std::get<FFTTruncatedGreenSolver_t<T, Dim>>(this->getSolver()).solve();
137 }
138 } else if (this->getStype() == "OPEN") {
139 if constexpr (Dim == 3) {
140 std::get<OpenSolver_t<T, Dim>>(this->getSolver()).solve();
141 }
142 } else {
143 throw std::runtime_error("Unknown solver type");
144 }

147 template <typename Solver>
148 void initSolverWithParams(const ippl::ParameterList& sp) {
149 this->getSolver().template emplace<Solver>();
150 Solver& solver = std::get<Solver>(this->getSolver());
151
152 solver.mergeParameters(sp);
153
154 solver.setRhs(*rho_m);
155
156 if constexpr ((std::is_same_v<Solver, CGSolver_t<T, Dim>>) ||
157 (std::is_same_v<Solver, FEMSolver_t<T, Dim>>) ||
158 (std::is_same_v<Solver, FEMPreconSolver_t<T, Dim>>)) {
159 // The CG solver and FEMPoissonSolver compute the potential
160 // directly and use this to get the electric field
161 solver.setLhs(*phi_m);
162 solver.setGradient(*E_m);
163 } else {
164 // The periodic Poisson solver, Open boundaries solver,
165 // and the TG solver compute the electric field directly
166 solver.setLhs(*E_m);
167 }
168 }
