40 // Parse the preconditioner params (PCG and FEMPrecon solvers);
41 // [type, optional numeric args...]
42 // Type validation and numeric sanitization are delegated to the shared
43 // preconditioner validation helper to avoid duplication with Poisson solvers.
44 // If the type is known but the numeric args are invalid or the number
45 // of arguments is wrong, a warning is printed and default parameters
46 // are used.
47 ParsedPreconditionerParams parsePreconditionerParams(const std::string& solver_name) const {
48 Inform warn("FieldSolver");
49 ParsedPreconditionerParams parsed;
50 if (preconditioner_params_m.empty()) {
51 warn << "No preconditioner type provided for solver " << solver_name
52 << "; using default identity preconditioner." << endl;
53 return parsed;
54 }
55
56 parsed.type = preconditioner_params_m[0];
57 ippl::preconditioner_validation::throwIfUnknownType(
58 parsed.type, "FieldSolver::parsePreconditionerParams");
59
60 size_t expected_arg_count = 0;
61 if (parsed.type == "newton" || parsed.type == "chebyshev"
62 || parsed.type == "richardson_alt") {
63 expected_arg_count = 1;
64 } else if (parsed.type == "richardson") {
65 expected_arg_count = 2;
66 } else if (parsed.type == "gauss-seidel" || parsed.type == "ssor") {
67 expected_arg_count = 3;
68 } else if (parsed.type == "multigrid") {
69 expected_arg_count = 4;
70 }
71
72 const size_t provided_arg_count = preconditioner_params_m.size() - 1;
73 if (provided_arg_count == 0) {
74 return parsed;
75 }
76
77 auto warnAndUseDefaults = [&](const std::string& detail) {
78 warn << "Invalid parameters for preconditioner '" << parsed.type << "' in solver "
79 << solver_name << " (" << detail << "). Using default preconditioner parameters."
80 << endl;
81 parsed.use_defaults = true;
82 };
83
84 if (provided_arg_count != expected_arg_count) {
85 warnAndUseDefaults("expected " + std::to_string(expected_arg_count)
86 + " argument(s), got " + std::to_string(provided_arg_count));
87 return parsed;
88 }
89
90 try {
91 int arg = 1;
92 if (parsed.type == "newton") {
93 parsed.newton_level = std::stoi(preconditioner_params_m[arg++]);
94 } else if (parsed.type == "chebyshev") {
95 parsed.chebyshev_degree = std::stoi(preconditioner_params_m[arg++]);
96 } else if (parsed.type == "richardson" || parsed.type == "richardson_alt") {
97 parsed.richardson_iterations = std::stoi(preconditioner_params_m[arg++]);
98 if (parsed.type == "richardson") {
99 parsed.communication = std::stoi(preconditioner_params_m[arg++]);
100 }
101 } else if (parsed.type == "gauss-seidel") {
102 parsed.gauss_seidel_inner_iterations = std::stoi(preconditioner_params_m[arg++]);
103 parsed.gauss_seidel_outer_iterations = std::stoi(preconditioner_params_m[arg++]);
104 parsed.communication = std::stoi(preconditioner_params_m[arg++]);
105 } else if (parsed.type == "ssor") {
106 parsed.gauss_seidel_inner_iterations = std::stoi(preconditioner_params_m[arg++]);
107 parsed.gauss_seidel_outer_iterations = std::stoi(preconditioner_params_m[arg++]);
108 parsed.ssor_omega = std::stod(preconditioner_params_m[arg++]);
109 } else if (parsed.type == "multigrid") {
110 parsed.mg_pre_smooth_iters = std::stoi(preconditioner_params_m[arg++]);
111 parsed.mg_post_smooth_iters = std::stoi(preconditioner_params_m[arg++]);
112 parsed.mg_omega = std::stod(preconditioner_params_m[arg++]);
113 parsed.mg_min_cells =
114 static_cast<unsigned>(std::stoul(preconditioner_params_m[arg++]));
115 }
116 parsed.use_defaults = false;
117 } catch (const std::exception& ex) {
118 warnAndUseDefaults(std::string("failed to parse numeric values: ") + ex.what());
119 }
120
121 ippl::preconditioner_validation::sanitizeParams(
122 parsed.type, warn, parsed.newton_level, parsed.chebyshev_degree,
123 parsed.richardson_iterations, parsed.gauss_seidel_inner_iterations,
124 parsed.gauss_seidel_outer_iterations, parsed.ssor_omega, &parsed.communication,
125 parsed.mg_pre_smooth_iters, parsed.mg_post_smooth_iters, parsed.mg_omega,
126 parsed.mg_min_cells);
127
128 return parsed;
129 }

175 // CG requires explicit periodic boundary conditions while the periodic Poisson solver
176 // simply assumes them
177 typedef ippl::BConds<Field<T, Dim>, Dim> bc_type;

191 void runSolver() override {
192 if ((this->getStype() == "CG") || (this->getStype() == "PCG") || (this->getStype() == "FEM")
193 || (this->getStype() == "FEM_PRECON")) {
194 int iterations = 0;
195 int residue = 0;
196
197 if (this->getStype() == "FEM") {
198 FEMSolver_t<T, Dim>& solver = std::get<FEMSolver_t<T, Dim>>(this->getSolver());
199 solver.solve();
200
201 iterations = solver.getIterationCount();
202 residue = solver.getResidue();
203 } else if (this->getStype() == "FEM_PRECON") {
204 FEMPreconSolver_t<T, Dim>& solver =
205 std::get<FEMPreconSolver_t<T, Dim>>(this->getSolver());
206 solver.solve();
207
208 iterations = solver.getIterationCount();
209 residue = solver.getResidue();
210 } else {
211 CGSolver_t<T, Dim>& solver = std::get<CGSolver_t<T, Dim>>(this->getSolver());
212 solver.solve();
213
214 iterations = solver.getIterationCount();
215 residue = solver.getResidue();
216 }
217
218 if (ippl::Comm->rank() == 0) {
219 std::filesystem::create_directory("data_CG");
220 std::stringstream fname;
221 if ((this->getStype() == "CG") || (this->getStype() == "FEM")
222 || (this->getStype() == "FEM_PRECON")) {
223 fname << "data_CG/CG_";
224 } else {
225 fname << "data_CG/";
226 fname << preconditioner_params_m[0];
227 fname << "_";
228 }
229 fname << ippl::Comm->size();
230 fname << ".csv";
231
232 Inform log(NULL, fname.str().c_str(), Inform::APPEND);
233 // Assume the dummy solve is the first call
234 if (iterations == 0) {
235 log << "residue,iterations" << endl;
236 }
237 // Don't print the dummy solve
238 if (iterations > 0) {
239 log << residue << "," << iterations << endl;
240 }
241 }
242 ippl::Comm->barrier();
243 } else if (this->getStype() == "FFT") {
244 if constexpr (Dim == 2 || Dim == 3) {
245 std::get<FFTSolver_t<T, Dim>>(this->getSolver()).solve();
246 }
247 } else if (this->getStype() == "TG") {
248 if constexpr (Dim == 3) {
249 std::get<FFTTruncatedGreenSolver_t<T, Dim>>(this->getSolver()).solve();

260 template <typename Solver>
261 void initSolverWithParams(const ippl::ParameterList& sp) {
262 this->getSolver().template emplace<Solver>();
263 Solver& solver = std::get<Solver>(this->getSolver());
264
265 solver.mergeParameters(sp);
266
267 solver.setRhs(*rho_m);
268
269 if constexpr ((std::is_same_v<Solver, CGSolver_t<T, Dim>>)
270 || (std::is_same_v<Solver, FEMSolver_t<T, Dim>>)
271 || (std::is_same_v<Solver, FEMPreconSolver_t<T, Dim>>)) {
272 // The CG solver and FEMPoissonSolver compute the potential
273 // directly and use this to get the electric field
274 solver.setLhs(*phi_m);
275 solver.setGradient(*E_m);
276 } else {
277 // The periodic Poisson solver, Open boundaries solver,
278 // and the TG solver compute the electric field directly
279 solver.setLhs(*E_m);
280 }
