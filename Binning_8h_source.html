59 KOKKOS_INLINE_FUNCTION int compute_tile_1d(RealType val, int dim) const {
60 const RealType grid_pos = transform.toGridCoordinate(val, dim);
61 const int center =
62 transform.getStencilCenter(grid_pos - RealType(0.5), kernel_width);
63 const int local_c = center - local_offset[dim];
64 return Kokkos::clamp(local_c / tile_size[dim], 0, num_tiles[dim] - 1);
65 }

73 template <typename PositionType>
74 KOKKOS_INLINE_FUNCTION int operator()(const PositionType& pos) const {
75 int bin_idx = 0;
76 int stride = 1;
77
78 for (int d = Dim - 1; d >= 0; --d) {
79 const int tile_d = compute_tile_1d(pos[d], d);
80 bin_idx += tile_d * stride;
81 stride *= num_tiles[d];
82 }
83
84 return bin_idx;
85 }

119 template <unsigned Dim, typename RealType, typename PositionViewType,
120 typename ExecSpace, typename PermuteViewType, typename OffsetViewType,
121 typename KeyViewType, typename CursorViewType>
122 void bin_sort(PositionViewType positions, Vector<int, Dim> n_grid_global,
123 [[maybe_unused]] Vector<int, Dim> n_grid_local,
124 Vector<int, Dim> local_offset, Vector<int, Dim> tile_size,
125 int kernel_width, Vector<RealType, Dim> origin,
126 Vector<RealType, Dim> invdx, PermuteViewType& permute,
127 OffsetViewType& bin_offsets, KeyViewType& bin_keys,
128 CursorViewType& cursor, size_t n_particles,
129 Vector<int, Dim> num_tiles) {
130 using key_type = typename KeyViewType::non_const_value_type;
131
132 static IpplTimings::TimerRef binSortTimer = IpplTimings::getTimer("binSort");
133 IpplTimings::startTimer(binSortTimer);
134
135 // Total number of bins
136 size_t n_bins = 1;
137 for (unsigned d = 0; d < Dim; ++d) {
138 n_bins *= num_tiles[d];
139 }
140
141 CoordinateTransform<RealType, Dim> transform(origin, invdx, n_grid_global);
142 BinComputer<Dim, RealType> bin_computer{n_grid_global, local_offset, tile_size,
143 num_tiles, kernel_width, transform};
144
145 // Pass A: bin index per particle + atomic histogram into bin_offsets.
146 static IpplTimings::TimerRef keyTimer = IpplTimings::getTimer("binComputeKeys");
147 IpplTimings::startTimer(keyTimer);
148
149 {
150 auto offsets_zero = Kokkos::subview(
151 bin_offsets, std::make_pair(size_t(0), n_bins + 1));
152 Kokkos::deep_copy(ExecSpace(), offsets_zero,
153 typename OffsetViewType::value_type(0));
154 }
155
156 if (n_particles > 0) {
157 Kokkos::parallel_for(
158 "BinSort::HistogramAndKeys",
159 Kokkos::RangePolicy<ExecSpace>(0, n_particles),
160 KOKKOS_LAMBDA(const size_t i) {
161 const key_type k =
162 static_cast<key_type>(bin_computer(positions(i)));
163 bin_keys(i) = k;
164 Kokkos::atomic_inc(&bin_offsets(static_cast<size_t>(k)));
165 });
166 Kokkos::fence();
167 }
168 IpplTimings::stopTimer(keyTimer);
169
170 // Pass B: exclusive scan turns the histogram into per-bin start
171 // offsets. Kokkos dispatches this to cub::DeviceScan on CUDA
172 // and rocPRIM on HIP, so we get the vendor-tuned scan for free.
173 static IpplTimings::TimerRef scanTimer = IpplTimings::getTimer("binSortByKey");
174 IpplTimings::startTimer(scanTimer);
175
176 using offset_value_type = typename OffsetViewType::value_type;
177 Kokkos::parallel_scan(
178 "BinSort::ExclusiveScan",
179 Kokkos::RangePolicy<ExecSpace>(0, n_bins + 1),
180 KOKKOS_LAMBDA(const size_t i, offset_value_type& upd, const bool final) {
181 const offset_value_type cnt = bin_offsets(i);
182 if (final) {
183 bin_offsets(i) = upd;
184 }
185 upd += cnt;
186 });
187 Kokkos::fence();
188 IpplTimings::stopTimer(scanTimer);
189
190 // Pass C: scatter particle ids into permute, using cursor as a
191 // per-bin atomic write head that starts at the bin's offset.
192 static IpplTimings::TimerRef offsetTimer =
193 IpplTimings::getTimer("binComputeOffsets");
194 IpplTimings::startTimer(offsetTimer);
195
196 if (n_particles > 0) {
197 auto cursor_sub = Kokkos::subview(cursor,
198 std::make_pair(size_t(0), n_bins));
199 auto offsets_sub = Kokkos::subview(bin_offsets,
200 std::make_pair(size_t(0), n_bins));
201 Kokkos::deep_copy(ExecSpace(), cursor_sub, offsets_sub);
202
203 Kokkos::parallel_for(
204 "BinSort::Scatter",
205 Kokkos::RangePolicy<ExecSpace>(0, n_particles),
206 KOKKOS_LAMBDA(const size_t i) {
207 const size_t k = static_cast<size_t>(bin_keys(i));
208 const size_t pos =
209 Kokkos::atomic_fetch_add(&cursor(k), size_t(1));
210 permute(pos) = i;
211 });
212 Kokkos::fence();
213 }
214 IpplTimings::stopTimer(offsetTimer);
215
216 IpplTimings::stopTimer(binSortTimer);
217 }

238 template <typename ParticleT, typename FieldT, class... ParticleProperties,
239 unsigned Dim>
240 auto bin_particles(
241 const ParticleAttrib<Vector<ParticleT, Dim>, ParticleProperties...>& particles,
242 FieldLayout<Dim> fieldLayout, UniformCartesian<FieldT, Dim> mesh,
243 Vector<int, Dim> tile_size, int kernel_width) {
244 using AttribType = std::decay_t<decltype(particles)>;
245 using ExecSpace = typename AttribType::execution_space;
246 using memory_space = typename AttribType::memory_space;
247
248 // Extract grid information
249 const NDIndex<Dim>& lDom = fieldLayout.getLocalNDIndex();
250 const NDIndex<Dim>& gDom = fieldLayout.getDomain();
251
252 Vector<int, Dim> ngrid_global;
253 Vector<int, Dim> ngrid_local;
254 Vector<int, Dim> local_offset;
255 for (unsigned d = 0; d < Dim; ++d) {
256 ngrid_global[d] = gDom[d].length();
257 ngrid_local[d] = lDom[d].length();
258 local_offset[d] = lDom[d].first();
259 }
260
261 // Compute number of tiles (+1 for boundary particles).
262 Vector<int, Dim> num_tiles;
263 size_t total_tiles = 1;
264 for (unsigned d = 0; d < Dim; ++d) {
265 num_tiles[d] = (ngrid_local[d] + tile_size[d] - 1) / tile_size[d] + 1;
266 total_tiles *= num_tiles[d];
267 }
268
269 auto particle_view = particles.getView();
270 const auto invdx = 1.0 / mesh.getMeshSpacing();
271 const size_t n_particles = particles.getParticleCount();
272
273 auto& bufs = ippl::detail::getDefaultBinSortBuffers<memory_space>();
274 // n_bins + 1 slots needed for bin_offsets and cursor
275 bufs.ensureCapacity(n_particles, total_tiles + 1);
276
277 auto& permute = bufs.permute();
278 auto& bin_offsets = bufs.binOffsets();
279 auto& bin_keys = bufs.binKeys();
280 auto& cursor = bufs.cursor();
281
282 bin_sort<Dim, ParticleT, std::decay_t<decltype(particle_view)>, ExecSpace>(
283 particle_view, ngrid_global, ngrid_local, local_offset, tile_size,
284 kernel_width, mesh.getOrigin(), invdx, permute, bin_offsets, bin_keys,
285 cursor, n_particles, num_tiles);
286
287 return std::make_tuple(permute, bin_offsets, num_tiles);
288 }
