74 virtual void operator()(lhs_type& lhs, rhs_type& rhs,
75 const ParameterList& params) override {
76 constexpr unsigned Dim = lhs_type::dim;
77 typename lhs_type::Mesh_t mesh = lhs.get_mesh();
78 typename lhs_type::Layout_t layout = lhs.getLayout();
79
80 iterations_m = 0;
81 const int maxIterations = params.get<int>("max_iterations");
82
83 // Variable names mostly based on description in
84 // https://www.cs.cmu.edu/~quake-papers/painless-conjugate-gradient.pdf
85 lhs_type r(mesh, layout);
86 lhs_type d(mesh, layout);
87
88 using bc_type = BConds<lhs_type, Dim>;
89 bc_type lhsBCs = lhs.getFieldBC();
90 bc_type bc;
91
92 bool allFacesPeriodic = true;
93 for (unsigned int i = 0; i < 2 * Dim; ++i) {
94 FieldBC bcType = lhsBCs[i]->getBCType();
95 if (bcType == PERIODIC_FACE) {
96 // If the LHS has periodic BCs, so does the residue
97 bc[i] = std::make_shared<PeriodicFace<lhs_type>>(i);
98 } else if (bcType & CONSTANT_FACE) {
99 // If the LHS has constant BCs, the residue is zero on the BCs
100 // Bitwise AND with CONSTANT_FACE will succeed for ZeroFace or ConstantFace
101 bc[i] = std::make_shared<ZeroFace<lhs_type>>(i);
102 allFacesPeriodic = false;
103 } else {
104 throw IpplException("PCG::operator()",
105 "Only periodic or constant BCs for LHS supported.");
106 return;
107 }
108 }
109
110 r = rhs - op_m(lhs);
111 d = r.deepCopy();
112 d.setFieldBC(bc);
113
114 T delta1 = innerProduct(r, d);
115 T delta0 = delta1;
116 residueNorm = std::sqrt(delta1);
117 const T tolerance = params.get<T>("tolerance") * norm(rhs);
118
119 lhs_type q(mesh, layout);
120
121 while (iterations_m < maxIterations && residueNorm > tolerance) {
122 q = op_m(d);
123
124 T alpha = delta1 / innerProduct(d, q);
125 lhs = lhs + alpha * d;
126
127 // The exact residue is given by
128 // r = rhs - op_m(lhs);
129 // This correction is generally not used in practice because
130 // applying the Laplacian is computationally expensive and
131 // the correction does not have a significant effect on accuracy;
132 // in some implementations, the correction may be applied every few
133 // iterations to offset accumulated floating point errors
134 r = r - alpha * q;
135 delta0 = delta1;
136 delta1 = innerProduct(r, r);
137 T beta = delta1 / delta0;
138
139 residueNorm = std::sqrt(delta1);
140 d = r + beta * d;
141 ++iterations_m;
142 }
143
144 if (allFacesPeriodic) {
145 T avg = lhs.getVolumeAverage();
146 lhs = lhs - avg;
147 }
148 }

154 T residueNorm = 0;

228 preconditioner_m =

231 std::move(upper_and_lower), std::move(inverse_diagonal),

233 } else if (preconditioner_type == "gauss-seidel") {
234 preconditioner_m = std::move(

236 std::move(lower), std::move(upper), std::move(inverse_diagonal), inner,

238 } else if (preconditioner_type == "ssor") {

244 } else {
245 preconditioner_m = std::move(std::make_unique<preconditioner<FieldLHS>>());
