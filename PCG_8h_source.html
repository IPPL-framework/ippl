62 virtual void operator()(lhs_type& lhs, rhs_type& rhs,
63 const ParameterList& params) override {
64 constexpr unsigned Dim = lhs_type::dim;
65 typename lhs_type::Mesh_t mesh = lhs.get_mesh();
66 typename lhs_type::Layout_t layout = lhs.getLayout();
67
68 iterations_m = 0;
69 const int maxIterations = params.get<int>("max_iterations");
70
71 // Variable names mostly based on description in
72 // https://www.cs.cmu.edu/~quake-papers/painless-conjugate-gradient.pdf
73 lhs_type r(mesh, layout);
74 lhs_type d(mesh, layout);
75
76 using bc_type = BConds<lhs_type, Dim>;
77 bc_type lhsBCs = lhs.getFieldBC();
78 bc_type bc;
79
80 bool allFacesPeriodic = true;
81 for (unsigned int i = 0; i < 2 * Dim; ++i) {
82 FieldBC bcType = lhsBCs[i]->getBCType();
83 if (bcType == PERIODIC_FACE) {
84 // If the LHS has periodic BCs, so does the residue
85 bc[i] = std::make_shared<PeriodicFace<lhs_type>>(i);
86 } else if (bcType & CONSTANT_FACE) {
87 // If the LHS has constant BCs, the residue is zero on the BCs
88 // Bitwise AND with CONSTANT_FACE will succeed for ZeroFace or ConstantFace
89 bc[i] = std::make_shared<ZeroFace<lhs_type>>(i);
90 allFacesPeriodic = false;
91 } else {
92 throw IpplException("PCG::operator()",
93 "Only periodic or constant BCs for LHS supported.");
94 return;
95 }
96 }
97
98 r = rhs - op_m(lhs);
99 d = r.deepCopy();
100 d.setFieldBC(bc);
101
102 T delta1 = innerProduct(r, d);
103 T delta0 = delta1;
104 residueNorm = std::sqrt(delta1);
105 const T tolerance = params.get<T>("tolerance") * norm(rhs);
106
107 lhs_type q(mesh, layout);
108
109 while (iterations_m < maxIterations && residueNorm > tolerance) {
110 q = op_m(d);
111 T alpha = delta1 / innerProduct(d, q);
112 lhs = lhs + alpha * d;
113
114 // The exact residue is given by
115 // r = rhs - op_m(lhs);
116 // This correction is generally not used in practice because
117 // applying the Laplacian is computationally expensive and
118 // the correction does not have a significant effect on accuracy;
119 // in some implementations, the correction may be applied every few
120 // iterations to offset accumulated floating point errors
121 r = r - alpha * q;
122 delta0 = delta1;
123 delta1 = innerProduct(r, r);
124 T beta = delta1 / delta0;
125
126 residueNorm = std::sqrt(delta1);
127 d = r + beta * d;
128 ++iterations_m;
129 }
130
131 if (allFacesPeriodic) {
132 T avg = lhs.getVolumeAverage();
133 lhs = lhs - avg;
134 }
135 }
